28  package org.treetank.axis;
29  
30  import java.util.Stack;
31  
32  import org.treetank.api.INodeReadTrx;
33  import org.treetank.node.IConstants;
34  import org.treetank.node.interfaces.IStructNode;
35  
36  /**
37   * <h1>PrecedingAxis</h1>
38   * 
39   * <p>
40   * Iterate over all preceding nodes of kind ELEMENT or TEXT starting at a given node. Self is not included.
41   * </p>
42   */
43  public class PrecedingAxis extends AbsAxis {
44  
45      private boolean mIsFirst;
46  
47      private Stack<Long> mStack;
48  
49      /**
50       * Constructor initializing internal state.
51       * 
52       * @param rtx
53       *            Exclusive (immutable) trx to iterate with.
54       */
55      public PrecedingAxis(final INodeReadTrx rtx) {
56  
57          super(rtx);
58          mIsFirst = true;
59          mStack = new Stack<Long>();
60  
61      }
62  
63      /**
64       * {@inheritDoc}
65       */
66      @Override
67      public final void reset(final long mNodeKey) {
68  
69          super.reset(mNodeKey);
70          mIsFirst = true;
71          mStack = new Stack<Long>();
72  
73      }
74  
75      /**
76       * {@inheritDoc}
77       */
78      @Override
79      public final boolean hasNext() {
80  
81          // assure, that preceding is not evaluated on an attribute or a
82          // namespace
83          if (mIsFirst) {
84              mIsFirst = false;
85              if (getNode().getKind() == IConstants.ATTRIBUTE
86              // || getTransaction().isNamespaceKind()
87              ) {
88                  resetToStartKey();
89                  return false;
90              }
91          }
92  
93          resetToLastKey();
94  
95          if (!mStack.empty()) {
96              // return all nodes of the current subtree in reverse document order
97              moveTo(mStack.pop());
98              return true;
99          }
100 
101         if (((IStructNode)getNode()).hasLeftSibling()) {
102             moveTo(((IStructNode)getNode()).getLeftSiblingKey());
103             // because this axis return the precedings in reverse document
104             // order, we
105             // need to travel to the node in the subtree, that comes last in
106             // document
107             // order.
108             getLastChild();
109             return true;
110         }
111 
112         while (getNode().hasParent()) {
113             // ancestors are not part of the preceding set
114             moveTo(getNode().getParentKey());
115             if (((IStructNode)getNode()).hasLeftSibling()) {
116                 moveTo(((IStructNode)getNode()).getLeftSiblingKey());
117                 // move to last node in the subtree
118                 getLastChild();
119                 return true;
120             }
121         }
122 
123         resetToStartKey();
124         return false;
125 
126     }
127 
128     /**
129      * Moves the transaction to the node in the current subtree, that is last in
130      * document order and pushes all other node key on a stack. At the end the
131      * stack contains all node keys except for the last one in reverse document
132      * order.
133      */
134     private void getLastChild() {
135 
136         // nodekey of the root of the current subtree
137         final long parent = getNode().getDataKey();
138 
139         // traverse tree in pre order to the leftmost leaf of the subtree and
140         // push
141         // all nodes to the stack
142         if (((IStructNode)getNode()).hasFirstChild()) {
143             while (((IStructNode)getNode()).hasFirstChild()) {
144                 mStack.push(getNode().getDataKey());
145                 moveTo(((IStructNode)getNode()).getFirstChildKey());
146             }
147 
148             // traverse all the siblings of the leftmost leave and all their
149             // descendants and push all of them to the stack
150             while (((IStructNode)getNode()).hasRightSibling()) {
151                 mStack.push(getNode().getDataKey());
152                 moveTo(((IStructNode)getNode()).getRightSiblingKey());
153                 getLastChild();
154             }
155 
156             // step up the path till the root of the current subtree and process
157             // all
158             // right siblings and their descendants on each step
159             if (getNode().hasParent() && (getNode().getParentKey() != parent)) {
160 
161                 mStack.push(getNode().getDataKey());
162                 while (getNode().hasParent() && (getNode().getParentKey() != parent)) {
163 
164                     moveTo(getNode().getParentKey());
165 
166                     // traverse all the siblings of the leftmost leave and all
167                     // their
168                     // descendants and push all of them to the stack
169                     while (((IStructNode)getNode()).hasRightSibling()) {
170 
171                         moveTo(((IStructNode)getNode()).getRightSiblingKey());
172                         getLastChild();
173                         mStack.push(getNode().getDataKey());
174                     }
175                 }
176 
177                 // set transaction to the node in the subtree that is last in
178                 // document
179                 // order
180                 moveTo(mStack.pop());
181             }
182         }
183     }
