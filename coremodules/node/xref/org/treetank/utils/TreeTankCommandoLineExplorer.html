73      /**
74       * Finding file for a given command.z.
75       * 
76       * @param mCommandLine
77       *            the line to be analysed
78       * @return the corresponding file
79       */
80      private static File findFile(final String mCommandLine) {
81          final String[] command = mCommandLine.split(COMMANDDELIM);
82          if (command.length != 2) {
83              return null;
84          }
85          return new File(command[1]);
86      }

88      /**
89       * Main for all the knowledge and input.
90       * 
91       * @param args
92       *            only one arg allowed, the tnk file
93       * @throws Exception
94       *             of any kind
95       */
96      public static void main(final String[] args) throws Exception {
97          IStorage storage = null;
98          ISession session = null;
99          INodeReadTrx nRtx = null;
100         IPageReadTrx pRtx = null;
101         if (args.length > 0) {
102 
103             long revision = 0;
104             if (args.length > 1) {
105                 revision = Long.parseLong(args[1]);
106             }
107 
108             final File file = new File(args[0]);
109             final StorageConfiguration config = new StorageConfiguration(file);
110             Storage.createStorage(config);
111             storage = Storage.openStorage(file);
112             session = storage.getSession(new SessionConfiguration("TMP", null));
113             if (revision != 0) {
114                 pRtx = session.beginPageReadTransaction(session.getMostRecentVersion());
115             } else {
116                 pRtx = session.beginPageReadTransaction(revision);
117             }
118             nRtx = new NodeReadTrx(pRtx);
119         } else {
120             System.out.println("Usage: java TreeTankCommandoLineExplorer \"tnk-file\" [revision] "
121                 + "(if revision not given, explorer works in write mode");
122             System.exit(-1);
123         }
124 
125         try {
126             System.out.println("Welcome to TTCommand");
127             System.out.println("Type in \"help\" for help for usage..");
128 
129             String line = null;
130             final BufferedReader bufferIn = new BufferedReader(new InputStreamReader(System.in));
131             System.out.print(">");
132             while ((line = bufferIn.readLine()) != null) {
133 
134                 final Command command = Command.toCommand(line);
135                 switch (command) {
136                 case LOGIN:
137                     if (nRtx != null) {
138                         nRtx.close();
139                     }
140                     if (session != null) {
141                         session.close();
142                     }
143                     final File file = findFile(line);
144                     if (file != null) {
145                         storage = Storage.openStorage(file);
146                         session = storage.getSession(new SessionConfiguration("TMP", null));
147                         pRtx = session.beginPageReadTransaction(session.getMostRecentVersion());
148                         nRtx = new NodeReadTrx(pRtx);
149                         System.out.println(command.executeCommand(nRtx));
150                     } else {
151                         System.out.println("Invalid path to tt-file! Please use other!");
152                     }
153                     break;
154                 case LOGOUT:
155                     System.out.println(command.executeCommand(nRtx));
156                     if (nRtx != null) {
157                         nRtx.close();
158                     }
159                     if (session != null) {
160                         session.close();
161                     }
162                     break;
163                 case EXIT:
164                     System.out.println(command.executeCommand(nRtx));
165                     if (nRtx != null) {
166                         nRtx.close();
167                     }
168                     if (session != null) {
169                         session.close();
170                     }
171                     System.exit(1);
172                     break;
173                 default:
174                     if (session == null || nRtx == null) {
175                         System.out.println(new StringBuilder("No database loaded!, Please use ").append(
176                             Command.LOGIN.mCommand).append(" to load tt-database").toString());
177                     } else {
178                         System.out.println(command.executeCommand(nRtx));
179                     }
180                     System.out.print(">");
181                 }
182             }
183 
184         } catch (final Exception e) {
185             if (nRtx != null) {
186                 nRtx.close();
187             }
188             if (session != null) {
189                 session.close();
190             }
191             System.exit(-1);
192         }

202     private enum Command {
203         HELP("help") {
204             @Override
205             String executeCommand(final INodeReadTrx mCurrentRtx, final String mParameter) {
206                 final StringBuilder builder = new StringBuilder("Help for ");
207                 if (mParameter.equals(INFO.mCommand)) {
208                     builder.append("info:\n");
209                     builder.append("prints out nodeKey, child count, parent key, ").append(
210                         "first child key, left sibling key, right sibling key\n");
211                 } else if (mParameter.equals(CONTENT.mCommand)) {
212                     builder.append("content:\n");
213                     builder.append("prints out kind of node plus relevant content\n");
214                 } else if (mParameter.equals(LOGOUT.mCommand)) {
215                     builder.append("logout:\n");
216                     builder.append("Logout from database\n");
217                 } else if (mParameter.equals(LOGIN.mCommand)) {
218                     builder.append("login:\n");
219                     builder.append("Parameter is the path to the tt-database").append("\"login:[path]\"\n");
220                 } else if (mParameter.equals(EXIT.mCommand)) {
221                     builder.append("exit:\n");
222                     builder.append("Exits the program\n");
223                 } else if (mParameter.equals(MOVE.mCommand)) {
224                     builder.append("move:\n");
225                     builder.append("Below a concrete parameter list\n");
226                     builder.append("up\t\t:\tGo to father if possible\n");
227                     builder.append("down\t\t:\tGo to first child if possible\n");
228                     builder.append("left\t\t:\tGo to left sibling if possible\n");
229                     builder.append("right\t\t:\tGo to right sibling if possible\n");
230                     builder.append("root\t\t:\tGo to document root if possible\n");
231                     builder.append("[nodekey]\t:\tGo to specified node key if possible, ").append(
232                         "[nodekey] has to be a long\n");
233                 } else {
234                     builder.append("common usage\n Usage: [COMMAND]:[PARAMETER]\n");
235                     builder.append("For concrete parameter-list, type ").append(HELP.mCommand).append(
236                         COMMANDDELIM).append("[COMMAND]\n");
237                     builder.append("Below a list of all commands:\n");
238                     builder.append(LOGIN.mCommand).append("\t:\t").append("Login into database.\n");
239                     builder.append(LOGOUT.mCommand).append("\t:\t").append("Logout from database.\n");
240                     builder.append(EXIT.mCommand).append("\t:\t").append("Exits the programm.\n");
241                     builder.append(INFO.mCommand).append("\t:\t").append(
242                         "Offers info about the current node.\n");
243                     builder.append(MOVE.mCommand).append("\t:\t").append("Moving to given node.\n");
244                 }
245                 return builder.toString();
246             }
247         },
248         CONTENT("content") {
249             @Override
250             String executeCommand(final INodeReadTrx mCurrentRtx, final String mParameter) {
251                 final StringBuilder builder = new StringBuilder("Kind: ");
252                 switch (mCurrentRtx.getNode().getKind()) {
253                 case ELEMENT:
254                     builder.append("Element\n");
255                     builder.append(mCurrentRtx.getQNameOfCurrentNode());
256                     break;
257                 case ATTRIBUTE:
258                     builder.append("Attribute\n");
259                     builder.append(mCurrentRtx.getQNameOfCurrentNode());
260                     builder.append("=");
261                     builder.append(mCurrentRtx.getValueOfCurrentNode());
262                     break;
263                 case TEXT:
264                     builder.append("Text\n");
265                     builder.append(mCurrentRtx.getValueOfCurrentNode());
266                     break;
267                 case NAMESPACE:
268                     builder.append("Namespace\n");
269                     builder.append(mCurrentRtx.getQNameOfCurrentNode());
270                     break;
271                 case PROCESSING:
272                     builder.append("Processing instruction\n");
273                     break;
274                 case COMMENT:
275                     builder.append("Comment\n");
276                     break;
277                 case ROOT:
278                     builder.append("Document Root\n");
279                     break;
280                 default:
281                     builder.append("unknown!");
282                 }
283                 return builder.toString();
284             }
285         },
286         INFO("info") {
287             @Override
288             String executeCommand(final INodeReadTrx mCurrentRtx, final String mParameter) {
289                 final StringBuilder builder = new StringBuilder();
290                 builder.append(mCurrentRtx.toString());
291                 return builder.toString();
292             }
293         },
294         LOGIN("login") {
295             @Override
296             String executeCommand(final INodeReadTrx mCurrentRtx, final String mParameter) {
297                 return new StringBuilder("Loggin into database ").append(mParameter).append("\n").toString();
298             }
299         },
300         LOGOUT("logout") {
301             @Override
302             String executeCommand(final INodeReadTrx mCurrentRtx, final String mParameter) {
303                 return new StringBuilder("Logout from database.").toString();
304             }
305         },
306         EXIT("exit") {
307             @Override
308             String executeCommand(final INodeReadTrx mCurrentRtx, final String mParameter) {
309                 return new StringBuilder("Exiting the program.").toString();
310             }
311         },
312         MOVE("move") {
313             @Override
314             String executeCommand(final INodeReadTrx mCurrentRtx, final String mParameter) {
315                 boolean succeed = false;
316                 final StringBuilder builder = new StringBuilder("Move to ");
317                 if (mParameter.equals("up")) {
318                     builder.append("parent ");
319                     succeed = mCurrentRtx.moveTo(mCurrentRtx.getNode().getParentKey());
320                 } else if (mParameter.equals("down")) {
321                     builder.append("first child ");
322                     succeed = mCurrentRtx.moveTo(((IStructNode)mCurrentRtx.getNode()).getFirstChildKey());
323                 } else if (mParameter.equals("right")) {
324                     builder.append("right sibling ");
325                     succeed = mCurrentRtx.moveTo(((IStructNode)mCurrentRtx.getNode()).getRightSiblingKey());
326                 } else if (mParameter.equals("left")) {
327                     builder.append("left sibling ");
328                     succeed = mCurrentRtx.moveTo(((IStructNode)mCurrentRtx.getNode()).getLeftSiblingKey());
329                 } else if (mParameter.equals("root")) {
330                     builder.append("document root ");
331                     succeed = mCurrentRtx.moveTo(ROOT_NODE);
332                 } else {
333                     try {
334                         final long nodeKey = Long.parseLong(mParameter);
335                         builder.append("node with key ").append(nodeKey).append(" ");
336                         succeed = mCurrentRtx.moveTo(nodeKey);
337                     } catch (final NumberFormatException e) {
338                         builder.append("invalid node ");
339                         succeed = false;
340                     }
341                 }
342                 if (succeed) {
343                     builder.append("succeeded");
344                 } else {
345                     builder.append("not succeeded");
346                 }
347 
348                 return builder.toString();
349             }
350         },
351         MODIFICATION("modification") {
352             @Override
353             String executeCommand(final INodeReadTrx mCurrentRtx, final String mParameter) {
354                 final StringBuilder builder = new StringBuilder("Insert ");
355                 try {
356                     if (mCurrentRtx instanceof INodeWriteTrx) {
357                         final INodeWriteTrx wtx = (INodeWriteTrx)mCurrentRtx;
358 
359                         if (mParameter.equals("commit")) {
360                             wtx.commit();
361                             builder.append(" operation: commit succeed.");
362                         } else if (mParameter.equals("abort")) {
363                             wtx.abort();
364                             builder.append(" operation: abort succeed. ");
365                         }
366 
367                     } else {
368                         builder.append(" not succeed, Please login with write-right "
369                             + "(that means without revision parameter");
370                     }
371                 } catch (final TTException exc) {
372                     builder.append(" throws exception: ").append(exc);
373                 }
374                 return builder.toString();
375             }
376         },
377         NOVALUE("") {
378             @Override
379             String executeCommand(final INodeReadTrx mCurrentRtx, final String mParameter) {
380                 return new StringBuilder("Command not known. Try ").append(Command.HELP.getCommand()).append(
381                     " for known commands!").toString();
382             }
383         };

385         private final String mCommand;

389         Command(final String paramCommand) {

393         private static Command toCommand(final String mCommandString) {
394             try {
395                 final String[] commandStrings = mCommandString.split(COMMANDDELIM);
396                 final Command command = valueOf(commandStrings[0].toUpperCase());
397                 if (commandStrings.length == 2) {
398                     command.setAdvise(commandStrings[1].toLowerCase());
399                 }
400 
401                 return command;
402             } catch (final Exception e) {
403                 return NOVALUE;
404             }
405         }

407         private String executeCommand(final INodeReadTrx read) {
408             return executeCommand(read, mParameter);
409         }
