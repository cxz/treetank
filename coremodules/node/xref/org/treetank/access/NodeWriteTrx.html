82      /**
83       * How is the Hash for this storage computed?
84       */
85      public enum HashKind {
86          /** Rolling hash, only nodes on ancestor axis are touched. */
87          Rolling,
88          /**
89           * Postorder hash, all nodes on ancestor plus postorder are at least
90           * read.
91           */
92          Postorder,
93          /** No hash structure after all. */
94          None;
95      }

130     @Override
131     public long insertElementAsFirstChild(final QName pQName) throws TTException, NullPointerException {
132         checkState(!mDelegate.isClosed(), "Transaction is already closed.");
133         checkNotNull(pQName);
134         checkState(mDelegate.getCurrentNode() instanceof ElementNode
135             || mDelegate.getCurrentNode() instanceof DocumentRootNode,
136             "Insert is not allowed if current node is not an ElementNode, but was %s", mDelegate
137                 .getCurrentNode());
138 
139         final long parentKey = mDelegate.getCurrentNode().getNodeKey();
140         final long leftSibKey = NULL_NODE;
141         final long rightSibKey = ((IStructNode)mDelegate.getCurrentNode()).getFirstChildKey();
142         final ElementNode node = createElementNode(parentKey, leftSibKey, rightSibKey, 0, pQName);
143 
144         mDelegate.setCurrentNode(node);
145         adaptForInsert(node, true);
146         adaptHashesWithAdd();
147 
148         return node.getNodeKey();
149     }

241         final NameNodeDelegate nameDel = new NameNodeDelegate(nodeDel, nameKey, namespaceKey);

245         getPtx().setNode(node);
246 
247         final INode parentNode = (org.treetank.node.interfaces.INode)getPtx().getNode(node.getParentKey());
248         ((ElementNode)parentNode).insertAttribute(node.getNodeKey());
249         getPtx().setNode(parentNode);
250 
251         mDelegate.setCurrentNode(node);
252         adaptForInsert(node, false);
253 
254         adaptHashesWithAdd();

290     private ElementNode createElementNode(final long parentKey, final long mLeftSibKey,
291         final long rightSibKey, final long hash, final QName mName) throws TTException {
292 
293         final int nameKey = insertName(buildName(mName));
294         final int namespaceKey = insertName(mName.getNamespaceURI());
295 
296         final NodeDelegate nodeDel = new NodeDelegate(getPtx().incrementNodeKey(), parentKey, 0);
297         final StructNodeDelegate structDel =
298             new StructNodeDelegate(nodeDel, NULL_NODE, rightSibKey, mLeftSibKey, 0);
299         final NameNodeDelegate nameDel = new NameNodeDelegate(nodeDel, nameKey, namespaceKey);
300         final ElementNode node =
301             new ElementNode(nodeDel, structDel, nameDel, new ArrayList<Long>(), new ArrayList<Long>());
302         getPtx().setNode(node);
303         return node;
304     }

306     private TextNode createTextNode(final long mParentKey, final long mLeftSibKey, final long rightSibKey,
307         final byte[] mValue) throws TTException {
308         final NodeDelegate nodeDel = new NodeDelegate(getPtx().incrementNodeKey(), mParentKey, 0);
309         final ValNodeDelegate valDel = new ValNodeDelegate(nodeDel, mValue);
310         final StructNodeDelegate structDel =
311             new StructNodeDelegate(nodeDel, NULL_NODE, rightSibKey, mLeftSibKey, 0);
312         final TextNode node = new TextNode(nodeDel, structDel, valDel);
313         getPtx().setNode(node);
314         return node;
315     }

317     /**
318      * {@inheritDoc}
319      */
320     @Override
321     public void remove() throws TTException {
322         checkState(!mDelegate.isClosed(), "Transaction is already closed.");
323         checkState(mDelegate.getCurrentNode().getKind() != IConstants.ROOT,
324             "Document root can not be removed.");
325         if (mDelegate.getCurrentNode() instanceof IStructNode) {
326             final IStructNode node = (IStructNode)mDelegate.getCurrentNode();
327             if (node.getKind() == IConstants.ELEMENT) {
328                 long currentKey = node.getNodeKey();
329                 ElementNode element = (ElementNode)node;
330                 for (int i = 0; i < element.getAttributeCount(); i++) {
331                     moveTo(element.getAttributeKey(i));
332                     getPtx().removeNode(mDelegate.getCurrentNode());
333                 }
334                 for (int i = 0; i < element.getNamespaceCount(); i++) {
335                     moveTo(element.getNamespaceKey(i));
336                     getPtx().removeNode(mDelegate.getCurrentNode());
337                 }
338                 moveTo(currentKey);
339             }
340             moveTo(node.getNodeKey());
341             adaptForRemove(node);
342             adaptHashesWithRemove();
343 
344             // Set current node.
345             if (node.hasRightSibling()) {
346                 moveTo(node.getRightSiblingKey());
347             } else if (node.hasLeftSibling()) {
348                 moveTo(node.getLeftSiblingKey());
349             } else {
350                 moveTo(node.getParentKey());
351             }
352         } else if (mDelegate.getCurrentNode().getKind() == IConstants.ATTRIBUTE) {
353             final INode node = mDelegate.getCurrentNode();
354 
355             final ElementNode parentNode = (ElementNode)getPtx().getNode(node.getParentKey());
356             parentNode.removeAttribute(node.getNodeKey());
357             getPtx().setNode(parentNode);
358             adaptHashesWithRemove();
359             moveTo(mDelegate.getCurrentNode().getParentKey());
360         } else if (mDelegate.getCurrentNode().getKind() == IConstants.NAMESPACE) {
361             final INode node = mDelegate.getCurrentNode();
362             final ElementNode parentNode = (ElementNode)getPtx().getNode(node.getParentKey());
363             parentNode.removeNamespace(node.getNodeKey());
364             getPtx().setNode(parentNode);
365             adaptHashesWithRemove();
366             moveTo(mDelegate.getCurrentNode().getParentKey());
367         }
368     }

491     private int insertName(final String pName) throws TTException {
492         final String string = (pName == null ? "" : pName);
493         final int nameKey = NamePageHash.generateHashForString(string);
494         NodeMetaPageFactory.MetaKey key = new NodeMetaPageFactory.MetaKey(nameKey);

504     /**
505      * Adapting everything for insert operations.
506      * 
507      * @param paramNewNode
508      *            pointer of the new node to be inserted
509      * @param addAsFirstChild
510      *            determines the position where to insert
511      * @throws TTIOException
512      *             if anything weird happens
513      */
514     private void adaptForInsert(final INode paramNewNode, final boolean addAsFirstChild) throws TTException {
515         assert paramNewNode != null;
516 
517         if (paramNewNode instanceof IStructNode) {
518             final IStructNode strucNode = (IStructNode)paramNewNode;
519             final IStructNode parent = (IStructNode)getPtx().getNode(paramNewNode.getParentKey());
520             parent.incrementChildCount();
521             if (addAsFirstChild) {
522                 parent.setFirstChildKey(paramNewNode.getNodeKey());
523             }
524             getPtx().setNode(parent);
525 
526             if (strucNode.hasRightSibling()) {
527                 final IStructNode rightSiblingNode =
528                     (IStructNode)getPtx().getNode(strucNode.getRightSiblingKey());
529                 rightSiblingNode.setLeftSiblingKey(paramNewNode.getNodeKey());
530                 getPtx().setNode(rightSiblingNode);
531             }
532             if (strucNode.hasLeftSibling()) {
533                 final IStructNode leftSiblingNode =
534                     (IStructNode)getPtx().getNode(strucNode.getLeftSiblingKey());
535                 leftSiblingNode.setRightSiblingKey(paramNewNode.getNodeKey());
536                 getPtx().setNode(leftSiblingNode);
537             }
538         }
539 
540     }

550     /**
551      * Adapting everything for remove operations.
552      * 
553      * @param pOldNode
554      *            pointer of the old node to be replaces
555      * @throws TTIOException
556      *             if anything weird happens
557      */
558     private void adaptForRemove(final IStructNode pOldNode) throws TTException {
559         assert pOldNode != null;
560 
561         // Adapt left sibling node if there is one.
562         if (pOldNode.hasLeftSibling()) {
563             final IStructNode leftSibling = (IStructNode)getPtx().getNode(pOldNode.getLeftSiblingKey());
564             leftSibling.setRightSiblingKey(pOldNode.getRightSiblingKey());
565             getPtx().setNode(leftSibling);
566         }
567 
568         // Adapt right sibling node if there is one.
569         if (pOldNode.hasRightSibling()) {
570             final IStructNode rightSibling = (IStructNode)getPtx().getNode(pOldNode.getRightSiblingKey());
571             rightSibling.setLeftSiblingKey(pOldNode.getLeftSiblingKey());
572             getPtx().setNode(rightSibling);
573         }
574 
575         // Adapt parent, if node has now left sibling it is a first child.
576         final IStructNode parent = (IStructNode)getPtx().getNode(pOldNode.getParentKey());
577         if (!pOldNode.hasLeftSibling()) {
578             parent.setFirstChildKey(pOldNode.getRightSiblingKey());
579         }
580         parent.decrementChildCount();
581         getPtx().setNode(parent);
582 
583         if (pOldNode.getKind() == IConstants.ELEMENT) {
584             // removing attributes
585             for (int i = 0; i < ((ElementNode)pOldNode).getAttributeCount(); i++) {
586                 moveTo(((ElementNode)pOldNode).getAttributeKey(i));
587                 getPtx().removeNode(mDelegate.getCurrentNode());
588             }
589             // removing namespaces
590             moveTo(pOldNode.getNodeKey());
591             for (int i = 0; i < ((ElementNode)pOldNode).getNamespaceCount(); i++) {
592                 moveTo(((ElementNode)pOldNode).getNamespaceKey(i));
593                 getPtx().removeNode(mDelegate.getCurrentNode());
594             }
595         }
596 
597         // Remove old node.
598         getPtx().removeNode(pOldNode);
599     }

651     /**
652      * Adapting the structure with a hash for all ancestors only with update.
653      * 
654      * @param paramOldHash
655      *            paramOldHash to be removed
656      * @throws TTIOException
657      *             of anything weird happened.
658      */
659     private void adaptHashedWithUpdate(final long paramOldHash) throws TTException {
660         switch (mHashKind) {
661         case Rolling:
662             rollingUpdate(paramOldHash);
663             break;
664         case Postorder:

682     /**
683      * Adapting the structure with a rolling hash for all ancestors only with
684      * insert.
685      * 
686      * @throws TTIOException
687      *             if anything weird happened
688      */
689     private void postorderAdd() throws TTException {
690         // start with hash to add
691         final long startKey = mDelegate.getCurrentNode().getNodeKey();
692         // long for adapting the hash of the parent
693         long hashCodeForParent = 0;
694         // adapting the parent if the current node is no structural one.
695         if (!(mDelegate.getCurrentNode() instanceof IStructNode)) {
696             getPtx().getNode(mDelegate.getCurrentNode().getNodeKey());
697             mDelegate.getCurrentNode().setHash(mDelegate.getCurrentNode().hashCode());
698             getPtx().setNode(mDelegate.getCurrentNode());
699             moveTo(mDelegate.getCurrentNode().getParentKey());
700         }
701         // Cursor to root
702         IStructNode cursorToRoot;
703         do {
704             synchronized (mDelegate.getCurrentNode()) {
705                 cursorToRoot = (IStructNode)getPtx().getNode(mDelegate.getCurrentNode().getNodeKey());
706                 hashCodeForParent = mDelegate.getCurrentNode().hashCode() + hashCodeForParent * PRIME;
707                 // Caring about attributes and namespaces if node is an element.
708                 if (cursorToRoot.getKind() == IConstants.ELEMENT) {
709                     final ElementNode currentElement = (ElementNode)cursorToRoot;
710                     // setting the attributes and namespaces
711                     for (int i = 0; i < ((ElementNode)cursorToRoot).getAttributeCount(); i++) {
712                         moveTo(currentElement.getAttributeKey(i));
713                         hashCodeForParent = mDelegate.getCurrentNode().hashCode() + hashCodeForParent * PRIME;
714                     }
715                     for (int i = 0; i < ((ElementNode)cursorToRoot).getNamespaceCount(); i++) {
716                         moveTo(currentElement.getNamespaceKey(i));
717                         hashCodeForParent = mDelegate.getCurrentNode().hashCode() + hashCodeForParent * PRIME;
718                     }
719                     moveTo(cursorToRoot.getNodeKey());
720                 }
721 
722                 // Caring about the children of a node
723                 if (moveTo(((IStructNode)getNode()).getFirstChildKey())) {
724                     do {
725                         hashCodeForParent = mDelegate.getCurrentNode().getHash() + hashCodeForParent * PRIME;
726                     } while (moveTo(((IStructNode)getNode()).getRightSiblingKey()));
727                     moveTo(((IStructNode)getNode()).getParentKey());
728                 }
729 
730                 // setting hash and resetting hash
731                 cursorToRoot.setHash(hashCodeForParent);
732                 getPtx().setNode(cursorToRoot);
733                 hashCodeForParent = 0;
734             }
735         } while (moveTo(cursorToRoot.getParentKey()));
736 
737         moveTo(startKey);
738     }

741      * Adapting the structure with a rolling hash for all ancestors only with
742      * update.
743      * 
744      * @param paramOldHash
745      *            paramOldHash to be removed
746      * @throws TTIOException
747      *             if anything weird happened
748      */
749     private void rollingUpdate(final long paramOldHash) throws TTException {
750         final INode newNode = mDelegate.getCurrentNode();
751         final long newNodeHash = newNode.hashCode();
752         long resultNew = newNode.hashCode();
753 
754         // go the path to the root
755         do {
756             synchronized (mDelegate.getCurrentNode()) {
757                 getPtx().getNode(mDelegate.getCurrentNode().getNodeKey());
758                 if (mDelegate.getCurrentNode().getNodeKey() == newNode.getNodeKey()) {
759                     resultNew = mDelegate.getCurrentNode().getHash() - paramOldHash;
760                     resultNew = resultNew + newNodeHash;
761                 } else {
762                     resultNew = mDelegate.getCurrentNode().getHash() - (paramOldHash * PRIME);
763                     resultNew = resultNew + newNodeHash * PRIME;
764                 }
765                 mDelegate.getCurrentNode().setHash(resultNew);
766                 getPtx().setNode(mDelegate.getCurrentNode());
767             }
768         } while (moveTo(mDelegate.getCurrentNode().getParentKey()));
769 
770         mDelegate.setCurrentNode(newNode);

776      * 
777      * @throws TTIOException
778      *             if anything weird happened
779      */
780     private void rollingRemove() throws TTException {
781         final INode startNode = mDelegate.getCurrentNode();
782         long hashToRemove = startNode.getHash();
783         long hashToAdd = 0;
784         long newHash = 0;
785         // go the path to the root
786         do {
787             synchronized (mDelegate.getCurrentNode()) {
788                 getPtx().getNode(mDelegate.getCurrentNode().getNodeKey());
789                 if (mDelegate.getCurrentNode().getNodeKey() == startNode.getNodeKey()) {
790                     // the begin node is always null
791                     newHash = 0;
792                 } else if (mDelegate.getCurrentNode().getNodeKey() == startNode.getParentKey()) {
793                     // the parent node is just removed
794                     newHash = mDelegate.getCurrentNode().getHash() - (hashToRemove * PRIME);
795                     hashToRemove = mDelegate.getCurrentNode().getHash();
796                 } else {
797                     // the ancestors are all touched regarding the modification
798                     newHash = mDelegate.getCurrentNode().getHash() - (hashToRemove * PRIME);
799                     newHash = newHash + hashToAdd * PRIME;
800                     hashToRemove = mDelegate.getCurrentNode().getHash();
801                 }
802                 mDelegate.getCurrentNode().setHash(newHash);
803                 hashToAdd = newHash;
804                 getPtx().setNode(mDelegate.getCurrentNode());
805             }
806         } while (moveTo(mDelegate.getCurrentNode().getParentKey()));
807 
808         mDelegate.setCurrentNode(startNode);
809     }

811     /**
812      * Adapting the structure with a rolling hash for all ancestors only with
813      * insert.
814      * 
815      * @throws TTIOException
816      *             if anything weird happened
817      */
818     private void rollingAdd() throws TTException {
819         // start with hash to add
820         final INode startNode = mDelegate.getCurrentNode();
821         long hashToAdd = startNode.hashCode();
822         long newHash = 0;
823         long possibleOldHash = 0;
824         // go the path to the root
825         do {
826             synchronized (mDelegate.getCurrentNode()) {
827                 getPtx().getNode(mDelegate.getCurrentNode().getNodeKey());
828                 if (mDelegate.getCurrentNode().getNodeKey() == startNode.getNodeKey()) {
829                     // at the beginning, take the hashcode of the node only
830                     newHash = hashToAdd;
831                 } else if (mDelegate.getCurrentNode().getNodeKey() == startNode.getParentKey()) {
832                     // at the parent level, just add the node
833                     possibleOldHash = mDelegate.getCurrentNode().getHash();
834                     newHash = possibleOldHash + hashToAdd * PRIME;
835                     hashToAdd = newHash;
836                 } else {
837                     // at the rest, remove the existing old key for this element
838                     // and add the new one
839                     newHash = mDelegate.getCurrentNode().getHash() - (possibleOldHash * PRIME);
840                     newHash = newHash + hashToAdd * PRIME;
841                     hashToAdd = newHash;
842                     possibleOldHash = mDelegate.getCurrentNode().getHash();
843                 }
844                 mDelegate.getCurrentNode().setHash(newHash);
845                 getPtx().setNode(mDelegate.getCurrentNode());
846             }
847         } while (moveTo(mDelegate.getCurrentNode().getParentKey()));
848         mDelegate.setCurrentNode(startNode);
849     }
