57  @Guice(moduleFactory = ModuleFactory.class)
58  public final class OverallTest {
59  
60      private static int NUM_CHARS = 3;
61      private static int ELEMENTS = 1000;
62      private static int COMMITPERCENTAGE = 20;
63      private static int REMOVEPERCENTAGE = 20;
64      private static final Random ran = new Random(0l);
65      public static String chars = "abcdefghijklm";
66  
67      private Holder holder;
68  
69      @Inject
70      private IResourceConfigurationFactory mResourceConfig;
71  
72      private ResourceConfiguration mResource;
73  
74      @BeforeMethod
75      public void setUp() throws TTException {
76          CoreTestHelper.deleteEverything();
77          final CoreTestHelper.Holder holder = CoreTestHelper.Holder.generateStorage();
78          Properties props =
79              StandardSettings.getProps(CoreTestHelper.PATHS.PATH1.getFile().getAbsolutePath(),
80                  CoreTestHelper.RESOURCENAME);
81          mResource = mResourceConfig.create(props);
82          this.holder = Holder.generateWtx(holder,mResource);
83      }
84  
85      @Test
86      public void testJustEverything() throws TTException {
87          NodeTestHelper.createDocumentRootNode(holder.getNWtx());
88          holder.getNWtx().insertElementAsFirstChild(new QName(getString()));
89          for (int i = 0; i < ELEMENTS; i++) {
90              if (ran.nextBoolean()) {
91                  switch (holder.getNWtx().getNode().getKind()) {
92                  case ELEMENT:
93                      holder.getNWtx().setQName(new QName(getString()));
94                      holder.getNWtx().setURI(getString());
95                      break;
96                  case ATTRIBUTE:
97                      holder.getNWtx().setQName(new QName(getString()));
98                      holder.getNWtx().setURI(getString());
99                      holder.getNWtx().setValue(getString());
100                     break;
101                 case NAMESPACE:
102                     holder.getNWtx().setQName(new QName(getString()));
103                     holder.getNWtx().setURI(getString());
104                     break;
105                 case TEXT:
106                     holder.getNWtx().setValue(getString());
107                     break;
108                 default:
109                 }
110             } else {
111                 if (holder.getNWtx().getNode() instanceof ElementNode) {
112                     if (ran.nextBoolean()) {
113                         holder.getNWtx().insertElementAsFirstChild(new QName(getString()));
114                     } else {
115                         holder.getNWtx().insertElementAsRightSibling(new QName(getString()));
116                     }
117                     while (ran.nextBoolean()) {
118                         holder.getNWtx().insertAttribute(new QName(getString()), getString());
119                         holder.getNWtx().moveTo(holder.getNWtx().getNode().getParentKey());
120                     }
121                     while (ran.nextBoolean()) {
122                         holder.getNWtx().insertNamespace(new QName(getString(), getString()));
123                         holder.getNWtx().moveTo(holder.getNWtx().getNode().getParentKey());
124                     }
125                 }
126 
127                 if (ran.nextInt(100) < REMOVEPERCENTAGE) {
128                     holder.getNWtx().remove();
129                 }
130 
131                 if (ran.nextInt(100) < COMMITPERCENTAGE) {
132                     holder.getNWtx().commit();
133                 }
134                 do {
135                     final int newKey = ran.nextInt(i + 1) + 1;
136                     holder.getNWtx().moveTo(newKey);
137                 } while (holder.getNWtx().getNode() == null);
138                 if (holder.getNWtx().getNode().getKind() != IConstants.ELEMENT) {
139                     holder.getNWtx().moveTo(holder.getNWtx().getNode().getParentKey());
140                 }
141             }
142         }
143         final long key = holder.getNWtx().getNode().getNodeKey();
144         holder.getNWtx().remove();
145         holder.getNWtx().insertElementAsFirstChild(new QName(getString()));
146         holder.getNWtx().moveTo(key);
147         holder.getNWtx().commit();
148         holder.getNWtx().close();
149     }

151     @AfterMethod

154         CoreTestHelper.deleteEverything();
155     }

157     private static String getString() {
158         char[] buf = new char[NUM_CHARS];
159 
160         for (int i = 0; i < buf.length; i++) {
161             buf[i] = chars.charAt(ran.nextInt(chars.length()));
162         }
163 
164         return new String(buf);
165     }
