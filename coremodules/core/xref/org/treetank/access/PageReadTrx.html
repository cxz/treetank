54  /**
55   * <h1>PageReadTrx</h1>
56   * 
57   * <p>
58   * State of a reading transaction. The only thing shared amongst transactions is the page cache. Everything
59   * else is exclusive to this transaction. It is required that only a single thread has access to this
60   * transaction.
61   * </p>
62   * 
63   * <p>
64   * A path-like cache boosts sequential operations.
65   * </p>
66   */
67  public class PageReadTrx implements IPageReadTrx {
68  
69      /** Page reader exclusively assigned to this transaction. */
70      private final IReader mPageReader;
71  
72      /** Uber page this transaction is bound to. */
73      private final UberPage mUberPage;
74  
75      /** Cached name page of this revision. */
76      private final RevisionRootPage mRootPage;
77  
78      /** Internal reference to cache. */
79      private final Cache<Long, NodePageContainer> mCache;
80  
81      /** Configuration of the session */
82      protected final Session mSession;
83  
84      /** Boolean for determinc close. */
85      private boolean mClose;

87      /**
88       * Standard constructor.
89       * 
90       * @param pSession
91       *            State of state.
92       * @param pUberpage
93       *            Uber page to start reading with.
94       * @param pRevision
95       *            Key of revision to read from uber page.
96       * @param paramItemList
97       *            List of non-persistent items.
98       * @param pReader
99       *            for this transaction
100      * @throws TTIOException
101      *             if the read of the persistent storage fails
102      */
103     protected PageReadTrx(final Session pSession, final UberPage pUberpage, final long pRevision,
104         final IReader pReader) throws TTIOException {
105         mCache = CacheBuilder.newBuilder().maximumSize(10000).build();
106         mSession = pSession;
107         mPageReader = pReader;
108         mUberPage = pUberpage;
109         mRootPage = loadRevRoot(pRevision);
110         initializeNamePage();
111         mClose = false;
112     }

114     /**
115      * Getting the node related to the given node key.
116      * 
117      * @param paramNodeKey
118      *            searched for
119      * @return the related Node
120      * @throws TTIOException
121      *             if the read to the persistent storage fails
122      */
123     public INode getNode(final long paramNodeKey) throws TTIOException {
124 
125         // Immediately return node from item list if node key negative.
126         if (paramNodeKey < 0) {
127             throw new IllegalArgumentException();
128         }
129 
130         // Calculate page and node part for given nodeKey.
131         final long nodePageKey = nodePageKey(paramNodeKey);
132         final int nodePageOffset = nodePageOffset(paramNodeKey);
133 
134         NodePageContainer cont = mCache.getIfPresent(nodePageKey);
135 
136         if (cont == null) {
137             final NodePage[] revs = getSnapshotPages(nodePageKey);
138             if (revs.length == 0) {
139                 return null;
140             }
141             final int mileStoneRevision = mSession.mResourceConfig.mRevisionsToRestore;
142 
143             // Build up the complete page.
144             final ERevisioning revision = mSession.mResourceConfig.mRevision;
145             final NodePage completePage = revision.combinePages(revs, mileStoneRevision);
146             cont = new NodePageContainer(completePage);
147             mCache.put(nodePageKey, cont);
148         }
149         // If nodePage is a weak one, the moveto is not cached
150         final INode returnVal = cont.getComplete().getNode(nodePageOffset);
151         return checkItemIfDeleted(returnVal);
152     }

207     /**
208      * Get revision root page belonging to revision key.
209      * 
210      * @param revisionKey
211      *            Key of revision to find revision root page for.
212      * @return Revision root page of this revision key.
213      * 
214      * @throws TTIOException
215      *             if something odd happens within the creation process.
216      */
217     protected final RevisionRootPage loadRevRoot(final long revisionKey) throws TTIOException {
218 
219         final PageReference ref = dereferenceLeafOfTree(mUberPage.getIndirectPageReference(), revisionKey);
220         RevisionRootPage page = (RevisionRootPage)ref.getPage();
221 
222         // If there is no page, get it from the storage and cache it.
223         if (page == null) {
224             page = (RevisionRootPage)mPageReader.read(ref.getKey());
225         }
226 
227         // Get revision root page which is the leaf of the indirect tree.
228         return page;
229     }

253     /**
254      * Dereference node page reference.
255      * 
256      * @param mNodePageKey
257      *            Key of node page.
258      * @return Dereferenced page.
259      * 
260      * @throws TTIOException
261      *             if something odd happens within the creation process.
262      */
263     protected final NodePage[] getSnapshotPages(final long mNodePageKey) throws TTIOException {
264 
265         // ..and get all leaves of nodepages from the revision-trees.
266         final List<PageReference> refs = new ArrayList<PageReference>();
267         final Set<Long> keys = new HashSet<Long>();
268 
269         for (long i = mRootPage.getRevision(); i >= 0; i--) {
270             final PageReference ref =
271                 dereferenceLeafOfTree(loadRevRoot(i).getIndirectPageReference(), mNodePageKey);
272             if (ref != null && (ref.getPage() != null || ref.getKey() != null)) {
273                 if (ref.getKey() == null || (!keys.contains(ref.getKey().getIdentifier()))) {
274                     refs.add(ref);
275                     if (ref.getKey() != null) {
276                         keys.add(ref.getKey().getIdentifier());
277                     }
278                 }
279                 if (refs.size() == mSession.mResourceConfig.mRevisionsToRestore) {
280                     break;
281                 }
282 
283             } else {
284                 break;
285             }
286         }
287 
288         // Afterwards read the nodepages if they are not dereferences...
289         final NodePage[] pages = new NodePage[refs.size()];
290         for (int i = 0; i < pages.length; i++) {
291             final PageReference ref = refs.get(i);
292             pages[i] = (NodePage)ref.getPage();
293             if (pages[i] == null) {
294                 pages[i] = (NodePage)mPageReader.read(ref.getKey());
295             }
296         }
297         return pages;
298 
299     }

301     /**
302      * Dereference indirect page reference.
303      * 
304      * @param reference
305      *            Reference to dereference.
306      * @return Dereferenced page.
307      * 
308      * @throws TTIOException
309      *             if something odd happens within the creation process.
310      */
311     protected final IndirectPage dereferenceIndirectPage(final PageReference ref) throws TTIOException {
312 
313         IndirectPage page = (IndirectPage)ref.getPage();
314 
315         // If there is no page, get it from the storage and cache it.
316         if (page == null) {
317             page = (IndirectPage)mPageReader.read(ref.getKey());
318             ref.setPage(page);
319         }
320 
321         return page;
322     }

324     /**
325      * Find reference pointing to leaf page of an indirect tree.
326      * 
327      * @param paramStartReference
328      *            Start reference pointing to the indirect tree.
329      * @param paramKey
330      *            Key to look up in the indirect tree.
331      * @return Reference denoted by key pointing to the leaf page.
332      * 
333      * @throws TTIOException
334      *             if something odd happens within the creation process.
335      */
336     protected final PageReference dereferenceLeafOfTree(final PageReference paramStartReference,
337         final long paramKey) throws TTIOException {
338 
339         // Initial state pointing to the indirect page of level 0.
340         PageReference reference = paramStartReference;
341         int offset = 0;
342         long levelKey = paramKey;
343 
344         // Iterate through all levels.
345         for (int level = 0, height = IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT.length; level < height; level++) {
346             offset = (int)(levelKey >> IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT[level]);
347             levelKey -= offset << IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT[level];
348             final IPage page = dereferenceIndirectPage(reference);
349             if (page == null) {
350                 reference = null;
351                 break;
352             } else {
353                 reference = page.getReferences()[offset];
354             }
355         }
356 
357         // Return reference to leaf of indirect tree.
358         return reference;
359     }
