92      /**
93       * Standard constructor.
94       * 
95       * 
96       * @param pSession
97       *            {@link ISession} reference
98       * @param pUberBucket
99       *            root of resource

101      *            writer where this transaction should write to
102      * @param pRepresentRev
103      *            revision represent
104      * @throws TTIOException
105      *             if IO Error
106      */
107     protected BucketWriteTrx(final ISession pSession, final UberBucket pUberBucket,
108         final IBackendWriter pWriter, final long pRepresentRev) throws TTException {
109         mBucketWriter =
110             new BackendWriterProxy(pWriter, new File(pSession.getConfig().mProperties
111                 .getProperty(org.treetank.access.conf.ConstructorProps.RESOURCEPATH)),
112                 pSession.getConfig().mNodeFac, pSession.getConfig().mMetaFac);
113 
114         final long revkey =
115             BucketReadTrx.dereferenceLeafOfTree(pWriter,
116                 pUberBucket.getReferenceKeys()[IReferenceBucket.GUARANTEED_INDIRECT_OFFSET], pRepresentRev);
117         final RevisionRootBucket revBucket = (RevisionRootBucket)pWriter.read(revkey);
118         final MetaBucket metaBucket =
119             (MetaBucket)pWriter.read(revBucket.getReferenceKeys()[RevisionRootBucket.META_REFERENCE_OFFSET]);
120 
121         mCommitInProgress = Executors.newSingleThreadExecutor();
122         mDelegate = new BucketReadTrx(pSession, pUberBucket, revBucket, metaBucket, pWriter);
123         setUpTransaction(pUberBucket, revBucket, metaBucket, pSession, pRepresentRev, mBucketWriter);
124     }

129     public long setNode(final INode pNode) throws TTException {
130         checkState(!mDelegate.isClosed(), "Transaction already closed");
131         // Allocate node key and increment node count.
132         final long nodeKey = pNode.getNodeKey();
133         final long seqBucketKey = nodeKey >> IConstants.INP_LEVEL_BUCKET_COUNT_EXPONENT[3];
134         final int nodeBucketOffset = nodeBucketOffset(nodeKey);
135         final LogValue container = prepareNodeBucket(nodeKey);
136         final NodeBucket modified = ((NodeBucket)container.getModified());
137         final NodeBucket complete = ((NodeBucket)container.getComplete());
138         modified.setNode(nodeBucketOffset, pNode);
139         complete.setNode(nodeBucketOffset, pNode);
140         mBucketWriter.put(new LogKey(false, IConstants.INP_LEVEL_BUCKET_COUNT_EXPONENT.length, seqBucketKey),
141             container);
142         return nodeKey;
143     }

148     @Override
149     public void removeNode(final INode pNode) throws TTException {
150         checkState(!mDelegate.isClosed(), "Transaction already closed");
151         checkNotNull(pNode);
152         final long nodeBucketKey = pNode.getNodeKey() >> IConstants.INP_LEVEL_BUCKET_COUNT_EXPONENT[3];
153         LogValue container = prepareNodeBucket(pNode.getNodeKey());
154         final INode delNode = new DeletedNode(pNode.getNodeKey());
155         ((NodeBucket)container.getComplete()).setNode(nodeBucketOffset(pNode.getNodeKey()), delNode);
156         ((NodeBucket)container.getModified()).setNode(nodeBucketOffset(pNode.getNodeKey()), delNode);
157 
158         mBucketWriter.put(
159             new LogKey(false, IConstants.INP_LEVEL_BUCKET_COUNT_EXPONENT.length, nodeBucketKey), container);
160     }

162     /**
163      * {@inheritDoc}
164      */
165     public INode getNode(final long pNodeKey) throws TTIOException {
166         checkState(!mDelegate.isClosed(), "Transaction already closed");
167         // Calculate bucket and node part for given nodeKey.
168         final long nodeBucketKey = pNodeKey >> IConstants.INP_LEVEL_BUCKET_COUNT_EXPONENT[3];
169         final int nodeBucketOffset = nodeBucketOffset(pNodeKey);
170 
171         final LogKey key =
172             new LogKey(false, IConstants.INP_LEVEL_BUCKET_COUNT_EXPONENT.length, nodeBucketKey);
173         LogValue container = mBucketWriter.get(key);
174         INode item = null;
175         // Bucket was modified...
176         if (container.getModified() != null) {
177             // ..check if the real node was touched and set it or..
178             if (((NodeBucket)container.getModified()).getNode(nodeBucketOffset) == null) {
179                 item = ((NodeBucket)container.getComplete()).getNode(nodeBucketOffset);
180             }// ..take the node from the complete status of the page.
181             else {
182                 item = ((NodeBucket)container.getModified()).getNode(nodeBucketOffset);
183             }
184             checkNotNull(item, "Item must be set!");
185             item = mDelegate.checkItemIfDeleted(item);
186         }// ...bucket was modified within a former version,...
187         else {
188             // check the former version as..
189             container = mBucketWriter.getFormer(key);
190             // ..modified element within this version or...
191             if (container.getModified() != null) {
192                 // ..check if the real node was touched and set it or..
193                 if (((NodeBucket)container.getModified()).getNode(nodeBucketOffset) == null) {
194                     item = ((NodeBucket)container.getComplete()).getNode(nodeBucketOffset);
195                 }// ..take the node from the complete status of the page.
196                 else {
197                     item = ((NodeBucket)container.getComplete()).getNode(nodeBucketOffset);
198                 }
199                 checkNotNull(item, "Item must be set!");
200                 item = mDelegate.checkItemIfDeleted(item);
201             }// bucket was modified long long before, read it normally.
202             else {
203                 item = mDelegate.getNode(pNodeKey);
204             }
205         }
206         return item;
207     }

216     @Override
217     public void commit() throws TTException {
218         checkState(!mDelegate.isClosed(), "Transaction already closed");
219 
220         mDelegate.mSession.waitForRunningCommit();
221 
222         final UberBucket uber = UberBucket.copy(mNewUber);
223         final MetaBucket meta = MetaBucket.copy(mNewMeta);
224         final RevisionRootBucket rev = RevisionRootBucket.copy(mNewRoot);
225         final Future<Void> commitInProgress = mBucketWriter.commit(uber, meta, rev);
226         mDelegate.mSession.setRunningCommit(mCommitInProgress.submit(new Callable<Void>() {
227             @Override
228             public Void call() throws Exception {
229                 // serializing of new UberPage including its Subtree is concluded.
230                 commitInProgress.get();
231                 ((Session)mDelegate.mSession).setLastCommittedUberBucket(uber);
232                 mDelegate = new BucketReadTrx(mDelegate.mSession, uber, rev, meta, mBucketWriter);
233                 mBucketWriter.closeFormerLog();
234                 return null;
235             }
236         }));
237 
238         mDelegate.mSession.waitForRunningCommit();
239 
240         setUpTransaction(uber, rev, meta, mDelegate.mSession, uber.getRevisionNumber(), mBucketWriter);
241 
242     }

251     @Override
252     public boolean close() throws TTIOException {
253         mCommitInProgress.shutdown();
254         if (!mDelegate.isClosed()) {
255             mDelegate.close();
256 
257             try {
258                 // Try to close the log.
259                 // It may already be closed if a commit
260                 // was the last operation.
261                 mBucketWriter.close();
262             } catch (IllegalStateException e) {
263                 // Do nothing
264             }
265             mDelegate.mSession.deregisterBucketTrx(this);
266             return true;
267         } else {
268             return false;
269         }
270     }

307     private LogValue prepareNodeBucket(final long pNodeKey) throws TTException {
308 
309         final long seqNodeBucketKey = pNodeKey >> IConstants.INP_LEVEL_BUCKET_COUNT_EXPONENT[3];
310 
311         final LogKey key =
312             new LogKey(false, IConstants.INP_LEVEL_BUCKET_COUNT_EXPONENT.length, seqNodeBucketKey);
313         // See if on nodeBucketLevel, there are any buckets.
314         LogValue container = mBucketWriter.get(key);
315         // if not,...
316         if (container.getModified() == null) {
317             // ..start preparing a new container to be logged
318             final LogKey indirectKey = preparePathToLeaf(false, mNewRoot, pNodeKey);
319             final LogValue indirectContainer = mBucketWriter.get(indirectKey);
320             final int nodeOffset = nodeBucketOffset(seqNodeBucketKey);
321             final long bucketKey =
322                 ((IndirectBucket)indirectContainer.getModified()).getReferenceKeys()[nodeOffset];
323             final long newBucketKey = mNewUber.incrementBucketCounter();
324             // if there is not any bucket already existing...
325             if (bucketKey != 0) {
326                 // ...just denote the number of elements necessary to restore (only for visibility reasons).
327                 final int revToRestore =
328                     Integer.parseInt(mDelegate.mSession.getConfig().mProperties
329                         .getProperty(ConstructorProps.NUMBERTORESTORE));
330 
331                 // Gather all data, from the former log..
332                 final LogValue formerModified = mBucketWriter.getFormer(key);
333                 // ..and from the former revision.
334                 final List<NodeBucket> formerBuckets = mDelegate.getSnapshotBuckets(seqNodeBucketKey);
335                 // declare summarized buckets.
336                 final List<NodeBucket> bucketList = new ArrayList<NodeBucket>();
337 
338                 // Look, if a former log is currently in process to be written...
339                 if (formerModified.getModified() != null) {
340                     // ..if so, check if the modified one...
341                     final NodeBucket currentlyInProgress = (NodeBucket)formerModified.getModified();
342                     // ... is the same one than recently written (to avoid race conditions).
343                     if (formerBuckets.isEmpty()
344                         || formerBuckets.get(0).getBucketKey() == currentlyInProgress.getBucketKey()) {
345                         bucketList.add(currentlyInProgress);
346                     }
347                 }
348 
349                 // All currently written elements are inserted so if no elements are in the bucketlist...
350                 if (bucketList.isEmpty()) {
351                     // ...add all former ones...
352                     bucketList.addAll(formerBuckets);
353                 }// ..otherwise, take all elements starting index 1 into account
354                 else {
355                     if (formerBuckets.size() > 1) {
356                         bucketList.addAll(formerBuckets.subList(1, formerBuckets.size() - 1));
357                     }
358                 }
359 
360                 // Transform into array..
361                 final NodeBucket[] buckets = bucketList.toArray(new NodeBucket[bucketList.size()]);
362                 // ..and check that the number of buckets are valid and return the entire bucket.
363                 checkState(buckets.length > 0);
364                 container =
365                     mDelegate.mSession.getConfig().mRevision.combineBucketsForModification(revToRestore,
366                         newBucketKey, buckets, mNewRoot.getRevision() % revToRestore == 0);
367 
368 //                // DEBUG CODE!!!!!
369 //                INode[] toCheck = ((NodeBucket)container.getComplete()).getNodes();
370 //                boolean nullFound = false;
371 //                for (int i = 0; i < toCheck.length && !nullFound; i++) {
372 //                    if ((i < toCheck.length - 1 && i > 0)
373 //                        && (toCheck[i + 1] != null && toCheck[i] == null && toCheck[i - 1] != null)) {
374 //                        nullFound = true;
375 //                    }
376 //                }
377 //                if (nullFound) {
378 //                    System.out.println("-----FAILURE------");
379 //                    for (int i = 0; i < buckets.length; i++) {
380 //                        System.out.println("+++++++++++++++");
381 //                        System.out.println(buckets[i].toString());
382 //                        System.out.println("+++++++++++++++");
383 //                    }
384 //                    System.out.println("-----------");
385 //                    System.exit(-1);
386 //                }
387 
388             }// ...if no bucket is existing, create an entirely new one.
389             else {
390                 final NodeBucket newBucket = new NodeBucket(newBucketKey, IConstants.NULL_NODE);
391                 container = new LogValue(newBucket, newBucket);
392             }
393             ((IndirectBucket)indirectContainer.getModified()).setReferenceKey(nodeOffset, newBucketKey);
394             mBucketWriter.put(indirectKey, indirectContainer);
395             mBucketWriter.put(key, container);
396         }
397         return container;

400     /**
401      * Getting a {@link LogKey} containing the last IndirectBucket with the reference to any new/modified
402      * bucket.
403      * 
404      * @param pIsRootLevel
405      *            is this dereferencing walk based on the the search after a RevRoot or a NodeBucket. Needed
406      *            because of the same keys in both subtrees.
407      * @param pBucket
408      *            where to start the tree-walk: either from an UberBucket (related to new
409      *            RevisionRootBuckets) or from a RevisionRootBucket (related to new NodeBuckets).
410      * @param pElementKey
411      *            key to be dereferenced
412      * @return the key the container representing the last level
413      * @throws TTException
414      */
415     private LogKey preparePathToLeaf(final boolean pIsRootLevel, final IReferenceBucket pBucket,
416         final long pElementKey) throws TTException {
417 
418         // computing the ordernumbers within all level. The ordernumbers are the position in the sequence of
419         // all buckets within the same level.
420         // ranges are for level 0: 0-127; level 1: 0-16383; level 2: 0-2097151; level 3: 0-268435455; ;level
421         // 4: 0-34359738367
422         long seqBucketKey = -1;
423         // since the revision points to a bucket, the sequence-key bases on the last indirect-layer directly
424         // within the search after a revision,...
425         if (pIsRootLevel) {
426             seqBucketKey = pElementKey >> IConstants.INP_LEVEL_BUCKET_COUNT_EXPONENT[3];
427         } // ...whereas one layer above is used for the nodes based on the offsets pointing to nodes
428           // instead of buckets.
429         else {
430             seqBucketKey = pElementKey >> IConstants.INP_LEVEL_BUCKET_COUNT_EXPONENT[2];
431         }
432 
433         long[] orderNumber = new long[IConstants.INP_LEVEL_BUCKET_COUNT_EXPONENT.length];
434         for (int level = 0; level < orderNumber.length; level++) {
435             orderNumber[level] = seqBucketKey >> IConstants.INP_LEVEL_BUCKET_COUNT_EXPONENT[level];
436         }
437 
438         IReferenceBucket bucket = null;
439         IReferenceBucket parentBucket = pBucket;
440         LogKey key = null;
441         LogKey parentKey = new LogKey(pIsRootLevel, -1, 0);
442 
443         // Iterate through all levels...
444         for (int level = 0; level < orderNumber.length; level++) {
445             // ...see if the actual bucket requested is already in the log
446             key = new LogKey(pIsRootLevel, level, orderNumber[level]);
447             LogValue container = mBucketWriter.get(key);
448             // if the bucket is not existing,..
449             if (container.getModified() == null) {
450                 // ..create a new bucket
451                 final long newKey = mNewUber.incrementBucketCounter();
452                 bucket = new IndirectBucket(newKey);
453 
454                 // compute the offset of the new bucket
455                 int offset = nodeBucketOffset(orderNumber[level]);
456 
457                 // if there existed the same bucket in former versions in a former log or
458                 container = mBucketWriter.getFormer(key);
459                 IReferenceBucket oldBucket = null;
460                 if (container.getModified() != null) {
461                     oldBucket = (IReferenceBucket)container.getModified();
462                 }
463                 // over the former log or the offset within
464                 // the parent...
465                 else if (parentBucket.getReferenceKeys()[offset] != 0) {
466                     oldBucket = (IReferenceBucket)mBucketWriter.read(parentBucket.getReferenceKeys()[offset]);
467                 }
468                 // ..copy all references to the new log.
469                 if (oldBucket != null) {
470                     for (int i = 0; i < oldBucket.getReferenceKeys().length; i++) {
471                         bucket.setReferenceKey(i, oldBucket.getReferenceKeys()[i]);
472                     }
473                 }
474 
475                 // Set the newKey on the computed offset...
476                 parentBucket.setReferenceKey(offset, newKey);
477                 // .. and put the parent-reference to the log...
478                 container = new LogValue(parentBucket, parentBucket);
479                 // ..if the parent is not referenced as UberBucket or RevisionRootBucket within the Wtx
480                 // itself...
481                 // if (level > 0) {
482                 mBucketWriter.put(parentKey, container);
483                 // }
484                 // ..but set the reference of the current bucket in every case.
485                 container = new LogValue(bucket, bucket);
486                 mBucketWriter.put(key, container);
487 
488             } // if the bucket is already in the log, get it simply from the log.
489             else {
490                 bucket = (IReferenceBucket)container.getModified();
491             }
492             // finally, set the new bucketkey for the next level
493             parentKey = key;
494             parentBucket = bucket;
495         }
496 
497         // Return reference to leaf of indirect tree.
498         return key;
499     }

501     private void setUpTransaction(final UberBucket pUberOld, final RevisionRootBucket pRootToRepresent,
502         final MetaBucket pMetaOld, final ISession pSession, final long pRepresentRev,
503         final BackendWriterProxy pWriter) throws TTException {
504 
505         mNewUber =
506             new UberBucket(pUberOld.getBucketCounter() + 1, pUberOld.getRevisionNumber() + 1, pUberOld
507                 .getBucketCounter() + 1);
508         mNewUber.setReferenceKey(IReferenceBucket.GUARANTEED_INDIRECT_OFFSET,
509             pUberOld.getReferenceKeys()[IReferenceBucket.GUARANTEED_INDIRECT_OFFSET]);
510 
511         // Prepare indirect tree to hold reference to prepared revision root
512         // nodeBucketReference.
513         final LogKey indirectKey = preparePathToLeaf(true, mNewUber, mNewUber.getRevisionNumber());
514         final LogValue indirectContainer = mBucketWriter.get(indirectKey);
515         final int offset = nodeBucketOffset(mNewUber.getRevisionNumber());
516 
517         // Get previous revision root bucket and using this data to initialize a fresh revision root including
518         // the pointers.
519         mNewRoot =
520             new RevisionRootBucket(mNewUber.incrementBucketCounter(), pRepresentRev + 1, pRootToRepresent
521                 .getMaxNodeKey());
522         mNewRoot.setReferenceKey(IReferenceBucket.GUARANTEED_INDIRECT_OFFSET, pRootToRepresent
523             .getReferenceKeys()[IReferenceBucket.GUARANTEED_INDIRECT_OFFSET]);
524 
525         // setting the new revRoot to the correct offset
526         ((IndirectBucket)indirectContainer.getModified()).setReferenceKey(offset, mNewRoot.getBucketKey());
527         mBucketWriter.put(indirectKey, indirectContainer);
528 
529         // Setting up a new metabucket and link it to the new root
530         Map<IMetaEntry, IMetaEntry> oldMap = pMetaOld.getMetaMap();
531         mNewMeta = new MetaBucket(mNewUber.incrementBucketCounter());
532         for (IMetaEntry key : oldMap.keySet()) {
533             mNewMeta.setEntry(key, oldMap.get(key));
534         }
535         mNewRoot.setReferenceKey(RevisionRootBucket.META_REFERENCE_OFFSET, mNewMeta.getBucketKey());
536 
537     }
