79       * How is the Hash for this storage computed?

81      public enum HashKind {
82          /** Rolling hash, only nodes on ancestor axis are touched. */

123     /**
124      * {@inheritDoc}
125      */
126     @Override
127     public synchronized long insertElementAsFirstChild(final QName mQName)
128             throws AbsTTException, NullPointerException {
129         if (mQName == null) {
130             throw new NullPointerException("mQName may not be null!");
131         }
132         if (mDelegate.getCurrentNode() instanceof ElementNode
133                 || mDelegate.getCurrentNode() instanceof DocumentRootNode) {
134 
135             checkAccessAndCommit();
136 
137             final long parentKey = mDelegate.getCurrentNode().getNodeKey();
138             final long leftSibKey = NULL_NODE;
139             final long rightSibKey = ((IStructNode) mDelegate.getCurrentNode())
140                     .getFirstChildKey();
141             final ElementNode node = createElementNode(parentKey, leftSibKey,
142                     rightSibKey, 0, mQName);
143 
144             mDelegate.setCurrentNode(node);
145             adaptForInsert(node, true);
146             adaptHashesWithAdd();
147 
148             return node.getNodeKey();
149         } else {
150             throw new TTUsageException(
151                     "Insert is not allowed if current node is not an ElementNode!");
152         }
153     }

258     public synchronized long insertAttribute(final QName paramQName,
259             final String paramValueAsString) throws AbsTTException {
260         if (mDelegate.getCurrentNode() instanceof ElementNode) {
261 
262             checkAccessAndCommit();
263 
264             final byte[] value = TypedValue.getBytes(paramValueAsString);
265             final long elementKey = mDelegate.getCurrentNode().getNodeKey();
266 
267             final int nameKey = getPageTransaction().createNameKey(
268                     PageWriteTrx.buildName(paramQName));
269             final int namespaceKey = getPageTransaction().createNameKey(
270                     paramQName.getNamespaceURI());
271             final NodeDelegate nodeDel = new NodeDelegate(getPageTransaction()
272                     .getMaxNodeKey() + 1, elementKey, 0);
273             final NameNodeDelegate nameDel = new NameNodeDelegate(nodeDel,
274                     nameKey, namespaceKey);
275             final ValNodeDelegate valDel = new ValNodeDelegate(nodeDel, value);
276 
277             final AttributeNode node = getPageTransaction().createNode(
278                     new AttributeNode(nodeDel, nameDel, valDel));
279 
280             final INode parentNode = (org.treetank.node.interfaces.INode) getPageTransaction()
281                     .prepareNodeForModification(node.getParentKey());
282             ((ElementNode) parentNode).insertAttribute(node.getNodeKey());
283             getPageTransaction().finishNodeModification(parentNode);
284 
285             mDelegate.setCurrentNode(node);
286             adaptForInsert(node, false);
287 
288             adaptHashesWithAdd();
289             return node.getNodeKey();
290 
291         } else {
292             throw new TTUsageException(
293                     "Insert is not allowed if current node is not an ElementNode!");

342     private ElementNode createElementNode(final long parentKey,
343             final long mLeftSibKey, final long rightSibKey, final long hash,
344             final QName mName) throws TTIOException {
345 
346         final int nameKey = getPageTransaction().createNameKey(
347                 PageWriteTrx.buildName(mName));
348         final int namespaceKey = getPageTransaction().createNameKey(
349                 mName.getNamespaceURI());
350 
351         final NodeDelegate nodeDel = new NodeDelegate(getPageTransaction()
352                 .getMaxNodeKey() + 1, parentKey, 0);
353         final StructNodeDelegate structDel = new StructNodeDelegate(nodeDel,
354                 NULL_NODE, rightSibKey, mLeftSibKey, 0);
355         final NameNodeDelegate nameDel = new NameNodeDelegate(nodeDel, nameKey,
356                 namespaceKey);
357 
358         return getPageTransaction().createNode(
359                 new ElementNode(nodeDel, structDel, nameDel,
360                         new ArrayList<Long>(), new ArrayList<Long>()));
361     }

363     private TextNode createTextNode(final long mParentKey,
364             final long mLeftSibKey, final long rightSibKey, final byte[] mValue)
365             throws TTIOException {
366         final NodeDelegate nodeDel = new NodeDelegate(getPageTransaction()
367                 .getMaxNodeKey() + 1, mParentKey, 0);
368         final ValNodeDelegate valDel = new ValNodeDelegate(nodeDel, mValue);
369         final StructNodeDelegate structDel = new StructNodeDelegate(nodeDel,
370                 NULL_NODE, rightSibKey, mLeftSibKey, 0);
371 
372         return getPageTransaction().createNode(
373                 new TextNode(nodeDel, valDel, structDel));

376     /**
377      * {@inheritDoc}
378      */
379     @Override
380     public synchronized void remove() throws AbsTTException {
381         checkAccessAndCommit();
382         if (mDelegate.getCurrentNode().getKind() == ENode.ROOT_KIND) {
383             throw new TTUsageException("Document root can not be removed.");
384         } else if (mDelegate.getCurrentNode() instanceof IStructNode) {
385             final IStructNode node = (IStructNode) mDelegate.getCurrentNode();
386             // Remove subtree, excluded since 1. axis is now moved to extra
387             // bundle and 2. attributes and
388             // namespaces are ignored
389             // for (final AbsAxis desc = new DescendantAxis(this, false); desc
390             // .hasNext(); desc.next()) {
391             // getTransactionState().removeNode(getCurrentNode());
392             // }
393             moveTo(node.getNodeKey());
394             adaptForRemove(node);
395             adaptHashesWithRemove();
396 
397             // Set current node.
398             if (node.hasRightSibling()) {
399                 moveTo(node.getRightSiblingKey());
400             } else if (node.hasLeftSibling()) {
401                 moveTo(node.getLeftSiblingKey());
402             } else {
403                 moveTo(node.getParentKey());
404             }
405         } else if (mDelegate.getCurrentNode().getKind() == ENode.ATTRIBUTE_KIND) {
406             final INode node = mDelegate.getCurrentNode();
407 
408             final ElementNode parent = (ElementNode) getPageTransaction()
409                     .prepareNodeForModification(node.getParentKey());
410             parent.removeAttribute(node.getNodeKey());
411             getPageTransaction().finishNodeModification(parent);
412             adaptHashesWithRemove();
413             moveTo(mDelegate.getCurrentNode().getParentKey());
414         } else if (mDelegate.getCurrentNode().getKind() == ENode.NAMESPACE_KIND) {
415             final INode node = mDelegate.getCurrentNode();
416 
417             final ElementNode parent = (ElementNode) getPageTransaction()
418                     .prepareNodeForModification(node.getParentKey());
419             parent.removeNamespace(node.getNodeKey());
420             getPageTransaction().finishNodeModification(parent);
421             adaptHashesWithRemove();
422             moveTo(mDelegate.getCurrentNode().getParentKey());
423         }
424     }

501      * {@inheritDoc}

504      *             if paramRevision < 0 or paramRevision > maxCommitedRev

506      *             if an I/O operation fails

508     @Override
509     public void revertTo(final long paramRevision) throws TTUsageException,
510             TTIOException {
511         if (paramRevision < 0) {
512             throw new IllegalArgumentException(
513                     "paramRevision parameter must be >= 0");
514         }
515         mDelegate.assertNotClosed();
516         mSession.assertAccess(paramRevision);
517         final long revNumber = getRevisionNumber();
518         getPageTransaction().close();
519         // Reset internal transaction state to new uber page.
520         mDelegate.setPageTransaction(mSession.beginPageWriteTransaction(
521                 paramRevision, revNumber - 1));
522         moveTo(ROOT_NODE);
523 
524     }

526     /**
527      * {@inheritDoc}
528      */
529     @Override
530     public synchronized void commit() throws AbsTTException {
531 
532         mDelegate.assertNotClosed();
533 
534         // Commit uber page.
535         final UberPage uberPage = getPageTransaction().commit();
536 
537         // Remember succesfully committed uber page in session state.
538         mSession.setLastCommittedUberPage(uberPage);
539 
540         final long revNumber = getRevisionNumber();
541 
542         getPageTransaction().close();
543         // Reset internal transaction state to new uber page.
544         mDelegate.setPageTransaction(mSession.beginPageWriteTransaction(
545                 revNumber, revNumber));
546 
547     }

550      * {@inheritDoc}
551      */
552     @Override
553     public synchronized void abort() throws TTIOException {
554 
555         mDelegate.assertNotClosed();
556 
557         long revisionToSet = 0;
558         if (!getPageTransaction().getUberPage().isBootstrap()) {
559             revisionToSet = getRevisionNumber() - 1;
560         }
561 
562         getPageTransaction().close();
563 
564         // Reset internal transaction state to last committed uber page.
565         mDelegate.setPageTransaction(mSession.beginPageWriteTransaction(
566                 revisionToSet, revisionToSet));
567     }

593     /**
594      * Adapting everything for insert operations.
595      * 
596      * @param paramNewNode
597      *            pointer of the new node to be inserted
598      * @param addAsFirstChild
599      *            determines the position where to insert
600      * @throws TTIOException
601      *             if anything weird happens
602      */
603     private void adaptForInsert(final INode paramNewNode,
604             final boolean addAsFirstChild) throws TTIOException {
605         assert paramNewNode != null;
606 
607         if (paramNewNode instanceof IStructNode) {
608             final IStructNode strucNode = (IStructNode) paramNewNode;
609             final IStructNode parent = (IStructNode) getPageTransaction()
610                     .prepareNodeForModification(paramNewNode.getParentKey());
611             parent.incrementChildCount();
612             if (addAsFirstChild) {
613                 parent.setFirstChildKey(paramNewNode.getNodeKey());
614             }
615             getPageTransaction().finishNodeModification(parent);
616 
617             if (strucNode.hasRightSibling()) {
618                 final IStructNode rightSiblingNode = (IStructNode) getPageTransaction()
619                         .prepareNodeForModification(
620                                 strucNode.getRightSiblingKey());
621                 rightSiblingNode.setLeftSiblingKey(paramNewNode.getNodeKey());
622                 getPageTransaction().finishNodeModification(rightSiblingNode);
623             }
624             if (strucNode.hasLeftSibling()) {
625                 final IStructNode leftSiblingNode = (IStructNode) getPageTransaction()
626                         .prepareNodeForModification(
627                                 strucNode.getLeftSiblingKey());
628                 leftSiblingNode.setRightSiblingKey(paramNewNode.getNodeKey());
629                 getPageTransaction().finishNodeModification(leftSiblingNode);
630             }
631         }
632 
633     }

643     /**
644      * Adapting everything for remove operations.
645      * 
646      * @param paramOldNode
647      *            pointer of the old node to be replaces
648      * @throws TTIOException
649      *             if anything weird happens
650      */
651     private void adaptForRemove(final IStructNode paramOldNode)
652             throws TTIOException {
653         assert paramOldNode != null;
654 
655         // Adapt left sibling node if there is one.
656         if (paramOldNode.hasLeftSibling()) {
657             final IStructNode leftSibling = (IStructNode) getPageTransaction()
658                     .prepareNodeForModification(
659                             paramOldNode.getLeftSiblingKey());
660             leftSibling.setRightSiblingKey(paramOldNode.getRightSiblingKey());
661             getPageTransaction().finishNodeModification(leftSibling);
662         }
663 
664         // Adapt right sibling node if there is one.
665         if (paramOldNode.hasRightSibling()) {
666             final IStructNode rightSibling = (IStructNode) getPageTransaction()
667                     .prepareNodeForModification(
668                             paramOldNode.getRightSiblingKey());
669             rightSibling.setLeftSiblingKey(paramOldNode.getLeftSiblingKey());
670             getPageTransaction().finishNodeModification(rightSibling);
671         }
672 
673         // Adapt parent, if node has now left sibling it is a first child.
674         final IStructNode parent = (IStructNode) getPageTransaction()
675                 .prepareNodeForModification(paramOldNode.getParentKey());
676         if (!paramOldNode.hasLeftSibling()) {
677             parent.setFirstChildKey(paramOldNode.getRightSiblingKey());
678         }
679         parent.decrementChildCount();
680         getPageTransaction().finishNodeModification(parent);
681 
682         if (paramOldNode.getKind() == ENode.ELEMENT_KIND) {
683             // removing attributes
684             for (int i = 0; i < ((ElementNode) paramOldNode)
685                     .getAttributeCount(); i++) {
686                 moveTo(((ElementNode) paramOldNode).getAttributeKey(i));
687                 getPageTransaction().removeNode(mDelegate.getCurrentNode());
688             }
689             // removing namespaces
690             moveTo(paramOldNode.getNodeKey());
691             for (int i = 0; i < ((ElementNode) paramOldNode)
692                     .getNamespaceCount(); i++) {
693                 moveTo(((ElementNode) paramOldNode).getNamespaceKey(i));
694                 getPageTransaction().removeNode(mDelegate.getCurrentNode());
695             }
696         }
697 
698         // Remove old node.
699         getPageTransaction().removeNode(paramOldNode);
700     }

785      * Adapting the structure with a rolling hash for all ancestors only with
786      * insert.
787      * 
788      * @throws TTIOException
789      *             if anything weird happened
790      */
791     private void postorderAdd() throws TTIOException {
792         // start with hash to add
793         final INode startNode = mDelegate.getCurrentNode();
794         // long for adapting the hash of the parent
795         long hashCodeForParent = 0;
796         // adapting the parent if the current node is no structural one.
797         if (!(mDelegate.getCurrentNode() instanceof IStructNode)) {
798             getPageTransaction().prepareNodeForModification(
799                     mDelegate.getCurrentNode().getNodeKey());
800             mDelegate.getCurrentNode().setHash(
801                     mDelegate.getCurrentNode().hashCode());
802             getPageTransaction().finishNodeModification(
803                     mDelegate.getCurrentNode());
804             moveTo(mDelegate.getCurrentNode().getParentKey());
805         }
806         // Cursor to root
807         IStructNode cursorToRoot;
808         do {
809             synchronized (mDelegate.getCurrentNode()) {
810                 cursorToRoot = (IStructNode) getPageTransaction()
811                         .prepareNodeForModification(
812                                 mDelegate.getCurrentNode().getNodeKey());
813                 hashCodeForParent = mDelegate.getCurrentNode().hashCode()
814                         + hashCodeForParent * PRIME;
815                 // Caring about attributes and namespaces if node is an element.
816                 if (cursorToRoot.getKind() == ENode.ELEMENT_KIND) {
817                     final ElementNode currentElement = (ElementNode) cursorToRoot;
818                     // setting the attributes and namespaces
819                     for (int i = 0; i < ((ElementNode) cursorToRoot)
820                             .getAttributeCount(); i++) {
821                         moveTo(currentElement.getAttributeKey(i));
822                         hashCodeForParent = mDelegate.getCurrentNode()
823                                 .hashCode() + hashCodeForParent * PRIME;
824                     }
825                     for (int i = 0; i < ((ElementNode) cursorToRoot)
826                             .getNamespaceCount(); i++) {
827                         moveTo(currentElement.getNamespaceKey(i));
828                         hashCodeForParent = mDelegate.getCurrentNode()
829                                 .hashCode() + hashCodeForParent * PRIME;
830                     }
831                     moveTo(cursorToRoot.getNodeKey());
832                 }
833 
834                 // Caring about the children of a node
835                 if (moveTo(((IStructNode) getNode()).getFirstChildKey())) {
836                     do {
837                         hashCodeForParent = mDelegate.getCurrentNode()
838                                 .getHash() + hashCodeForParent * PRIME;
839                     } while (moveTo(((IStructNode) getNode())
840                             .getRightSiblingKey()));
841                     moveTo(((IStructNode) getNode()).getParentKey());
842                 }
843 
844                 // setting hash and resetting hash
845                 cursorToRoot.setHash(hashCodeForParent);
846                 getPageTransaction().finishNodeModification(cursorToRoot);
847                 hashCodeForParent = 0;
848             }
849         } while (moveTo(cursorToRoot.getParentKey()));
850 
851         mDelegate.setCurrentNode(startNode);
852     }

943     private void rollingAdd() throws TTIOException {
944         // start with hash to add
945         final INode startNode = mDelegate.getCurrentNode();
946         long hashToAdd = startNode.hashCode();
947         long newHash = 0;
948         long possibleOldHash = 0;
949         // go the path to the root
950         do {
951             synchronized (mDelegate.getCurrentNode()) {
952                 getPageTransaction().prepareNodeForModification(
953                         mDelegate.getCurrentNode().getNodeKey());
954                 if (mDelegate.getCurrentNode().getNodeKey() == startNode
955                         .getNodeKey()) {
956                     // at the beginning, take the hashcode of the node only
957                     newHash = hashToAdd;
958                 } else if (mDelegate.getCurrentNode().getNodeKey() == startNode
959                         .getParentKey()) {
960                     // at the parent level, just add the node
961                     possibleOldHash = mDelegate.getCurrentNode().getHash();
962                     newHash = possibleOldHash + hashToAdd * PRIME;
963                     hashToAdd = newHash;
964                 } else {
965                     // at the rest, remove the existing old key for this element
966                     // and add the new one
967                     newHash = mDelegate.getCurrentNode().getHash()
968                             - (possibleOldHash * PRIME);
969                     newHash = newHash + hashToAdd * PRIME;
970                     hashToAdd = newHash;
971                     possibleOldHash = mDelegate.getCurrentNode().getHash();
972                 }
973                 mDelegate.getCurrentNode().setHash(newHash);
974                 getPageTransaction().finishNodeModification(
975                         mDelegate.getCurrentNode());
976             }
977         } while (moveTo(mDelegate.getCurrentNode().getParentKey()));
