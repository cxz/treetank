34      public static Properties getProps(final String pathToStorage, final String resource) throws TTIOException {
35          Properties properties = new Properties();
36          properties.setProperty(ConstructorProps.STORAGEPATH, pathToStorage);
37          properties.setProperty(ConstructorProps.RESOURCE, resource);
38          properties.setProperty(ConstructorProps.RESOURCEPATH, FileSystems.getDefault().getPath(pathToStorage,
39              StorageConfiguration.Paths.Data.getFile().getName(), resource).toString());
40          properties.setProperty(ConstructorProps.NUMBERTORESTORE, Integer.toString(4));
41  
42          // properties.setProperty(ConstructorProps.JCLOUDSTYPE, "aws-s3");
43          properties.setProperty(ConstructorProps.JCLOUDSTYPE, "filesystem");
44          properties.setProperty(FilesystemConstants.PROPERTY_BASEDIR, Files.createTempDir().getAbsolutePath());
45          String[] awsCredentials = getCredentials();
46          if (awsCredentials.length == 0) {
47              properties.setProperty(Constants.PROPERTY_CREDENTIAL, "test");
48          } else {
49              properties.setProperty(Constants.PROPERTY_IDENTITY, awsCredentials[0]);
50              properties.setProperty(Constants.PROPERTY_CREDENTIAL, awsCredentials[1]);
51          }

66      private static String[] getCredentials() {
67          File userStore =
68              new File(System.getProperty("user.home"), new StringBuilder(".credentials")
69                  .append(File.separator).append("aws.properties").toString());
70          if (!userStore.exists()) {
71              return new String[0];
72          } else {
73              Properties props = new Properties();
74              try {
75                  props.load(new FileReader(userStore));
76                  return new String[] {
77                      props.getProperty("access"), props.getProperty("secret")
78                  };
79  
80              } catch (IOException exc) {
81                  throw new RuntimeException(exc);
82              }
83          }
84  
85      }
