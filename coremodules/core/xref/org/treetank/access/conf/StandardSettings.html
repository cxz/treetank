41      private static byte[] keyValue = new byte[] {
42          'k', 'k', 'k', 'k', 'k', 'k', 'k', 'k', 'k', 'k', 'k', 'k', 'k', 'k', 'k', 'k'
43      };
44      public static final Key KEY;
45      static {
46          KEY = new SecretKeySpec(keyValue, "AES");
47      }
48  
49      @Override
50      protected void configure() {
51          bind(INodeFactory.class).to(DumbNodeFactory.class);
52          configureNormal();
53      }
54  
55      public void configureNormal() {
56          install(new FactoryModuleBuilder().implement(IRevisioning.class, Differential.class).build(
57              IRevisioningFactory.class));
58          bind(IByteHandlerPipeline.class).toInstance(new ByteHandlerPipeline(new Zipper()));
59          install(new FactoryModuleBuilder().implement(IBackend.class, FileStorage.class).build(
60              IBackendFactory.class));
61          install(new FactoryModuleBuilder().build(IResourceConfigurationFactory.class));
62  
63          bind(Key.class).toInstance(KEY);
64          install(new FactoryModuleBuilder().build(ISessionConfigurationFactory.class));
65      }
66  
67      public static Properties getStandardProperties(final String mDatabase, final String resource) {
68          Properties properties = new Properties();
69          properties.setProperty(IConstants.DBFILE, mDatabase);
70          properties.setProperty(IConstants.RESOURCE, resource);
71          properties.setProperty(FilesystemConstants.PROPERTY_BASEDIR, new File(new File(new File(properties
72              .getProperty(IConstants.DBFILE), StorageConfiguration.Paths.Data.getFile().getName()), properties
73              .getProperty(IConstants.RESOURCE)), ResourceConfiguration.Paths.Data.getFile().getName())
74              .getAbsolutePath());
75          properties.setProperty(Constants.PROPERTY_CREDENTIAL, "test");
76          properties.setProperty(IConstants.JCLOUDSTYPE, "imagestore");
77          // Class name for painter for imagehost
78          properties.setProperty(ImageStoreConstants.PROPERTY_BYTEPAINTER,
79              "org.jclouds.imagestore.imagegenerator.bytepainter.HexadecimalBytesToImagePainter");
80          // Class name for imagehost
81          properties.setProperty(ImageStoreConstants.PROPERTY_IMAGEHOSTER,
82              "org.jclouds.imagestore.imagehoster.file.ImageHostFile");
83          // properties.setProperty(ImageStoreConstants.PROPERTY_IMAGEHOSTER,
84          // "org.jclouds.imagestore.imagehoster.flickr.ImageHostFlickr");
85          return properties;
86      }
