194     private static final String[] JSONNAMES = {
195         "metaentryClass", "revisioningClass", "nodeFactoryClass", "byteHandlerClasses", "storageClass",
196         "properties"
197     };
198 
199     public static void serialize(final ResourceConfiguration pConfig) throws TTIOException {
200         try {
201 
202             final File file =
203                 new File(new File(new File(pConfig.mProperties.getProperty(ConstructorProps.STORAGEPATH),
204                     StorageConfiguration.Paths.Data.getFile().getName()), pConfig.mProperties
205                     .getProperty(ConstructorProps.RESOURCE)), Paths.ConfigBinary.getFile().getName());
206 
207             FileWriter fileWriter = new FileWriter(file);
208             JsonWriter jsonWriter = new JsonWriter(fileWriter);
209             jsonWriter.beginObject();
210             // caring about the meta-entries
211             jsonWriter.name(JSONNAMES[0]).value(pConfig.mMetaFac.getClass().getName());
212             // caring about the versioning
213             jsonWriter.name(JSONNAMES[1]).value(pConfig.mRevision.getClass().getName());
214             // caring about the NodeFactory
215             jsonWriter.name(JSONNAMES[2]).value(pConfig.mNodeFac.getClass().getName());
216             // caring about the ByteHandlers
217             IByteHandlerPipeline byteHandler = pConfig.mBackend.getByteHandler();
218             jsonWriter.name(JSONNAMES[3]);
219             jsonWriter.beginArray();
220             for (IByteHandler handler : byteHandler) {
221                 jsonWriter.value(handler.getClass().getName());
222             }
223             jsonWriter.endArray();
224             // caring about the storage
225             jsonWriter.name(JSONNAMES[4]).value(pConfig.mBackend.getClass().getName());
226             jsonWriter.name(JSONNAMES[5]);
227             jsonWriter.beginObject();
228             for (String key : pConfig.mProperties.stringPropertyNames()) {
229                 jsonWriter.name(key).value(pConfig.mProperties.getProperty(key));
230             }
231             jsonWriter.endObject();
232             jsonWriter.endObject();
233             jsonWriter.close();
234             fileWriter.close();
235         } catch (final IOException exc) {
236             throw new TTIOException(exc);
237         }
238     }

240     /**
241      * Deserializing a Resourceconfiguration out of a JSON-file from the persistent storage.
242      * The order is important and the reader is passed through the objects as visitor.
243      * 
244      * @param pFile
245      *            where the resource lies in.
246      * @return a complete {@link ResourceConfiguration} instance.
247      * @throws TTIOException
248      */
249     public static ResourceConfiguration deserialize(final File pFile, final String pResource)
250         throws TTIOException {
251         try {
252 
253             final File file =
254                 new File(new File(new File(pFile, StorageConfiguration.Paths.Data.getFile().getName()),
255                     pResource), Paths.ConfigBinary.getFile().getName());
256 
257             FileReader fileReader = new FileReader(file);
258             JsonReader jsonReader = new JsonReader(fileReader);
259             jsonReader.beginObject();
260             // caring about the kind of the metabucket
261             jsonReader.nextName().equals(JSONNAMES[0]);
262             Class<?> metaBucketClazz = Class.forName(jsonReader.nextString());
263             // caring about the versioning
264             jsonReader.nextName().equals(JSONNAMES[1]);
265             Class<?> revClazz = Class.forName(jsonReader.nextString());
266             // caring about the NodeFactory
267             jsonReader.nextName().equals(JSONNAMES[2]);
268             Class<?> nodeFacClazz = Class.forName(jsonReader.nextString());
269 
270             // caring about the ByteHandlers
271             List<IByteHandler> handlerList = new ArrayList<IByteHandler>();
272             if (jsonReader.nextName().equals(JSONNAMES[3])) {
273                 jsonReader.beginArray();
274                 while (jsonReader.hasNext()) {
275                     Class<?> handlerClazz = Class.forName(jsonReader.nextString());
276                     Constructor<?> handlerCons = handlerClazz.getConstructors()[0];
277                     if (handlerClazz.getName().equals(Encryptor.class.getName())) {
278                         handlerList.add((IByteHandler)handlerCons.newInstance(StandardSettings.KEY));
279                     } else {
280                         handlerList.add((IByteHandler)handlerCons.newInstance());
281                     }
282 
283                 }
284                 jsonReader.endArray();
285             }
286             ByteHandlerPipeline pipeline =
287                 new ByteHandlerPipeline(handlerList.toArray(new IByteHandler[handlerList.size()]));
288             // caring about the storage
289             jsonReader.nextName().equals(JSONNAMES[4]);
290             Class<?> storageClazz = Class.forName(jsonReader.nextString());
291             jsonReader.nextName().equals(JSONNAMES[5]);
292             Properties props = new Properties();
293             jsonReader.beginObject();
294             while (jsonReader.hasNext()) {
295                 props.setProperty(jsonReader.nextName(), jsonReader.nextString());
296             }
297             jsonReader.endObject();
298             jsonReader.endObject();
299             jsonReader.close();
300             fileReader.close();
301 
302             Constructor<?> metaBucketCons = metaBucketClazz.getConstructors()[0];
303             IMetaEntryFactory metaFac = (IMetaEntryFactory)metaBucketCons.newInstance();
304 
305             Constructor<?> nodeFacCons = nodeFacClazz.getConstructors()[0];
306             INodeFactory nodeFactory = (INodeFactory)nodeFacCons.newInstance();
307 
308             Constructor<?> revCons = revClazz.getConstructors()[0];
309             IRevisioning revObject = (IRevisioning)revCons.newInstance();
310 
311             Constructor<?> storageCons = storageClazz.getConstructors()[0];
312             IBackend backend = (IBackend)storageCons.newInstance(props, nodeFactory, metaFac, pipeline);
313 
314             return new ResourceConfiguration(props, backend, revObject, nodeFactory, metaFac);
315 
316         } catch (IOException | ClassNotFoundException | InstantiationException | IllegalAccessException
317         | InvocationTargetException exc) {
318             throw new TTIOException(exc);
319         }
320     }
