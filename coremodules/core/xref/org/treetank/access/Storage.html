82      /**
83       * Creating a storage. This includes loading the storageconfiguration,
84       * building up the structure and preparing everything for login.
85       * 
86       * 
87       * @param pStorageConfig
88       *            which are used for the storage, including storage location
89       * @return true if creation is valid, false otherwise
90       * @throws TTIOException
91       *             if something odd happens within the creation process.
92       */
93      public static synchronized boolean createStorage(final StorageConfiguration pStorageConfig)
94          throws TTIOException {
95          boolean returnVal = true;
96          // if file is existing, skipping
97          if (pStorageConfig.mFile.exists()) {
98              return false;
99          } else {
100             returnVal = pStorageConfig.mFile.mkdirs();
101             if (returnVal) {
102                 // creation of folder structure
103                 for (StorageConfiguration.Paths paths : StorageConfiguration.Paths.values()) {
104                     final File toCreate = new File(pStorageConfig.mFile, paths.getFile().getName());
105                     if (paths.isFolder()) {
106                         returnVal = toCreate.mkdir();
107                     } else {
108                         try {
109                             returnVal = toCreate.createNewFile();
110                         } catch (final IOException exc) {
111                             throw new TTIOException(exc);
112                         }
113                     }
114                     if (!returnVal) {
115                         break;
116                     }
117                 }
118             }
119             // serialization of the config
120             StorageConfiguration.serialize(pStorageConfig);
121             // if something was not correct, delete the partly created
122             // substructure
123             if (!returnVal) {
124                 pStorageConfig.mFile.delete();
125             }
126             return returnVal;
127         }
128     }

130     /**
131      * Truncate a storage. This deletes all relevant data. All running sessions
132      * must be closed beforehand.
133      * 
134      * @param pConf
135      *            the storage at this path should be deleted.
136      * @throws TTException
137      *             any kind of false Treetank behaviour
138      */
139     public static synchronized void truncateStorage(final StorageConfiguration pConf) throws TTException {
140         // check that database must be closed beforehand
141         if (!STORAGEMAP.containsKey(pConf.mFile)) {
142             if (existsStorage(pConf.mFile)) {
143                 final IStorage storage = new Storage(pConf);
144                 final File[] resources =
145                     new File(pConf.mFile, StorageConfiguration.Paths.Data.getFile().getName()).listFiles();
146                 for (final File resource : resources) {
147                     storage.truncateResource(new SessionConfiguration(resource.getName(), null));
148                 }
149                 storage.close();
150                 // instantiate the database for deletion
151                 IOUtils.recursiveDelete(pConf.mFile);
152             }
153         }
154     }

181     /**
182      * {@inheritDoc}
183      */
184     @Override
185     public synchronized boolean createResource(final ResourceConfiguration pResConf) throws TTIOException {
186         boolean returnVal = true;
187         // Setting the missing params in the settings, this overrides already
188         // set data.
189         final File path =
190             new File(new File(pResConf.mProperties.getProperty(IConstants.DBFILE),
191                 StorageConfiguration.Paths.Data.getFile().getName()), pResConf.mProperties
192                 .getProperty(IConstants.RESOURCE));
193         // if file is existing, skipping
194         if (path.exists()) {
195             return false;
196         } else {
197             returnVal = path.mkdir();
198             if (returnVal) {
199                 // creation of the folder structure
200                 for (ResourceConfiguration.Paths paths : ResourceConfiguration.Paths.values()) {
201                     final File toCreate = new File(path, paths.getFile().getName());
202                     if (paths.isFolder()) {
203                         returnVal = toCreate.mkdir();
204                     } else {
205                         try {
206                             returnVal = toCreate.createNewFile();
207                         } catch (final IOException exc) {
208                             throw new TTIOException(exc);
209                         }
210                     }
211                     if (!returnVal) {
212                         break;
213                     }
214                 }
215             }
216             // serialization of the config
217             ResourceConfiguration.serialize(pResConf);
218             // if something was not correct, delete the partly created
219             // substructure
220             if (!returnVal) {
221                 throw new IllegalStateException(new StringBuilder("Failure, please remove folder ").append(
222                     pResConf.mProperties.getProperty(IConstants.DBFILE)).append(" manually!").toString());
223             }
224             return returnVal;
225         }
226     }

231      */
232     @Override
233     public synchronized void truncateResource(final SessionConfiguration pResConf) throws TTException {
234         final File resourceFile =
235             new File(new File(mStorageConfig.mFile, StorageConfiguration.Paths.Data.getFile().getName()),
236                 pResConf.getResource());
237         // check that database must be closed beforehand
238         if (!mSessions.containsKey(resourceFile) && existsResource(pResConf.getResource())) {
239             ISession session = getSession(pResConf);
240             session.close();
241             session.truncate();
242         }
243     }

252     /**
253      * Open database. A database can be opened only once. Afterwards the
254      * singleton instance bound to the File is given back.
255      * 
256      * @param pFile
257      *            where the database is located sessionConf a {@link SessionConfiguration} object to set up
258      *            the session
259      * @return {@link IStorage} instance.
260      * @throws TTException
261      *             if something odd happens
262      */
263     public static synchronized IStorage openStorage(final File pFile) throws TTException {
264         if (!existsStorage(pFile)) {
265             throw new TTUsageException("DB could not be opened (since it was not created?) at location",
266                 pFile.toString());
267         }
268         StorageConfiguration config = StorageConfiguration.deserialize(pFile);
269         final Storage storage = new Storage(config);
270         final IStorage returnVal = STORAGEMAP.putIfAbsent(pFile, storage);
271         if (returnVal == null) {
272             return storage;
273         } else {
274             return returnVal;
275         }
276     }

286     /**
287      * {@inheritDoc}
288      */
289     @Override
290     public synchronized ISession getSession(final SessionConfiguration pSessionConf) throws TTException {
291 
292         final File resourceFile =
293             new File(new File(mStorageConfig.mFile, StorageConfiguration.Paths.Data.getFile().getName()),
294                 pSessionConf.getResource());
295         Session returnVal = mSessions.get(resourceFile);
296         if (returnVal == null) {
297             if (!resourceFile.exists()) {
298                 throw new TTUsageException(
299                     "Resource could not be opened (since it was not created?) at location", resourceFile
300                         .toString());
301             }
302             ResourceConfiguration config =
303                 ResourceConfiguration.deserialize(mStorageConfig.mFile, pSessionConf.getResource());
304 
305             returnVal = new Session(this, config, pSessionConf);
306             mSessions.put(pSessionConf.getResource(), returnVal);
307         }
308         return returnVal;
309     }
310 
311     /**
312      * {@inheritDoc}
313      */
314     @Override
315     public synchronized void close() throws TTException {
316         for (final ISession session : mSessions.values()) {
317             session.close();
318         }
319         STORAGEMAP.remove(mStorageConfig.mFile);
320     }
