110     /**
111      * Prepare a node for modification. This is getting the node from the
112      * (persistence) layer, storing the page in the cache and setting up the
113      * node for upcoming modification. Note that this only occurs for {@link INode}s.
114      * 
115      * @param pNodeKey
116      *            key of the node to be modified
117      * @return an {@link INode} instance
118      * @throws TTIOException
119      *             if IO Error
120      */
121     protected INode prepareNodeForModification(final long pNodeKey) throws TTException {
122         if (pNodeKey < 0) {
123             throw new IllegalArgumentException("paramNodeKey must be >= 0!");
124         }
125         if (mNodePageCon != null) {
126             throw new IllegalStateException(
127                 "Another node page container is currently in the cache for updates!");
128         }
129 
130         final long nodePageKey = nodePageKey(pNodeKey);
131         final int nodePageOffset = nodePageOffset(pNodeKey);
132         prepareNodePage(nodePageKey);
133 
134         INode node = mNodePageCon.getModified().getNode(nodePageOffset);
135         if (node == null) {
136             final INode oldNode = mNodePageCon.getComplete().getNode(nodePageOffset);
137             if (oldNode == null) {
138                 throw new TTIOException("Cannot retrieve node from cache");
139             }
140             node = oldNode;
141             mNodePageCon.getModified().setNode(nodePageOffset, node);
142         }
143         return node;
144     }

146     /**
147      * Finishing the node modification. That is storing the node including the
148      * page in the cache.
149      * 
150      * @param pNode
151      *            the node to be modified
152      */
153     protected void finishNodeModification(final INode pNode) {
154         final long nodePageKey = nodePageKey(pNode.getNodeKey());
155         if (mNodePageCon == null || pNode == null || mLog.get(nodePageKey) == null) {
156             throw new IllegalStateException();
157         }
158 
159         mLog.put(nodePageKey, mNodePageCon);
160 
161         this.mNodePageCon = null;
162 
163     }

165     /**
166      * Create fresh node and prepare node nodePageReference for modifications
167      * (COW).
168      * 
169      * @param pNode
170      *            node to add
171      * @return Unmodified node from parameter for convenience.
172      * @throws TTIOException
173      *             if IO Error
174      */
175     public <T extends INode> T createNode(final T pNode) throws TTException {
176         // Allocate node key and increment node count.
177         mNewRoot.incrementMaxNodeKey();
178         final long nodeKey = mNewRoot.getMaxNodeKey();
179         final long nodePageKey = nodePageKey(nodeKey);
180         final int nodePageOffset = nodePageOffset(nodeKey);
181         prepareNodePage(nodePageKey);
182         final NodePage page = mNodePageCon.getModified();
183         page.setNode(nodePageOffset, pNode);
184         finishNodeModification(pNode);
185         return pNode;
186     }

188     /**
189      * Removing a node from the storage.
190      * 
191      * @param pNode
192      *            {@link INode} to be removed
193      * @throws TTIOException
194      *             if the removal fails
195      */
196     protected void removeNode(final INode pNode) throws TTException {
197         assert pNode != null;
198         final long nodePageKey = nodePageKey(pNode.getNodeKey());
199         prepareNodePage(nodePageKey);
200         final INode delNode = new DeletedNode(pNode.getNodeKey());
201         mNodePageCon.getModified().setNode(nodePageOffset(pNode.getNodeKey()), delNode);
202         mNodePageCon.getComplete().setNode(nodePageOffset(pNode.getNodeKey()), delNode);
203         finishNodeModification(pNode);
204     }

206     /**
207      * {@inheritDoc}
208      */
209     public INode getNode(final long pNodeKey) throws TTException {
210 
211         // Calculate page and node part for given nodeKey.
212         final long nodePageKey = nodePageKey(pNodeKey);
213         final int nodePageOffset = nodePageOffset(pNodeKey);
214 
215         final NodePageContainer pageCont = mLog.get(nodePageKey);
216         if (pageCont == null) {
217             return mDelegate.getNode(pNodeKey);
218         } else if (pageCont.getModified().getNode(nodePageOffset) == null) {
219             final INode item = pageCont.getComplete().getNode(nodePageOffset);
220             return mDelegate.checkItemIfDeleted(item);
221 
222         } else {
223             final INode item = pageCont.getModified().getNode(nodePageOffset);
224             return mDelegate.checkItemIfDeleted(item);
225         }
226 
227     }

236     public String getName(final int pNameKey) {
237         final NamePage currentNamePage = (NamePage)mNewRoot.getNamePageReference().getPage();
238         String returnVal;
239         // if currentNamePage == null -> state was commited and no
240         // prepareNodepage was invoked yet
241         if (currentNamePage == null || currentNamePage.getName(pNameKey) == null) {
242             returnVal = mDelegate.getName(pNameKey);
243         } else {
244             returnVal = currentNamePage.getName(pNameKey);
245         }
246         return returnVal;

271     /**
272      * Committing a a writetransaction. This method is recursivly invoked by all {@link PageReference}s.
273      * 
274      * @param pRef
275      *            to be commited
276      * @throws TTException
277      *             if the write fails
278      */
279     public void commit(final PageReference pRef) throws TTException {
280         IPage page = null;
281 
282         // if reference is not null, get one from the persistent storage.
283         if (pRef != null) {
284             // first, try to get one from the log
285             final NodePageContainer cont = mLog.get(pRef.getNodePageKey());
286             if (cont != null) {
287                 page = cont.getModified();
288             }
289             // if none is in the log, test if one is instantiated, if so, get
290             // the one flexible from the
291             // reference
292             if (page == null) {
293                 page = pRef.getPage();
294                 if (page == null) {
295                     return;
296                 }
297             }
298 
299             pRef.setPage(page);
300             // Recursively commit indirectely referenced pages and then
301             // write self.
302             page.commit(this);
303             mPageWriter.write(pRef);
304             pRef.setPage(null);
305         }
306     }

399     protected IndirectPage prepareIndirectPage(final PageReference pRef) throws TTException {
400 
401         IndirectPage page = (IndirectPage)pRef.getPage();
402         if (page == null) {
403             if (pRef.getKey() == IConstants.NULL_ID) {
404                 page = new IndirectPage(mDelegate.getUberPage().getRevision());
405             } else {
406                 page =
407                     new IndirectPage((IndirectPage)mDelegate.dereferenceIndirectPage(pRef),
408                         mNewRoot.getRevision() + 1);
409 
410             }
411             pRef.setPage(page);
412         }
413         return page;
414     }

416     protected NodePageContainer prepareNodePage(final long pPageKey) throws TTException {
417 
418         // Last level points to node nodePageReference.
419         NodePageContainer cont = mLog.get(pPageKey);
420         if (cont == null) {
421 
422             // Indirect reference.
423             final PageReference reference = prepareLeafOfTree(mNewRoot.getIndirectPageReference(), pPageKey);
424             NodePage page = (NodePage)reference.getPage();
425 
426             if (page == null) {
427                 if (reference.getKey() == IConstants.NULL_ID) {
428                     cont = new NodePageContainer(new NodePage(pPageKey, IConstants.UBP_ROOT_REVISION_NUMBER));
429                 } else {
430                     cont = dereferenceNodePageForModification(pPageKey);
431                 }
432             } else {
433                 cont = new NodePageContainer(page);
434             }
435 
436             reference.setNodePageKey(pPageKey);
437             mLog.put(pPageKey, cont);
438         }
439         mNodePageCon = cont;
440         return cont;
441     }

443     private RevisionRootPage preparePreviousRevisionRootPage(final long pRev,
444         final long pRepresentRev) throws TTException {
445 
446         if (mDelegate.getUberPage().isBootstrap()) {
447             return mDelegate.loadRevRoot(pRev);
448         } else {
449 
450             // Prepare revision root nodePageReference.
451             final RevisionRootPage revisionRootPage =
452                 new RevisionRootPage(mDelegate.loadRevRoot(pRev), pRepresentRev + 1);
453 
454             // Prepare indirect tree to hold reference to prepared revision root
455             // nodePageReference.
456             final PageReference revisionRootPageReference =
457                 prepareLeafOfTree(mDelegate.getUberPage().getIndirectPageReference(), mDelegate.getUberPage()
458                     .getRevisionNumber());
459 
460             // Link the prepared revision root nodePageReference with the
461             // prepared indirect tree.
462             revisionRootPageReference.setPage(revisionRootPage);
463 
464             revisionRootPage.getNamePageReference().setPage(
465                 (NamePage)mDelegate.getActualRevisionRootPage().getNamePageReference().getPage());
466 
467             // Return prepared revision root nodePageReference.
468             return revisionRootPage;
469         }
470     }

472     protected PageReference prepareLeafOfTree(final PageReference pStarRef, final long pKey)
473         throws TTException {
474 
475         // Initial state pointing to the indirect nodePageReference of level 0.
476 
477         PageReference reference = pStarRef;
478         int offset = 0;
479         long levelKey = pKey;
480 
481         // Iterate through all levels.
482         for (int level = 0, height = IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT.length; level < height; level++) {
483             offset = (int)(levelKey >> IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT[level]);
484             levelKey -= offset << IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT[level];
485             final IndirectPage page = prepareIndirectPage(reference);
486             reference = page.getReferences()[offset];
487 
488         }
489 
490         // Return reference to leaf of indirect tree.
491         return reference;
492     }

520     /**
521      * Building name consisting out of prefix and name. NamespaceUri is not used

523      *

525      *            the {@link QName} of an element
526      * @return a string with [prefix:]localname

529         if (pQName == null) {
530             throw new NullPointerException("mQName must not be null!");
531         }
532         String name;
533         if (pQName.getPrefix().isEmpty()) {
534             name = pQName.getLocalPart();
535         } else {
536             name =
537                 new StringBuilder(pQName.getPrefix()).append(":").append(pQName.getLocalPart())
538                     .toString();
539         }
540         return name;
