108     /**
109      * {@inheritDoc}
110      */
111     public long setNode(final INode pNode) throws TTException {
112         checkState(!mDelegate.isClosed(), "Transaction already closed");
113         // Allocate node key and increment node count.
114         final long nodeKey = pNode.getNodeKey();
115         final long seqPageKey = nodeKey >> IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT[3];
116         final int nodePageOffset = nodePageOffset(nodeKey);
117         LogValue container = prepareNodePage(nodeKey);
118         final NodePage page = ((NodePage)container.getModified());
119         page.setNode(nodePageOffset, pNode);
120         mLog.put(new LogKey(false, IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT.length, seqPageKey), container);
121         return nodeKey;
122     }

125      * {@inheritDoc}

127     @Override
128     public void removeNode(final INode pNode) throws TTException {
129         checkState(!mDelegate.isClosed(), "Transaction already closed");
130         checkNotNull(pNode);
131         final long nodePageKey = pNode.getNodeKey() >> IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT[3];
132         LogValue container = prepareNodePage(pNode.getNodeKey());
133         final INode delNode = new DeletedNode(pNode.getNodeKey());
134         ((NodePage)container.getComplete()).setNode(nodePageOffset(pNode.getNodeKey()), delNode);
135         ((NodePage)container.getModified()).setNode(nodePageOffset(pNode.getNodeKey()), delNode);
136 
137         mLog.put(new LogKey(false, IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT.length, nodePageKey), container);
138     }

140     /**
141      * {@inheritDoc}
142      */
143     public INode getNode(final long pNodeKey) throws TTIOException {
144         checkState(!mDelegate.isClosed(), "Transaction already closed");
145         // Calculate page and node part for given nodeKey.
146         final long nodePageKey = pNodeKey >> IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT[3];
147         final int nodePageOffset = nodePageOffset(pNodeKey);
148 
149         final LogValue container =
150             mLog.get(new LogKey(false, IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT.length, nodePageKey));
151         // Page was not modified yet, delegate to read or..
152         if (container == null) {
153             return mDelegate.getNode(pNodeKey);
154         }// ...page was modified, but not this node, take the complete part, or...
155         else if (((NodePage)container.getModified()).getNode(nodePageOffset) == null) {
156             final INode item = ((NodePage)container.getComplete()).getNode(nodePageOffset);
157             return mDelegate.checkItemIfDeleted(item);
158 
159         }// ...page was modified and the modification touched this node.
160         else {
161             final INode item = ((NodePage)container.getModified()).getNode(nodePageOffset);
162             return mDelegate.checkItemIfDeleted(item);
163         }

167     /**
168      * 
169      * {@inheritDoc}
170      */
171     @Override
172     public void commit() throws TTException {
173         checkState(!mDelegate.isClosed(), "Transaction already closed");
174         Iterator<LogValue> entries = mLog.getIterator();
175         while (entries.hasNext()) {
176             LogValue next = entries.next();
177             mPageWriter.write(next.getModified());
178         }
179         mPageWriter.write(mNewMeta);
180         mPageWriter.write(mNewRoot);
181         mPageWriter.writeUberPage(mNewUber);
182 
183         mLog.close();
184 
185         ((Session)mDelegate.mSession).setLastCommittedUberPage(mNewUber);
186         setUpTransaction(mNewUber, mDelegate.mSession, mNewUber.getRevisionNumber(), mPageWriter);
187 
188     }

240     private LogValue prepareNodePage(final long pNodeKey) throws TTException {
241 
242         final long seqNodePageKey = pNodeKey >> IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT[3];
243 
244         LogKey key = new LogKey(false, IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT.length, seqNodePageKey);
245         // See if on nodePageLevel, there are any pages...
246         LogValue container = mLog.get(key);
247         // ... and start dereferencing of not.
248         if (container == null) {
249             LogKey indirectKey = preparePathToLeaf(false, mNewRoot, pNodeKey);
250 
251             LogValue indirectContainer = mLog.get(indirectKey);
252             int nodeOffset = nodePageOffset(seqNodePageKey);
253             long pageKey = ((IndirectPage)indirectContainer.getModified()).getReferenceKeys()[nodeOffset];
254 
255             long newPageKey = mNewUber.incrementPageCounter();
256             if (pageKey != 0) {
257                 NodePage[] pages = mDelegate.getSnapshotPages(seqNodePageKey);
258                 checkState(pages.length > 0);
259                 if (mNewRoot.getRevision()
260                     % Integer.parseInt(mDelegate.mSession.getConfig().mProperties
261                         .getProperty(ConstructorProps.NUMBERTORESTORE)) == 0) {
262                     container =
263                         mDelegate.mSession.getConfig().mRevision.combinePagesForModification(Integer
264                             .parseInt(mDelegate.mSession.getConfig().mProperties
265                                 .getProperty(ConstructorProps.NUMBERTORESTORE)), newPageKey, pages, true);
266                 } else {
267                     container =
268                         mDelegate.mSession.getConfig().mRevision.combinePagesForModification(Integer
269                             .parseInt(mDelegate.mSession.getConfig().mProperties
270                                 .getProperty(ConstructorProps.NUMBERTORESTORE)), newPageKey, pages, false);
271                 }
272             } else {
273                 NodePage newPage = new NodePage(newPageKey, IConstants.NULL_NODE);
274                 container = new LogValue(newPage, newPage);
275             }
276             ((IndirectPage)indirectContainer.getModified()).setReferenceKey(nodeOffset, newPageKey);
277             mLog.put(indirectKey, indirectContainer);
278             mLog.put(key, container);
279         }
280         return container;
281     }

283     /**
284      * Getting a NodePageContainer containing the last IndirectPage with the reference to any new/modified
285      * page.
286      * 
287      * @param pIsRootLevel
288      *            is this dereferencing walk based on the the search after a RevRoot or a NodePage. Needed
289      *            because of the same keys in both subtrees.
290      * @param pPage
291      *            where to start the tree-walk: either from an UberPage (related to new
292      *            RevisionRootPages) or from a RevisionRootPage (related to new NodePages).
293      * @param pSeqPageKey
294      *            the key of the page mapped to the layer
295      * @return the key the container representing the last level
296      * @throws TTException
297      */
298     private LogKey preparePathToLeaf(final boolean pIsRootLevel, final IReferencePage pPage,
299         final long pElementKey) throws TTException {
300 
301         // computing the ordernumbers within all level. The ordernumbers are the position in the sequence of
302         // all pages within the same level.
303         // ranges are for level 0: 0-127; level 1: 0-16383; level 2: 0-2097151; level 3: 0-268435455; ;level
304         // 4: 0-34359738367
305         long seqPageKey = -1;
306         // since the revision points to a page, the sequence-key bases on the last indirect-layer directly
307         // within the search after a revision,...
308         if (pIsRootLevel) {
309             seqPageKey = pElementKey >> IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT[3];
310         } // ...whereas one layer above is used for the nodes based on the offsets pointing to nodes
311           // instead of pages.
312         else {
313             seqPageKey = pElementKey >> IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT[2];
314         }
315 
316         long[] orderNumber = new long[IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT.length];
317         for (int level = 0; level < orderNumber.length; level++) {
318             orderNumber[level] = seqPageKey >> IConstants.INP_LEVEL_PAGE_COUNT_EXPONENT[level];
319         }
320 
321         IReferencePage page = null;
322         IReferencePage parentPage = pPage;
323         LogKey key = null;
324         LogKey parentKey = new LogKey(pIsRootLevel, -1, 0);
325 
326         // Iterate through all levels...
327         for (int level = 0; level < orderNumber.length; level++) {
328             // ...see if the actual page requested is already in the log
329             key = new LogKey(pIsRootLevel, level, orderNumber[level]);
330             LogValue container = mLog.get(key);
331             // if the page is not existing,..
332             if (container == null) {
333                 // ..create a new page
334                 final long newKey = mNewUber.incrementPageCounter();
335                 page = new IndirectPage(newKey);
336 
337                 // compute the offset of the newpage
338                 int offset = nodePageOffset(orderNumber[level]);
339 
340                 // if there existed the same page in former versions (referencable over the offset within
341                 // the parent)...
342                 if (parentPage.getReferenceKeys()[offset] != 0) {
343                     IReferencePage oldPage =
344                         (IReferencePage)mPageWriter.read(parentPage.getReferenceKeys()[offset]);
345                     for (int i = 0; i < oldPage.getReferenceKeys().length; i++) {
346                         page.setReferenceKey(i, oldPage.getReferenceKeys()[i]);
347                     }
348                 }
349                 // Set the newKey on the computed offset
350                 parentPage.setReferenceKey(offset, newKey);
351                 // .. and put the parent-reference to the log as well as the reference of the..
352                 container = new LogValue(parentPage, parentPage);
353                 mLog.put(parentKey, container);
354                 // ...current page.
355                 container = new LogValue(page, page);
356                 mLog.put(key, container);
357 
358             } // if the page is already in the log, get it simply from the log.
359             else {
360                 page = (IReferencePage)container.getModified();
361             }
362             // finally, set the new pagekey for the next level
363             parentKey = key;
364             parentPage = page;
365         }
366 
367         // Return reference to leaf of indirect tree.
368         return key;
369     }

371     private void setUpTransaction(final UberPage pUberPage, final ISession pSession,
372         final long pRepresentRev, final IBackendWriter pWriter) throws TTException {
373 
374         mLog =
375             new LRULog(new File(pSession.getConfig().mProperties
376                 .getProperty(org.treetank.access.conf.ConstructorProps.RESOURCEPATH)),
377                 pSession.getConfig().mNodeFac, pSession.getConfig().mMetaFac);
378 
379         mNewUber =
380             new UberPage(pUberPage.incrementPageCounter(), pUberPage.getRevisionNumber() + 1, pUberPage
381                 .getPageCounter());
382         mNewUber.setReferenceKey(IReferencePage.GUARANTEED_INDIRECT_OFFSET,
383             pUberPage.getReferenceKeys()[IReferencePage.GUARANTEED_INDIRECT_OFFSET]);
384 
385         mDelegate = new PageReadTrx(pSession, pUberPage, pRepresentRev, pWriter);
386 
387         // Get previous revision root page..
388         final RevisionRootPage previousRevRoot = mDelegate.mRootPage;
389         // ...and using this data to initialize a fresh revision root including the pointers.
390         mNewRoot =
391             new RevisionRootPage(mNewUber.incrementPageCounter(), pRepresentRev + 1, previousRevRoot
392                 .getMaxNodeKey());
393         mNewRoot.setReferenceKey(IReferencePage.GUARANTEED_INDIRECT_OFFSET, previousRevRoot
394             .getReferenceKeys()[IReferencePage.GUARANTEED_INDIRECT_OFFSET]);
395 
396         // Prepare indirect tree to hold reference to prepared revision root
397         // nodePageReference.
398         LogKey indirectKey = preparePathToLeaf(true, mNewUber, mNewUber.getRevisionNumber());
399         LogValue indirectContainer = mLog.get(indirectKey);
400         int offset = nodePageOffset(mNewUber.getRevisionNumber());
401         ((IndirectPage)indirectContainer.getModified()).setReferenceKey(offset, mNewRoot.getPageKey());
402         mLog.put(indirectKey, indirectContainer);
403 
404         // Setting up a new metapage
405         Map<IMetaEntry, IMetaEntry> oldMap = mDelegate.mMetaPage.getMetaMap();
406         mNewMeta = new MetaPage(mNewUber.incrementPageCounter());
407 
408         for (IMetaEntry key : oldMap.keySet()) {
409             mNewMeta.setEntry(key, oldMap.get(key));
410         }
411 
412         mNewRoot.setReferenceKey(RevisionRootPage.NAME_REFERENCE_OFFSET, mNewMeta.getPageKey());
413 
414     }
