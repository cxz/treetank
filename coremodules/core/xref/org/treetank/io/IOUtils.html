21      /**
22       * Creating a folder structure based on a set of paths given as parameter and returning a boolean
23       * determining the success.
24       * 
25       * @param pFile
26       *            the root folder where the configuration should be created to.
27       * @param pPaths
28       *            to be created
29       * @return true if creations was successful, false otherwise.
30       * @throws TTIOException
31       */
32      public static boolean createFolderStructure(final File pFile, IConfigurationPath[] pPaths)
33          throws TTIOException {
34          boolean returnVal = true;
35          pFile.mkdirs();
36          // creation of folder structure
37          for (IConfigurationPath paths : pPaths) {
38              final File toCreate = new File(pFile, paths.getFile().getName());
39              if (paths.isFolder()) {
40                  returnVal = toCreate.mkdir();
41              } else {
42                  try {
43                      returnVal = toCreate.createNewFile();
44                  } catch (final IOException exc) {
45                      throw new TTIOException(exc);
46                  }
47              }
48              if (!returnVal) {
49                  break;
50              }
51          }
52          return returnVal;
53      }

55      /**
56       * Checking a structure in a folder to be equal with the data in this
57       * enum.
58       * 
59       * @param pFile
60       *            to be checked
61       * @param pPaths
62       *            containing the elements to be checked against
63       * @return -1 if less folders are there, 0 if the structure is equal to
64       *         the one expected, 1 if the structure has more folders
65       */
66      public static int compareStructure(final File pFile, IConfigurationPath[] pPaths) {
67          int existing = 0;
68          for (final IConfigurationPath path : pPaths) {
69              final File currentFile = new File(pFile, path.getFile().getName());
70              if (currentFile.exists()) {
71                  existing++;
72              }
73          }
74          return existing - pPaths.length;
75      }

77      /**
78       * Deleting a storage recursive. Used for deleting a databases
79       * 
80       * @param pFile
81       *            which should be deleted included descendants
82       * @return true if delete is valid
83       */
84      public static boolean recursiveDelete(final File pFile) {
85          if (pFile.isDirectory()) {
86              for (final File child : pFile.listFiles()) {
87                  if (!recursiveDelete(child)) {
88                      return false;
89                  }
90              }
91          }
92          return pFile.delete();
93      }
