29  public class JCloudsReader implements IBackendReader {
30  
31      /** Blob Store for Reading the data. */
32      protected final BlobStore mBlobStore;
33  
34      /** Factory for building Pages. */
35      private final PageFactory mFac;
36  
37      /** Inflater to decompress. */
38      protected final IByteHandlerPipeline mByteHandler;
39  
40      /** Resource name of this container and the database. */
41      protected final String mResourceName;
42  
43      public JCloudsReader(BlobStore pBlobStore, PageFactory pFac, IByteHandlerPipeline pByteHandler,
44          String pResourceName) throws TTException {
45          mBlobStore = pBlobStore;
46          mByteHandler = pByteHandler;
47          mFac = pFac;
48          mResourceName = pResourceName;
49      }
50  
51      /**
52       * {@inheritDoc}
53       */
54      @Override
55      public PageReference readFirstReference() throws TTIOException, TTByteHandleException {
56          try {
57              final PageReference uberPageReference = new PageReference();
58              Blob blobRetrieved = mBlobStore.getBlob(mResourceName, Long.toString(-1l));
59              InputStream in = blobRetrieved.getPayload().getInput();
60              DataInputStream datain = new DataInputStream(in);
61              long uberpagekey = datain.readLong();
62              uberPageReference.setKey(uberpagekey);
63              final UberPage page = (UberPage)read(uberPageReference.getKey());
64              uberPageReference.setPage(page);
65              datain.close();
66              in.close();
67              return uberPageReference;
68          } catch (final IOException exc) {
69              throw new TTIOException(exc);
70          }
71  
72      }
73  
74      /**
75       * {@inheritDoc}
76       * 
77       * @throws
78       */
79      @Override
80      public IPage read(long pKey) throws TTIOException {
81          try {
82              Blob blobRetrieved = mBlobStore.getBlob(mResourceName, Long.toString(pKey));
83              InputStream in = blobRetrieved.getPayload().getInput();
84              ByteArrayOutputStream out = new ByteArrayOutputStream();
85              ByteStreams.copy(in, out);
86              byte[] decryptedPage = mByteHandler.deserialize(out.toByteArray());
87              out.close();
88              in.close();
89              return mFac.deserializePage(decryptedPage);
90          } catch (final IOException | TTByteHandleException exc) {
91              throw new TTIOException(exc);
92          }
93      }

100     }
