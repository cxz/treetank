69  /**
70   * Factory class to build up {@link IBackendReader} {@link IBackendWriter} instances for the
71   * Treetank Framework.
72   * 
73   * After all this class is implemented as a Singleton to hold one {@link BerkeleyStorage} per
74   * {@link SessionConfiguration}.
75   * 
76   * @author Sebastian Graf, University of Konstanz
77   * 
78   */
79  public final class BerkeleyStorage implements IBackend {
80  
81      /** Name for the database. */
82      private static final String NAME = "berkeleyDatabase";
83  
84      /** Berkeley Environment for the database. */
85      private Environment mEnv;
86  
87      /** Storage instance per session. */
88      private Database mDatabase;
89  
90      /** Binding for de/-serializing buckets. */
91      private final TupleBinding<IBucket> mBucketBinding;
92  
93      /** Handling the byte-representation before serialization. */
94      private final IByteHandlerPipeline mByteHandler;
95  
96      /** Factory for Buckets. */
97      private final BucketFactory mFac;
98  
99      /** File for DB. */
100     private final File mFile;
101 
102     /**
103      * Simple constructor.
104      * 
105      * @param pProperties
106      *            not only the file associated with the database
107      * @param pNodeFac
108      *            factory for the nodes
109      * @param pMetaFac
110      *            factory for meta bucket
111      * @param pByteHandler
112      *            handling any bytes
113      * @throws TTIOException
114      *             of something odd happens while database-connection
115      */
116     @Inject
117     public BerkeleyStorage(@Assisted Properties pProperties, INodeFactory pNodeFac,
118         IMetaEntryFactory pMetaFac, IByteHandlerPipeline pByteHandler) throws TTIOException {
119 
120         mFile =
121             new File(pProperties.getProperty(ConstructorProps.RESOURCEPATH), ResourceConfiguration.Paths.Data
122                 .getFile().getName());
123 
124         mBucketBinding = new BucketBinding();
125         mByteHandler = pByteHandler;
126         mFac = new BucketFactory(pNodeFac, pMetaFac);
127 
128     }
129 
130     /**
131      * {@inheritDoc}
132      */
133     @Override
134     public void initialize() throws TTIOException {
135         try {
136             EnvironmentConfig config = new EnvironmentConfig();
137             config = config.setSharedCache(true);
138             config.setAllowCreate(true);
139             config.setTransactional(true);
140             config.setCachePercent(20);
141             mEnv = new Environment(mFile, config);
142 
143             final DatabaseConfig conf = new DatabaseConfig();
144             conf.setAllowCreate(true);
145             conf.setTransactional(true);
146 
147             mDatabase = mEnv.openDatabase(null, NAME, conf);
148 
149         } catch (final DatabaseException exc) {
150             throw new TTIOException(exc);
151         }
152     }
153 
154     /**
155      * {@inheritDoc}
156      */
157     @Override
158     public synchronized IBackendReader getReader() throws TTIOException {
159         try {
160             return new BerkeleyReader(mEnv, mDatabase, mBucketBinding);
161         } catch (final DatabaseException exc) {
162             throw new TTIOException(exc);
163         }
164     }
165 
166     /**
167      * {@inheritDoc}
168      */
169     @Override
170     public synchronized IBackendWriter getWriter() throws TTIOException {
171         return new BerkeleyWriter(mEnv, mDatabase, mBucketBinding);
172     }
173 
174     /**
175      * {@inheritDoc}
176      */
177     @Override
178     public synchronized void close() throws TTIOException {
179         try {
180             mEnv.sync();
181             mDatabase.close();
182             mEnv.close();
183         } catch (final DatabaseException exc) {
184             throw new TTIOException(exc);
185         }
186     }
187 
188     @Override
189     public IByteHandlerPipeline getByteHandler() {
190         return mByteHandler;
191     }
192 
193     /**
194      * {@inheritDoc}
195      */
196     @Override
197     public synchronized boolean truncate() throws TTException {
198         mDatabase.close();
199         if (mEnv.getDatabaseNames().contains(NAME)) {
200             mEnv.removeDatabase(null, NAME);
201         }
202         mEnv.close();
203         return IOUtils.recursiveDelete(mFile);
204     }
205 
206     /**
207      * {@inheritDoc}
208      */
209     @Override
210     public String toString() {
211         return toStringHelper(this).add("mByteHandler", mByteHandler).add("mFile", mFile).add("mFac", mFac)
212             .toString();
213     }

215     /**
216      * Binding for storing {@link IBucket} objects within the Berkeley DB.
217      * 
218      * @author Sebastian Graf, University of Konstanz
219      * 
220      */
221     class BucketBinding extends TupleBinding<IBucket>{
222 
223         /**
224          * {@inheritDoc}
225          */
226         @Override
227         public IBucket entryToObject(final TupleInput arg0) {
228             try {
229                 final InputStream handledStream = mByteHandler.deserialize(arg0);
230                 final DataInput in = new DataInputStream(handledStream);
231                 final IBucket returnVal = mFac.deserializeBucket(in);
232                 handledStream.close();
233                 arg0.close();
234                 return returnVal;
235             } catch (IOException | TTException exc) {
236                 throw new RuntimeException(exc);
237             }
238         }
239 
240         /**
241          * {@inheritDoc}
242          */
243         @Override
244         public void objectToEntry(final IBucket arg0, final TupleOutput arg1) {
245             try {
246                 final OutputStream handledStream = mByteHandler.serialize(arg1);
247                 final DataOutput output = new DataOutputStream(handledStream);
248                 arg0.serialize(output);
249                 handledStream.close();
250                 
251                 arg1.close();
252             } catch (IOException | TTException exc) {
253                 throw new RuntimeException(exc);
254             }
255         }
256     }
