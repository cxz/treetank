30  /**
31   * Holding all byte representations for building up a XML.
32   * 
33   * @author Sebastian Graf, University of Konstanz
34   * 
35   */
36  public enum ECharsForSerializing {
37  
38      /** " ". */
39      SPACE(new byte[] {
40          32
41      }),
42  
43      /** "&lt;". */
44      OPEN(new byte[] {
45          60
46      }),
47  
48      /** "&gt;". */
49      CLOSE(new byte[] {
50          62
51      }),
52  
53      /** "/". */
54      SLASH(new byte[] {
55          47
56      }),
57  
58      /** "=". */
59      EQUAL(new byte[] {
60          61
61      }),
62  
63      /** "\"". */
64      QUOTE(new byte[] {
65          34
66      }),
67  
68      /** "=\"". */
69      EQUAL_QUOTE(EQUAL.getBytes(), QUOTE.getBytes()),
70  
71      /** "&lt;/". */
72      OPEN_SLASH(OPEN.getBytes(), SLASH.getBytes()),
73  
74      /** "/&gt;". */
75      SLASH_CLOSE(SLASH.getBytes(), CLOSE.getBytes()),
76  
77      /** " rest:"". */
78      REST_PREFIX(SPACE.getBytes(), new byte[] {
79          114, 101, 115, 116, 58
80      }),
81  
82      /** "ttid". */
83      ID(new byte[] {
84          116, 116, 105, 100
85      }),
86  
87      /** " xmlns=\"". */
88      XMLNS(SPACE.getBytes(), new byte[] {
89          120, 109, 108, 110, 115
90      }, EQUAL.getBytes(), QUOTE.getBytes()),
91  
92      /** " xmlns:". */
93      XMLNS_COLON(SPACE.getBytes(), new byte[] {
94          120, 109, 108, 110, 115, 58
95      }),
96  
97      /** Newline. */
98      NEWLINE(System.getProperty("line.separator").getBytes());
99  
100     /** Getting the bytes for the char. */
101     private final byte[] mBytes;
102 
103     /**
104      * Private constructor.
105      * 
106      * @param paramBytes
107      *            the bytes for the chars
108      */
109     ECharsForSerializing(final byte[]... paramBytes) {
110         int index = 0;
111         for (final byte[] runner : paramBytes) {
112             index = index + runner.length;
113         }
114         this.mBytes = new byte[index];
115         index = 0;
116         for (final byte[] runner : paramBytes) {
117             System.arraycopy(runner, 0, mBytes, index, runner.length);
118             index = index + runner.length;
119         }
120     }
121 
122     /**
123      * Getting the bytes.
124      * 
125      * @return the bytes for the char.
126      */
127     public byte[] getBytes() {
128         return mBytes;
129     }
