314     @Test
315     public final void testFacets() {
316         assertEquals(true, string.facetIsSatisfiedBy("hallo welt!"));
317         assertEquals(true, string.facetIsSatisfiedBy("r7321741237r8gruqewfgducnb2138"));
318         assertEquals(true, string.facetIsSatisfiedBy("-12.E24"));
319         assertEquals(true, string.facetIsSatisfiedBy("&%)=1"));
320         assertEquals(true, string.facetIsSatisfiedBy("\""));
321 
322         assertEquals(true, integerT.facetIsSatisfiedBy("12345"));
323         assertEquals(true, integerT.facetIsSatisfiedBy("-12345"));
324         assertEquals(false, integerT.facetIsSatisfiedBy("123-45"));
325         assertEquals(false, integerT.facetIsSatisfiedBy("1234.5"));
326 
327         assertEquals(true, floatT.facetIsSatisfiedBy(".12345"));
328         assertEquals(true, floatT.facetIsSatisfiedBy("-.12345"));
329         assertEquals(true, floatT.facetIsSatisfiedBy("123E-45"));
330         assertEquals(true, floatT.facetIsSatisfiedBy("1234.5"));
331 
332         assertEquals(true, doubleT.facetIsSatisfiedBy(".12345"));
333         assertEquals(true, doubleT.facetIsSatisfiedBy("-.12345"));
334         assertEquals(true, doubleT.facetIsSatisfiedBy("123E-45"));
335         assertEquals(false, doubleT.facetIsSatisfiedBy("Hallo"));
336 
337         assertEquals(true, bool.facetIsSatisfiedBy("1"));
338         assertEquals(false, bool.facetIsSatisfiedBy("2"));
339         assertEquals(true, bool.facetIsSatisfiedBy("0"));
340         assertEquals(true, bool.facetIsSatisfiedBy("true"));
341     }
