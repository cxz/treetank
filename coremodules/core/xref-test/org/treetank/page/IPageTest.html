39       *            IPage as class
40       * @param pPages
41       *            different pages
42       * @throws TTIOException
43       */
44      @Test(dataProvider = "instantiatePages")
45      public void testByteRepresentation(Class<IPage> clazz, IPage[] pPages) throws TTIOException {
46          final PageFactory fac = new PageFactory(new DumbNodeFactory(), new DumbMetaEntryFactory());
47  
48          for (final IPage page : pPages) {
49              ByteArrayDataOutput output = ByteStreams.newDataOutput();
50              page.serialize(output);
51              byte[] firstSerialized = output.toByteArray();
52              ByteArrayDataInput input = ByteStreams.newDataInput(firstSerialized);
53  
54              final IPage serializedPage = fac.deserializePage(input);
55              output = ByteStreams.newDataOutput();
56              serializedPage.serialize(output);
57              byte[] secondSerialized = output.toByteArray();
58              assertTrue(new StringBuilder("Check for ").append(page.getClass()).append(" failed.").toString(),
59                  Arrays.equals(firstSerialized, secondSerialized));
60  
61          }
62      }

74      @Test(dataProvider = "instantiatePages")
75      public void testEqualsAndHashCode(Class<IPage> clazz, IPage[] pPages) throws TTIOException {
76  
77          for (int i = 0; i < pPages.length; i++) {
78              IPage onePage = pPages[i % pPages.length];
79              IPage secondPage = pPages[(i + 1) % pPages.length];
80              assertEquals(onePage.hashCode(), onePage.hashCode());
81              assertNotSame(onePage.hashCode(), secondPage.hashCode());
82              assertTrue(onePage.equals(onePage));
83              assertFalse(onePage.equals(secondPage));
84          }
85  
86      }
