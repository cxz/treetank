109         // creating a new resource, successful
110         assertTrue(mStorage.createResource(mResource));
111         // check if existing, successful
112         assertTrue(mStorage.existsResource(mResource.mProperties.getProperty(ConstructorProps.RESOURCE)));
113         // creating the same resource, unsuccessful
114         assertFalse(mStorage.createResource(mResource));
115         // getting the session
116         ISession session =
117             mStorage.getSession(new SessionConfiguration(mResource.mProperties
118                 .getProperty(ConstructorProps.RESOURCE), null));
119         try {
120             // try to truncate the resource
121             mStorage.truncateResource(new SessionConfiguration(mResource.mProperties
122                 .getProperty(ConstructorProps.RESOURCE), null));
123             fail();
124         } catch (IllegalStateException exc) {
125             // must be thrown
126         }
127         assertTrue(session.close());
128         // truncating resource
129         assertTrue(mStorage.truncateResource(new SessionConfiguration(mResource.mProperties
130             .getProperty(ConstructorProps.RESOURCE), null)));
131         // trying to truncate not existing resource
132         assertFalse(mStorage.truncateResource(new SessionConfiguration("notExistingResource", null)));
133         // check against resource, should not exist
134         assertFalse(mStorage.existsResource(mResource.mProperties.getProperty(ConstructorProps.RESOURCE)));

136         assertTrue(mStorage.createResource(mResource));
137         // check if existing, successful
