122      * {@link org.treetank.access.BucketReadTrx#dereferenceLeafOfTree(org.treetank.io.IBackendReader, long, long)}
123      * .
124      */
125     @Test
126     public void testDereferenceLeafOfTree() throws TTIOException {
127         int[] offsets = new int[5];
128 
129         IBackendReader reader = getFakedStructure(offsets);
130 
131         long key = BucketReadTrx.dereferenceLeafOfTree(reader, 1, 0);
132         // 6 is base key because of 5 layers plus the 1 as start key
133         assertEquals(6, key);
134 
135         offsets[4] = 127;
136         reader = getFakedStructure(offsets);
137         key = BucketReadTrx.dereferenceLeafOfTree(reader, 1, 127);
138         // 6 as base plus 127 as offset on last bucket
139         assertEquals(133, key);
140 
141         offsets[3] = 1;
142         offsets[4] = 0;
143         reader = getFakedStructure(offsets);
144         key = BucketReadTrx.dereferenceLeafOfTree(reader, 1, 128);
145         // 6 as base plus one additional offset on one level above
146         assertEquals(7, key);
147 
148         offsets[3] = 127;
149         offsets[4] = 127;
150         reader = getFakedStructure(offsets);
151         key = BucketReadTrx.dereferenceLeafOfTree(reader, 1, 16383);
152         // 6 as base plus two times 127 as offsets on level above
153         assertEquals(260, key);
154 
155         offsets[2] = 1;
156         offsets[3] = 0;
157         offsets[4] = 0;
158         reader = getFakedStructure(offsets);
159         key = BucketReadTrx.dereferenceLeafOfTree(reader, 1, 16384);
160         // 6 as base plus one additional offset on two levels above
161         assertEquals(7, key);
162 
163         offsets[2] = 127;
164         offsets[3] = 127;
165         offsets[4] = 127;
166         reader = getFakedStructure(offsets);
167         key = BucketReadTrx.dereferenceLeafOfTree(reader, 1, 2097151);
168         // 6 as base plus three times 127 as offsets on levels above
169         assertEquals(387, key);
170 
171         offsets[1] = 1;
172         offsets[2] = 0;
173         offsets[3] = 0;
174         offsets[4] = 0;
175         reader = getFakedStructure(offsets);
176         key = BucketReadTrx.dereferenceLeafOfTree(reader, 1, 2097152);
177         // 6 as base plus one additional offset on three levels above
178         assertEquals(7, key);
179 
180         offsets[1] = 127;
181         offsets[2] = 127;
182         offsets[3] = 127;
183         offsets[4] = 127;
184         reader = getFakedStructure(offsets);
185         key = BucketReadTrx.dereferenceLeafOfTree(reader, 1, 268435455);
186         // 6 as base plus four times 127 as offsets on levels above
187         assertEquals(514, key);
188 
189         offsets[0] = 1;
190         offsets[1] = 0;
191         offsets[2] = 0;
192         offsets[3] = 0;
193         offsets[4] = 0;
194         reader = getFakedStructure(offsets);
195         key = BucketReadTrx.dereferenceLeafOfTree(reader, 1, 268435456);
196         // 6 as base plus one additional offset on three levels above
197         assertEquals(7, key);
198 
199         offsets[0] = 127;
200         offsets[1] = 127;
201         offsets[2] = 127;
202         offsets[3] = 127;
203         offsets[4] = 127;
204         reader = getFakedStructure(offsets);
205         key = BucketReadTrx.dereferenceLeafOfTree(reader, 1, 34359738367l);
206         // 6 as base plus five times 127 as offsets on levels above
207         assertEquals(641, key);
208 
209         // false offset, not existing
210         offsets[0] = 0;
211         reader = getFakedStructure(offsets);
212         key = BucketReadTrx.dereferenceLeafOfTree(reader, 1, 34359738367l);
213         assertEquals(-1, key);
214     }

240     protected static void testGet(final ISession pSession, final DumbNode[][] pNodes) throws TTException {
241         // check against invalid nodekey

270     protected static void
271         testClose(final IStorage pStorage, final ISession pSession, final IBucketReadTrx pRtx)
272             throws TTException {
273 
274         IBucketReadTrx rtx = pRtx;
275 
276         // explicit closing of one transaction
277         rtx.getMetaBucket();
278         assertFalse(rtx.isClosed());
279         assertTrue(rtx.close());
280         try {
281             rtx.getMetaBucket();
282             fail();
283         } catch (IllegalStateException exc) {
284             // must be thrown
285         }
286         try {
287             rtx.getRevision();
288             fail();
289         } catch (IllegalStateException exc) {
290             // must be thrown
291         }
292         try {
293             rtx.getNode(0);
294             fail();
295         } catch (IllegalStateException exc) {
296             // must be thrown
297         }
298 
299         assertFalse(rtx.close());
300         assertTrue(rtx.isClosed());
301 
302         // implicit closing over session
303         rtx = pSession.beginBucketRtx(pSession.getMostRecentVersion());
304         assertFalse(rtx.isClosed());
305         assertTrue(pSession.close());
306         assertFalse(rtx.close());
307         assertTrue(rtx.isClosed());
308 
309         // implicit closing over storage
310         ISession session = pStorage.getSession(new SessionConfiguration(CoreTestHelper.RESOURCENAME, null));
311         rtx = session.beginBucketRtx(session.getMostRecentVersion());
312         assertFalse(rtx.isClosed());
313         assertTrue(pStorage.close());
314         assertFalse(rtx.close());
315         assertTrue(rtx.isClosed());
316         assertFalse(session.close());
317     }
