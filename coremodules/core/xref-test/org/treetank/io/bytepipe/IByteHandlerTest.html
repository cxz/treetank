34      /**
35       * Test method for {@link org.treetank.io.bytepipe.IByteHandler#deserialize(InputStream)} and for
36       * {@link org.treetank.io.bytepipe.IByteHandler#serialize(OutputStream)}.
37       * 
38       * @throws TTByteHandleException
39       * @throws IOException
40       */
41      @Test(dataProvider = "instantiateByteHandler")
42      public void testSerializeAndDeserialize(Class<IByteHandler> clazz, IByteHandler[] pHandlers)
43          throws TTByteHandleException, IOException {
44          for (final IByteHandler handler : pHandlers) {
45              final int datasize = 10000;
46              final byte[] bytes = CoreTestHelper.generateRandomBytes(datasize);
47  
48              ByteArrayOutputStream output = new ByteArrayOutputStream();
49              OutputStream handledOutout = handler.serialize(output);
50  
51              ByteArrayInputStream input = new ByteArrayInputStream(bytes);
52              ByteStreams.copy(input, handledOutout);
53              output.close();
54              handledOutout.close();
55              input.close();
56  
57              final byte[] encoded = output.toByteArray();
58              assertFalse(new StringBuilder("Check for ").append(handler.getClass()).append(" failed.")
59                  .toString(), Arrays.equals(bytes, encoded));
60  
61              input = new ByteArrayInputStream(encoded);
62              InputStream handledInput = handler.deserialize(input);
63              output = new ByteArrayOutputStream();
64              ByteStreams.copy(handledInput, output);
65              output.close();
66              handledInput.close();
67              input.close();
68  
69              final byte[] decoded = output.toByteArray();
70  
71              assertTrue(new StringBuilder("Check for ").append(handler.getClass()).append(" failed.")
72                  .toString(), Arrays.equals(bytes, decoded));
73          }
74      }
