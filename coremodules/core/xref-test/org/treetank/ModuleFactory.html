19  /**
20   * Module Factory for initializing the modules in correct order depending on the
21   * context. Main point for the orthogonal test setup.
22   * 
23   * @author Sebastian Graf, University of Konstanz
24   * 
25   */
26  public class ModuleFactory implements IModuleFactory {
27  
28      private final static String DATAFACTORYPARAMETER = "DataFactory";
29      private final static String METAFACTORYPARAMETER = "MetaFactory";
30      private final static String REVISIONINGPARAMETER = "Revisioning";
31      private final static String BACKENDPARAMETER = "Backend";
32  
33      // Standard Implementations
34      private final static String REVISIONING = SlidingSnapshot.class.getName();
35      private final static String BACKEND = JCloudsStorage.class.getName();
36  
37      /**
38       * {@inheritDoc}
39       */
40      @Override
41      @SuppressWarnings("unchecked")
42      public Module createModule(ITestContext context, Class<?> testClass) {
43          AbstractModule returnVal = null;
44          String dataFacName;
45          String metaFacName;
46          String revisioningName = REVISIONING;
47          String backendName = BACKEND;
48          // getting the parameters over testng.xml and setting it directly or...
49          if (context.getSuite().getParameter(DATAFACTORYPARAMETER) != null) {
50              final Map<String, String> params = context.getSuite().getXmlSuite().getAllParameters();
51              dataFacName = params.get(DATAFACTORYPARAMETER);
52              metaFacName = params.get(METAFACTORYPARAMETER);
53              revisioningName = params.get(REVISIONINGPARAMETER);
54              backendName = params.get(BACKENDPARAMETER);
55          }// ..determining standard factories based on bundle parsed from the testclass and...
56          else {
57              final String[] elements =
58                  testClass.getProtectionDomain().getCodeSource().getLocation().toString().split(
59                      Pattern.quote("/"));
60              String module = elements[elements.length - 3];
61              switch (module) {
62              case "core":
63                  dataFacName = "org.treetank.bucket.DumbDataFactory";
64                  metaFacName = "org.treetank.bucket.DumbMetaEntryFactory";
65                  break;
66              case "node":
67              case "xml":
68              case "saxon":
69              case "jax-rx":
70                  dataFacName = "org.treetank.node.TreeNodeFactory";
71                  metaFacName = "org.treetank.node.NodeMetaPageFactory";
72                  break;
73              case "iscsi":
74                  dataFacName = "org.treetank.data.BlockDataElementFactory";
75                  metaFacName = "org.treetank.data.ISCSIMetaPageFactory";
76                  break;
77              case "filelistener":
78                  dataFacName = "org.treetank.filelistener.file.data.FileDataFactory";
79                  metaFacName = "org.treetank.filelistener.file.data.FilelistenerMetaDataFactory";
80                  break;
81              default:
82                  throw new IllegalStateException("Suitable module not found");
83              }
84  
85          }
86          // ...invoking it over reflection and setting it to the ModuleSetter.
87          Class<IDataFactory> dataFac;
88          Class<IMetaEntryFactory> metaFac;
89          Class<IRevisioning> revisioning;
90          Class<IBackend> backend;
91          try {
92              dataFac = (Class<IDataFactory>)Class.forName(dataFacName);
93              metaFac = (Class<IMetaEntryFactory>)Class.forName(metaFacName);
94              revisioning = (Class<IRevisioning>)Class.forName(revisioningName);
95              backend = (Class<IBackend>)Class.forName(backendName);
96          } catch (ClassNotFoundException exc) {
97              throw new RuntimeException(exc);
98          }
99  
100         returnVal =
101             new ModuleSetter().setDataFacClass(dataFac).setMetaFacClass(metaFac)
102                 .setRevisioningClass(revisioning).setBackendClass(backend).createModule();
103         
104         return returnVal;
105 
106     }
