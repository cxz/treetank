125     /**
126      * Getting a database and create one of not existing. This includes the
127      * creation of a resource with the settings in the builder as standard.
128      * 
129      * @param file
130      *            to be created
131      * @return a database-obj
132      */
133     @Ignore
134     public static final IDatabase getDatabase(final File file) {
135         if (INSTANCES.containsKey(file)) {
136             return INSTANCES.get(file);
137         } else {
138             try {
139                 final DatabaseConfiguration config = new DatabaseConfiguration(file);
140                 if (!file.exists()) {
141                     Database.createDatabase(config);
142                 }
143                 final IDatabase database = Database.openDatabase(file);
144                 database.createResource(new ResourceConfiguration.Builder(RESOURCE, config).build());
145                 INSTANCES.put(file, database);
146                 return database;
147             } catch (final AbsTTException exc) {
148                 fail(exc.toString());
149                 return null;
150             }
151         }
152     }

242     /**
243      * Read a file into a StringBuilder.
244      * 
245      * @param paramFile
246      *            The file to read.
247      * @param paramWhitespaces
248      *            Retrieve file and don't remove any whitespaces.
249      * @return StringBuilder instance, which has the string representation of
250      *         the document.
251      * @throws IOException
252      *             throws an IOException if any I/O operation fails.
253      */
254     @Ignore("Not a test, utility method only")
255     public static StringBuilder readFile(final File paramFile, final boolean paramWhitespaces)
256         throws IOException {
257         final BufferedReader in = new BufferedReader(new FileReader(paramFile));
258         final StringBuilder sBuilder = new StringBuilder();
259         for (String line = in.readLine(); line != null; line = in.readLine()) {
260             if (paramWhitespaces) {
261                 sBuilder.append(line + ECharsForSerializing.NEWLINE);
262             } else {
263                 sBuilder.append(line.trim());
264             }
265         }
266 
267         // Remove last newline.
268         if (paramWhitespaces) {
269             sBuilder.replace(sBuilder.length() - 1, sBuilder.length(), "");
270         }
271         in.close();
272 
273         return sBuilder;
274     }

276     /**
277      * Creating a test document at {@link PATHS#PATH1}.
278      * 
279      * @throws AbsTTException
280      */
281     public static void createTestDocument() throws AbsTTException {
282         final IDatabase database = TestHelper.getDatabase(PATHS.PATH1.getFile());
283         database.createResource(new ResourceConfiguration.Builder(RESOURCE, PATHS.PATH1.config).build());
284         final ISession session = database.getSession(new SessionConfiguration.Builder(RESOURCE).build());
285         final IPageWriteTrx pWtx = session.beginPageWriteTransaction();
286         final INodeWriteTrx nWtx = new NodeWriteTrx(session, pWtx);
287         DocumentCreater.create(nWtx);
288         nWtx.commit();
289         nWtx.close();
290         session.close();
291     }
