49   * Helper class for offering convenient usage of {@link Session}s for test

57  public final class TestHelper {

62      /** Paths where the data is stored to. */
63      public enum PATHS {
64  
65          // PATH1
66              PATH1(new File(new StringBuilder(Files.createTempDir().getAbsolutePath()).append(File.separator)
67                  .append("tnk").append(File.separator).append("path1").toString())),
68  
69              // PATH2
70              PATH2(new File(new StringBuilder(Files.createTempDir().getAbsolutePath()).append(File.separator)
71                  .append("tnk").append(File.separator).append("path2").toString()));
72  
73          final File file;
74  
75          final DatabaseConfiguration config;
76  
77          PATHS(final File paramFile) {
78              file = paramFile;
79              config = new DatabaseConfiguration(paramFile);
80          }
81  
82          public File getFile() {
83              return file;
84          }
85  
86          public DatabaseConfiguration getConfig() {
87              return config;
88          }
89  
90      }

92      /** Common random instance for generating common tag names. */
93      public final static Random random = new Random();
94  
95      private final static Map<File, IDatabase> INSTANCES = new Hashtable<File, IDatabase>();
96  
97      /**
98       * Getting a database and create one of not existing. This includes the
99       * creation of a resource with the settings in the builder as standard.
100      * 
101      * @param file
102      *            to be created
103      * @return a database-obj
104      * @throws TTException
105      */
106     public static final IDatabase getDatabase(final File file) throws TTException {
107         if (INSTANCES.containsKey(file)) {
108             return INSTANCES.get(file);
109         } else {
110             final DatabaseConfiguration config = new DatabaseConfiguration(file);
111             if (!file.exists()) {
112                 Database.createDatabase(config);
113             }
114             final IDatabase database = Database.openDatabase(file);
115             INSTANCES.put(file, database);
116             return database;
117         }
118     }
119 
120     public static final boolean createResource(final ResourceConfiguration resConf) throws TTException {
121         final IDatabase database = TestHelper.getDatabase(TestHelper.PATHS.PATH1.getFile());
122         return database.createResource(resConf);
123     }
124 
125     /**
126      * Deleting all resources as defined in the enum {@link PATHS}.
127      * 
128      * @throws TTException
129      */
130     public static final void deleteEverything() throws TTException {
131         closeEverything();
132         Database.truncateDatabase(PATHS.PATH1.config);
133         Database.truncateDatabase(PATHS.PATH2.config);
134     }
135 
136     /**
137      * Closing all resources as defined in the enum {@link PATHS}.
138      * 
139      * @throws TTException
140      */
141     public static final void closeEverything() throws TTException {
142         if (INSTANCES.containsKey(PATHS.PATH1.getFile())) {
143             final IDatabase database = INSTANCES.remove(PATHS.PATH1.getFile());
144             database.close();
145         }
146         if (INSTANCES.containsKey(PATHS.PATH2.getFile())) {
147             final IDatabase database = INSTANCES.remove(PATHS.PATH2.getFile());
148             database.close();
149         }
150     }

152     /**
153      * Generating random bytes.
154      * 
155      * @return the random bytes

157     public static final byte[] generateRandomBytes(final int pSize) {

159         random.nextBytes(returnVal);
160         return returnVal;
161     }

163     /**
164      * Getting a node pages filled with nodes.
165      * 
166      * @param revision
167      *            of the page
168      * @param offset
169      *            offset to start within the page
170      * @param length
171      *            length of the page
172      * @param nodePageKey
173      *            key of the nodepage
174      * @return a {@link NodePage} filled
175      */
176     public static NodePage getNodePage(final long revision, final int offset, final int length,
177         final long nodePageKey) {
178         final NodePage page = new NodePage(nodePageKey, revision);
179         for (int i = offset; i < length; i++) {
180             page.setNode(i, new DumbNode(random.nextLong(), random.nextLong()));
181         }
182         return page;
183     }
