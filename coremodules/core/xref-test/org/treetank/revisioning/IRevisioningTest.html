135                             for (int i = 0; i < pFragments.length; i++) {
136                                 for (int j = i * 32; j < (i * 32) + 32; j++) {

138                                         pComplete.getNode(j));

140                             }
141                         }
142                     }// Checker for Differential
143                     , new IRevisionChecker() {
144                         @Override
145                         public void checkRevisions(NodePage pComplete, NodePage[] pFragments) {
146                             // Take the last version first, to get the data out there...
147                             for (int j = 0; j < 32; j++) {
148                                 assertEquals("Check for Incremental failed.", pFragments[0].getNode(j),
149                                     pComplete.getNode(j));
150                             }
151                             // ...and iterate through the first version afterwards for the rest of the

166     private static NodePage[] prepareNormal(final int length) {
167         final NodePage[] pages = new NodePage[length];
168         // filling one entire page with revision 0 and key 0
169         pages[pages.length - 1] = getNodePage(0, 0, 128, 0);
170         for (int i = 0; i < pages.length - 1; i++) {
171             // filling nodepages from end to start with 32 elements each slot
172             pages[i] = getNodePage(pages.length - i - 1, i * 32, (i * 32) + 32, 0);
173         }
174         return pages;
175     }
