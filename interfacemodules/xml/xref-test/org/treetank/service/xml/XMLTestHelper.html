19      /**
20       * Read a file into a StringBuilder.
21       * 
22       * @param paramFile
23       *            The file to read.
24       * @param paramWhitespaces
25       *            Retrieve file and don't remove any whitespaces.
26       * @return StringBuilder instance, which has the string representation of
27       *         the document.
28       * @throws IOException
29       *             throws an IOException if any I/O operation fails.
30       */
31      public static StringBuilder readFile(final File paramFile, final boolean paramWhitespaces)
32          throws IOException {
33          final BufferedReader in = new BufferedReader(new FileReader(paramFile));
34          final StringBuilder sBuilder = new StringBuilder();
35          for (String line = in.readLine(); line != null; line = in.readLine()) {
36              if (paramWhitespaces) {
37                  sBuilder.append(line + "\n");
38              } else {
39                  sBuilder.append(line.trim());
40              }
41          }
42  
43          // Remove last newline.
44          if (paramWhitespaces) {
45              sBuilder.replace(sBuilder.length() - 1, sBuilder.length(), "");
46          }
47          in.close();
48  
49          return sBuilder;
50      }
