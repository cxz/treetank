62      @Test
63      public void testIFilterConvetions() throws AbsTTException {
64  
65          holder.getRtx().moveTo(9L);
66          AbsFilterTest.testIFilterConventions(new NestedFilter(holder.getRtx(),
67              new ItemFilter(holder.getRtx()), new ElementFilter(holder.getRtx()), new NameFilter(holder
68                  .getRtx(), "b")), true);
69          AbsFilterTest.testIFilterConventions(new NestedFilter(holder.getRtx(),
70              new ItemFilter(holder.getRtx()), new AttributeFilter(holder.getRtx()), new NameFilter(holder
71                  .getRtx(), "b")), false);
72  
73          holder.getRtx().moveTo(4L);
74          AbsFilterTest.testIFilterConventions(new NestedFilter(holder.getRtx(),
75              new NodeFilter(holder.getRtx()), new ElementFilter(holder.getRtx())), false);
76          AbsFilterTest.testIFilterConventions(new NestedFilter(holder.getRtx(),
77              new NodeFilter(holder.getRtx()), new TextFilter(holder.getRtx())), true);
78  
79          holder.getRtx().moveTo(1L);
80          holder.getRtx().moveToAttribute(0);
81          AbsFilterTest.testIFilterConventions(new NestedFilter(holder.getRtx(), new AttributeFilter(holder
82              .getRtx()), new NameFilter(holder.getRtx(), "i")), true);
83  
84      }
