58      @AfterMethod
59      public void tearDown() throws TTException {
60          holder.close();
61          TestHelper.closeEverything();
62      }
63  
64      @Test
65      public void testComp() throws TTException {
66  
67          final AbsAxis axis1 = new XPathAxis(holder.getNRtx(), "1.0 = 1.0");
68          assertEquals(true, axis1.hasNext());
69          assertEquals(true, Boolean.parseBoolean(new String(((IValNode)axis1.getNode()).getRawValue())));
70          assertEquals(false, axis1.hasNext());
71  
72          final AbsAxis axis2 = new XPathAxis(holder.getNRtx(), "(1, 2, 3) < (2, 3)");
73          assertEquals(true, axis2.hasNext());
74          assertEquals(true, Boolean.parseBoolean(new String(((IValNode)axis2.getNode()).getRawValue())));
75          assertEquals(false, axis2.hasNext());
76  
77          final AbsAxis axis3 = new XPathAxis(holder.getNRtx(), "(1, 2, 3) > (3, 4)");
78          assertEquals(true, axis3.hasNext());
79          assertEquals(false, Boolean.parseBoolean(new String(((IValNode)axis3.getNode()).getRawValue())));
80          assertEquals(false, axis3.hasNext());
81  
82      }
