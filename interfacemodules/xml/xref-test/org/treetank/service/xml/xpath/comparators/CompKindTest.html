39      @Test
40      public void testEQ() throws TTXPathException {
41  
42          assertEquals(CompKind.EQ.compare("2.0", "2", Type.DOUBLE), true);
43          assertEquals(CompKind.EQ.compare("2.0", "2.01", Type.DOUBLE), false);
44          assertEquals(CompKind.EQ.compare("2.0", "4.0", Type.DOUBLE), false);
45          assertEquals(CompKind.EQ.compare(Double.toString(Double.MAX_VALUE),
46              Double.toString(Double.MAX_VALUE), Type.DOUBLE), true);
47          assertEquals(CompKind.EQ.compare(Double.toString(Double.MAX_VALUE),
48              Double.toString(Double.MIN_VALUE), Type.DOUBLE), false);
49  
50          assertEquals(CompKind.EQ.compare("2.0", "2", Type.DECIMAL), true);
51          assertEquals(CompKind.EQ.compare("2.0", "2.01", Type.DECIMAL), false);
52          assertEquals(CompKind.EQ.compare("2.0", "4.0", Type.DECIMAL), false);
53          assertEquals(CompKind.EQ.compare(Double.toString(Double.MAX_VALUE),
54              Double.toString(Double.MAX_VALUE), Type.DECIMAL), true);
55          assertEquals(CompKind.EQ.compare(Double.toString(Double.MAX_VALUE),
56              Double.toString(Double.MIN_VALUE), Type.DECIMAL), false);
57  
58          assertEquals(CompKind.EQ.compare("2.0", "2", Type.FLOAT), true);
59          assertEquals(CompKind.EQ.compare("2.0", "2.01", Type.FLOAT), false);
60          assertEquals(CompKind.EQ.compare("2.0", "4.0", Type.FLOAT), false);
61          assertEquals(CompKind.EQ.compare(Float.toString(Float.MAX_VALUE), Float.toString(Float.MAX_VALUE),
62              Type.FLOAT), true);
63          assertEquals(CompKind.EQ.compare(Float.toString(Float.MAX_VALUE), Float.toString(Float.MIN_VALUE),
64              Type.FLOAT), false);
65          assertEquals(CompKind.EQ.compare(Double.toString(Double.MAX_VALUE),
66              Double.toString(Double.MAX_VALUE), Type.FLOAT), true);
67          assertEquals(CompKind.EQ.compare(Double.toString(Double.MAX_VALUE),
68              Double.toString(Double.MIN_VALUE), Type.FLOAT), false);
69  
70          assertEquals(CompKind.EQ.compare("2.0", "2", Type.INTEGER), true);
71          assertEquals(CompKind.EQ.compare("2.0", "2.01", Type.INTEGER), true);
72          assertEquals(CompKind.EQ.compare("2.0", "4.0", Type.INTEGER), false);
73          assertEquals(CompKind.EQ.compare(Integer.toString(Integer.MAX_VALUE), Integer
74              .toString(Integer.MAX_VALUE), Type.INTEGER), true);
75          assertEquals(CompKind.EQ.compare(Integer.toString(Integer.MAX_VALUE), Integer
76              .toString(Integer.MIN_VALUE), Type.INTEGER), false);
77  
78          assertEquals(CompKind.EQ.compare("2.0", "2", Type.STRING), false);
79          assertEquals(CompKind.EQ.compare("2.01", "2.01", Type.STRING), true);
80          assertEquals(CompKind.EQ.compare("bla bla blubb", "bla bla blubb", Type.STRING), true);
81  
82          assertEquals(CompKind.EQ.compare("www.uni-konstanz.de", "www.uni-konstanz.de", Type.ANY_URI), true);
83          assertEquals(CompKind.EQ.compare("www.uni-konstanz.de/", "www.uni-konstanz.de", Type.ANY_URI), false);
84  
85          assertEquals(CompKind.EQ.compare("true", "false", Type.BOOLEAN), false);
86          assertEquals(CompKind.EQ.compare("true", "true", Type.BOOLEAN), true);
87  
88          try {
89              assertEquals(CompKind.EQ.compare("2.0", "2", Type.DATE), true);
90              Assert.fail("Expected exception");
91          } catch (IllegalStateException e) {
92              assertEquals(e.getMessage(), "Not implemented for this type yet");
93          }
94  
95          try {
96              assertEquals(CompKind.EQ.compare("2.0", "4.0", Type.G_MONTH), false);
97              Assert.fail("Expected exception");
98          } catch (IllegalStateException e) {
99              assertEquals(e.getMessage(), "Not implemented for this type yet");
100         }
101 
102         try {
103             assertEquals(CompKind.EQ.compare("2.0", "2.01", Type.YEAR_MONTH_DURATION), false);
104             Assert.fail("Expected exception");
105         } catch (TTXPathException e) {
106             assertEquals(e.getMessage(), "err:XPTY0004 The type is not appropriate "
107                 + "the expression or the typedoes not match a required type as"
108                 + " specified by the matching rules. ");
109         }
