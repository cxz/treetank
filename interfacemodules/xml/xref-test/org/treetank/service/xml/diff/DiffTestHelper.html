89      private static void initializeData(final Holder paramHolder, final File... paramFile)
90          throws TTException, IOException, XMLStreamException {
91  
92          int i = 0;
93          for (final File file : paramFile) {
94              XMLShredder init;
95              if (i == 0) {
96                  init =
97                      new XMLShredder(paramHolder.getNWtx(), XMLShredder.createFileReader(file),
98                          EShredderInsert.ADDASFIRSTCHILD);
99              } else {
100                 init =
101                     new XMLUpdateShredder(paramHolder.getNWtx(), XMLShredder.createFileReader(file),
102                         EShredderInsert.ADDASFIRSTCHILD, file, EShredderCommit.COMMIT);
103             }
104             i++;
105             init.call();
106         }
