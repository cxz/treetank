39  /**
40   * <h1>NodeComp</h1>
41   * <p>
42   * Node comparisons are used to compare two nodes, by their identity or by their document order.
43   * </p>
44   */
45  public class NodeComp extends AbsComparator {
46  
47      /**
48       * Constructor. Initializes the internal state.
49       * 
50       * @param rtx
51       *            Exclusive (immutable) trx to iterate with.
52       * @param mOperand1
53       *            First value of the comparison
54       * @param mOperand2
55       *            Second value of the comparison
56       * @param mComp
57       *            comparison kind
58       */
59      public NodeComp(final INodeReadTrx rtx, final AbsAxis mOperand1, final AbsAxis mOperand2,
60          final CompKind mComp) {
61  
62          super(rtx, mOperand1, mOperand2, mComp);
63      }
64  
65      /**
66       * {@inheritDoc}
67       */
68      @Override
69      protected AtomicValue[] atomize(final AbsAxis mOperand) throws TTXPathException {
70  
71          // store item key as atomic value
72          final AtomicValue mAtomized =
73              new AtomicValue(TypedValue.getBytes(((Long)getNode().getNodeKey()).toString()), NamePageHash
74                  .generateHashForString("xs:integer"));
75          final AtomicValue[] op = {
76              mAtomized
77          };
78  
79          // the operands must be singletons in case of a node comparison
80          if (mOperand.hasNext()) {
81              throw EXPathError.XPTY0004.getEncapsulatedException();
82          } else {
83              return op;
84          }
85  
86      }
87  
88      /**
89       * {@inheritDoc}
90       */
91      @Override
92      protected Type getType(final int mKey1, final int mKey2) {
93  
94          return Type.INTEGER;
95  
96      }
97  
98      /**
99       * {@inheritDoc}
100      * 
101      */
102     @Override
103     protected boolean compare(final AtomicValue[] mOperand1, final AtomicValue[] mOperand2)
104         throws TTXPathException {
105 
106         final String op1 = new String(mOperand1[0].getRawValue());
107         final String op2 = new String(mOperand2[0].getRawValue());
108 
109         return getCompKind().compare(op1, op2, getType(mOperand1[0].getTypeKey(), mOperand2[0].getTypeKey()));
110     }
111 
112 }
