100     /**
101      * XPath tests.
102      * 
103      * @param expressions
104      *            Expressions, which are used.
105      * @param doc
106      *            The test document.
107      * @param expectedResults
108      *            Expected result for each expression.
109      * @param result
110      *            Array with the result types for each expression.
111      * @param xpathConstants
112      *            XPathConstants for each expression.
113      * @param namespaces
114      *            Array of boolean values for each expression.
115      * @param namespace
116      *            The namespace Context, which is used.
117      * @throws Exception
118      *             Any Exception which maybe occurs.
119      */
120     @Test(enabled = false)
121     @SuppressWarnings("unchecked")
122     private void test(final String[] expressions, final Object doc, final Object[] expectedResults,
123         Object[] result, final QName[] xpathConstants, final boolean[] namespaces, final Object namespace)
124         throws Exception {
125 
126         // For every expected result.
127         for (int i = 0; i < expectedResults.length; i++) {
128 
129             // If namespace is required.
130             if (namespaces[i]) {
131                 xpe.setNamespaceContext((NamespaceContext)namespace);
132             }
133 
134             final XPathExpression findLine = xpe.compile(expressions[i]);
135 
136             // Cast the evaluated value.
137             if (xpathConstants[i].equals(XPathConstants.NODESET)) {
138                 result[i] = findLine.evaluate(doc, xpathConstants[i]);
139             } else if (xpathConstants[i].equals(XPathConstants.STRING)) {
140                 result[i] = (String)findLine.evaluate(doc, xpathConstants[i]);
141             } else if (xpathConstants[i].equals(XPathConstants.NUMBER)) {
142                 result[i] = Double.parseDouble(findLine.evaluate(doc, xpathConstants[i]).toString());
143             } else {
144                 throw new IllegalStateException("Unknown XPathConstant!");
145             }
146 
147             assertNotNull(result);
148 
149             if (xpathConstants[i].equals(XPathConstants.NODESET)) {
150                 final ArrayList<INode> test = (ArrayList<INode>)result[i];
151 
152                 final String res = (String)expectedResults[i];
153                 final String[] expRes = res.split(" ");
154 
155                 // Iterate over expected result and the actual result and
156                 // compare it.
157                 for (int j = 0; j < test.size(); j++) {
158                     final INode item = test.get(j);
159 
160                     mHolder.getNRtx().moveTo(item.getNodeKey());
161 
162                     final QName qName = mHolder.getNRtx().getQNameOfCurrentNode();
163 
164                     if (mHolder.getNRtx().getNode().getKind() == IConstants.ELEMENT) {
165                         assertEquals(expRes[j], qName.getPrefix() + ":" + qName.getLocalPart());
166                     } else if (mHolder.getNRtx().getNode().getKind() == IConstants.TEXT) {
167                         assertEquals(expRes[j], mHolder.getNRtx().getValueOfCurrentNode());
168                     }
169 
170                 }
171             } else {
172                 assertEquals(expectedResults[i], result[i]);
173             }
174         }
175     }

179 
180         final String[] expressions =
181             {
182                 "count(//b)", "count(//p:a)", "//p:a/@i", "//p:a/@p:i", "//b[1]/text()", "//b[2]", "//b[1]",
183                 "//b[2]/text()", "//p:a/text()"
184             };
185 
186         final NodeInfo doc = new DocumentWrapper(mHolder.getSession(), config);
187 
188         final Object[] expectedResults = {
189             2D, 1D, "j", "", "foo",
190             // "<b p:x=\"y\"><c/>bar</b>",
191             "bar", "foo",
192             // "<b>foo<c/></b>",
193             "bar", "oops1 oops2 oops3"
194         };
195 
196         Object[] result = {
197             "", "", "", 0D, "", "", "", "", ""
198         };
199 
200         final QName[] xpathConstants =
201             {
202                 XPathConstants.NUMBER, XPathConstants.NUMBER, XPathConstants.STRING, XPathConstants.STRING,
203                 XPathConstants.STRING, XPathConstants.STRING, XPathConstants.STRING, XPathConstants.STRING,
204                 XPathConstants.NODESET,
205             };
206 
207         final boolean[] namespaces = {
208             false, true, true, false, false, false, false, false, true
209         };
210 
211         test(expressions, doc, expectedResults, result, xpathConstants, namespaces, new DocNamespaceContext());
212     }

438     /**
439      * <h1>Namespace Context</h1>
440      * 
441      * <p>
442      * Namespace Context to test the NodeInfo implementation ( <code>NodeWrapper</code>). It's written for the
443      * test document which is written via <code></code>.
444      * </p>
445      * 
446      * @author Johannes Lichtenberger, University of Konstanz
447      */
448     static class DocNamespaceContext implements NamespaceContext {
449 
450         /**
451          * Get the Namespace URI.
452          * 
453          * @param prefix
454          *            Prefix of the current node.
455          * @return Return the Namespace URI.
456          * @throws IllegalArgumentException
457          *             if prefix is null.
458          */
459         public String getNamespaceURI(final String prefix) {
460             if (prefix == null) {
461                 throw new IllegalArgumentException("Prefix may not be null!");
462             } else if (prefix == XMLConstants.DEFAULT_NS_PREFIX) {
463                 return XMLConstants.NULL_NS_URI;
464             } else if (prefix == XMLConstants.XML_NS_PREFIX) {
465                 return XMLConstants.XML_NS_URI;
466             } else if (prefix == XMLConstants.XMLNS_ATTRIBUTE) {
467                 return XMLConstants.XMLNS_ATTRIBUTE_NS_URI;
468             } else if ("p".equals(prefix)) {
469                 return "ns";
470             } else {
471                 return XMLConstants.NULL_NS_URI;
472             }
473         }
474 
475         /**
476          * Get the prefix of the namespace.
477          * 
478          * @param namespace
479          *            Namespace of the current node.
480          * @return "p" if the Namespace equals "ns", otherwise returns null.
481          */
482         public String getPrefix(final String namespace) {
483             if (namespace == null) {
484                 throw new IllegalArgumentException("Namespace may not be null!");
485             } else if (XMLConstants.XML_NS_URI.equals(namespace)) {
486                 return XMLConstants.XML_NS_PREFIX;
487             } else if (XMLConstants.XMLNS_ATTRIBUTE_NS_URI.equals(namespace)) {
488                 return XMLConstants.XMLNS_ATTRIBUTE;
489             } else if (namespace.equals("ns")) {
490                 return "p";
491             } else {
492                 return null;
493             }
494         }
495 
496         /**
497          * Not needed/supported (only one prefix exists for a NS_URI).
498          */
499         public Iterator<String> getPrefixes(final String namespace) {
500             throw new UnsupportedOperationException("Currently not needed by the test document!");
501         }
502     }
