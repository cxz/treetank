134     @SuppressWarnings("unchecked")
135     protected void do_btnSubmit_widgetSelected(final SelectionEvent e) {
136         try {
137             this.name = text.getText();
138             
139             String backend = txtBackend.getText();
140             String rev = txtRevisioning.getText();
141             Class<? extends IBackend> backendClass;
142             Class<? extends IRevisioning> revClass;
143             
144             try {
145                 backendClass = (Class<IBackend>) Class.forName(backend);
146             } catch (ClassNotFoundException e1) {
147                 backendClass = JCloudsStorage.class;
148             }
149             
150             try {
151                 revClass = (Class<IRevisioning>) Class.forName(rev);
152             } catch (ClassNotFoundException e1) {
153                 revClass = SlidingSnapshot.class;
154             }
155             
156             StorageManager.createResource(this.name, new ModuleSetter().setDataFacClass(FileDataFactory.class).setMetaFacClass(FilelistenerMetaDataFactory.class)
157                 .setRevisioningClass(revClass).setBackendClass(backendClass).createModule());
158             
159         } catch (StorageAlreadyExistsException e1) {
160             lblTheStorageAlready.setVisible(true);
161             return;
162         } catch (TTException e1) {
163             lblTheStorageAlready
164                 .setText("Something went wrong. Please try another backend and/or another name.");
165             lblTheStorageAlready.setVisible(true);
166             return;
167         }
168 
169         this.getParent().close();
170     }
