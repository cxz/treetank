86      private void bench(int filebenchSize) throws FileNotFoundException, ClassNotFoundException, IOException,
87          ResourceNotExistingException, TTException, InterruptedException {
88          // Listening to the target folder
89          // Assert.assertNotNull(new File(BUCKETFOLDER).list());
90          final File tmpdir = Files.createTempDir();
91          Filelistener.addFilelistener(RESOURCE_1, tmpdir.getAbsolutePath());
92          filelistener.watchDir(tmpdir);
93          filelistener.startListening();
94  
95          // Benching creation of files on the filesystem and awaiting
96          // finalization in treetank.
97  
98          createEnds = new long[FILES];
99          readEnds = new long[FILES];
100         bucketCount = new long[FILES];
101         for (int i = 0; i < FILES; i++) {
102             String filename = tmpdir + File.separator + "file" + (i + 1) + ".data";
103             fileMap.put(File.separator + "file" + (i + 1) + ".data", i);
104 
105             byte[] fileBytes = new byte[filebenchSize];
106             Random rand = new Random(42 * (i + 1));
107             rand.nextBytes(fileBytes);
108             Files.write(fileBytes, new File(filename));
109 
110         }
111 
112         while (!finishedBench || !notifications.isEmpty()) {
113             FilesystemNotification n = notifications.poll();
114             if (n == null || n.getEvtType() == ENTRY_CREATE) {
115                 continue;
116             }
117             if (n.getRelativePath() != null) {
118                 createEnds[fileMap.get(n.getRelativePath())] = n.getTime();
119                 bucketCount[fileMap.get(n.getRelativePath())] = n.getBucketAmount();
120                 // System.out.println("Run finished.");
121                 if (fileMap.get(n.getRelativePath()) == FILES - 1)
122                     finishedBench = true;
123             }
124         }
125         
126         if (finishedBench) {
127             //All files in storage, time to check retrieval time
128             IFilelistenerWriteTrx trx = filelistener.getTrx(RESOURCE_1);
129             File file;
130             long time;
131             String filename;
132             for (int i = 0; i < FILES; i++) {
133                 filename = File.separator + "file" + (i + 1) + ".data";
134                 time = System.currentTimeMillis();
135                 file = trx.getFullFile(filename);
136                 time = System.currentTimeMillis() - time;
137                 readEnds[i] = time;
138 
139             }
140             
141             // Do something, analyze
142             printBench(filebenchSize + "");
143         } else {
144             fail("Bench was not finished but notified that it would be finished.");
145         }
146 
147         filelistener.shutDownListener();
148         IOUtils.recursiveDelete(tmpdir);
149     }

151     private void printBench(String string) throws IOException {
152         while (string.length() < 8) {
153             string = "0" + string;
154         }
155         String s = "";
156         String s2 = "";
157         String s3 = "";
158         for (int i = 0; i < createEnds.length - 1; i++) {
159             s   += createEnds[i] + ",";
160             s2  += bucketCount[i] + ",";
161             s3  += readEnds[i] + ",";
162 
163         }
164         s       +=  createEnds[createEnds.length - 1];
165         s2      += bucketCount[bucketCount.length - 1];
166         s3      +=    readEnds[readEnds.length - 1];
167 
168         Files.write(s   + "\n", new File(createBenchFile + string + "T.csv"), Charset.forName("UTF-8"));
169         Files.write(s2  + "\n", new File(createBenchFile + string + "B.csv"), Charset.forName("UTF-8"));
170         Files.write(s3  + "\n", new File(createBenchFile + string + "R.csv"), Charset.forName("UTF-8"));
171     }
