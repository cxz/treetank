27  package org.jaxrx.core;
28  
29  import java.util.HashMap;
30  import java.util.Map;
31  import org.jaxrx.JaxRx;
32  
33  /**
34   * This class organizes all implementations of the JAX-RX interface.
35   *
36   * @author Sebastian Graf, University of Konstanz
37   *
38   */
39  public final class Systems {
40  	/**
41  	 * Private constructor.
42  	 */
43  	private Systems() {
44  	}
45  
46  	/**
47  	 * System enumeration.
48  	 */
49  	private enum Sys {
50  		/**
51  		 * The implementation paths.
52  		 */
53  		IMPLEMENTATION(JaxRxConstants.PATHPROP, "org.jaxrx.dom.DOMJaxRx"),
54  
55  		/**
56  		 * Default package. This results in a look up related to the system
57  		 * properties for a org.jaxrx.systemPath-key
58  		 */
59  		SYSTEMNAME(JaxRxConstants.NAMEPROP, "dom");
60  
61  		/** System key. */
62  		final String key;
63  		/** System value. */
64  		final String value;
65  
66  		/**
67  		 * Constructor.
68  		 *
69  		 * @param paramKey
70  		 *            key
71  		 * @param paramValue
72  		 *            value
73  		 */
74  		private Sys(final String paramKey, final String paramValue) {
75  			this.key = paramKey;
76  			this.value = paramValue;
77  		}
78  	}
79  
80  	/**
81  	 * Delimiter for multiple values.
82  	 */
83  	private static final String VALDELIM = ";";
84  
85  	/**
86  	 * Block to load systems preference once. Note that in case of multivalue
87  	 * properties, all properties have to have the same multiple values.
88  	 */
89  	private static final Map<String, String> SYSTEMSIMPLS = new HashMap<String, String>();
90  
91  	/**
92  	 * The map holds the instances representing the classes of the
93  	 * implementation.
94  	 */
95  	private static final Map<String, JaxRx> INSTANCES = new HashMap<String, JaxRx>();
96  
97  	static {
98  		final String systemVal = System.getProperty(Sys.SYSTEMNAME.key);
99  		final String implVal = System.getProperty(Sys.IMPLEMENTATION.key);
100 
101 		if (systemVal != null && implVal != null) {
102 			final String[] systemValSplitted = systemVal.split(VALDELIM);
103 			final String[] implValSplitted = implVal.split(VALDELIM);
104 			for (int i = 0; i < systemVal.split(VALDELIM).length; i++) {
105 				SYSTEMSIMPLS.put(systemValSplitted[i], implValSplitted[i]);
106 			}
107 		} else if (systemVal == null && implVal == null) {
108 			SYSTEMSIMPLS.put(Sys.SYSTEMNAME.value, Sys.IMPLEMENTATION.value);
109 		} else {
110 			// TODO insert cool exception;
111 			throw new IllegalStateException();
112 		}
113 	}
114 
115 	/**
116 	 * Getting all available systems as keys and implementing packages as
117 	 * values.
118 	 *
119 	 * @return the systems implementations
120 	 */
121 	public static Map<String, String> getSystems() {
122 		return SYSTEMSIMPLS;
123 	}
124 
125 	/**
126 	 * Returns the instance for the specified implementation. If the system is
127 	 * unknown, throws an exception.
128 	 *
129 	 * @param impl
130 	 *            implementation to be checked.
131 	 * @return instances
132 	 */
133 	public static JaxRx getInstance(final String impl) {
134 		final String path = Systems.getSystems().get(impl);
135 		if (path == null) {
136 			throw new JaxRxException(404, "Unknown implementation: " + impl);
137 		}
138 
139 		JaxRx jaxrx = INSTANCES.get(path);
140 		if (jaxrx == null) {
141 			try {
142 				jaxrx = (JaxRx) Class.forName(path).newInstance();
143 				INSTANCES.put(path, jaxrx);
144 			} catch (final Exception ex) {
145 				throw new JaxRxException(ex);
146 			}
147 		}
148 		return jaxrx;
149 	}
