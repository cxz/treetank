27  package org.jaxrx.core;

29  import java.io.OutputStream;
30  import java.util.ArrayList;
31  import java.util.List;
32  import javax.ws.rs.core.StreamingOutput;
33  import javax.xml.parsers.DocumentBuilderFactory;
34  import javax.xml.parsers.ParserConfigurationException;
35  import javax.xml.transform.Transformer;
36  import javax.xml.transform.TransformerException;
37  import javax.xml.transform.TransformerFactory;
38  import javax.xml.transform.dom.DOMSource;
39  import javax.xml.transform.stream.StreamResult;
40  import org.w3c.dom.Document;
41  import org.w3c.dom.Element;

43  /**
44   * This class builds the response XML fragments for JAX-RX resources.
45   *
46   * @author Patrick Lang, Christian Gruen, Lukas Lewandowski, University of
47   *         Konstanz
48   *
49   */
50  public final class ResponseBuilder {
51  	/**
52  	 * The private empty constructor.
53  	 */
54  	private ResponseBuilder() {
55  		// i do nothing
56  	}
57  
58  	/**
59  	 * This method creates a new {@link Document} instance for the surrounding
60  	 * XML element for the client response.
61  	 *
62  	 * @return The created {@link Document} instance.
63  	 * @throws ParserConfigurationException
64  	 *             The exception occurred.
65  	 */
66  	private static Document createSurroundingXMLResp()
67  			throws ParserConfigurationException {
68  		return DocumentBuilderFactory.newInstance().newDocumentBuilder()
69  				.newDocument();
70  	}
71  
72  	/**
73  	 * This method creates the XML element containing a collection. This
74  	 * collection contains the available resources which are children of the
75  	 * database.
76  	 *
77  	 * @param pathResource
78  	 *            The list with the path of the available resources.
79  	 * @param document
80  	 *            The XML {@link Document} instance.
81  	 * @return A list of XML {@link Element} as the collection.
82  	 */
83  	private static List<Element> createCollectionElement(
84  			final List<String> pathResource, final Document document) {
85  
86  		final List<Element> collections = new ArrayList<Element>();
87  		for (final String path : pathResource) {
88  			final Element collection = document.createElementNS(
89  					JaxRxConstants.URL, JaxRxConstants.JAXRX + ":resource");
90  			collection.setAttribute("name", path);
91  			collections.add(collection);
92  		}
93  		return collections;
94  	}
95  
96  	/**
97  	 * This method creates the response XML element.
98  	 *
99  	 * @param document
100 	 *            The {@link Document} instance for the response.
101 	 * @return The created XML {@link Element}.
102 	 */
103 	private static Element createResultElement(final Document document) {
104 		return document.createElementNS(JaxRxConstants.URL, JaxRxConstants.JAXRX
105 				+ ":results");
106 	}
107 
108 	/**
109 	 * Builds a DOM response.
110 	 *
111 	 * @param availableResources
112 	 *            list of resources to include in the output
113 	 * @return streaming output
114 	 */
115 	public static StreamingOutput buildDOMResponse(
116 			final List<String> availableResources) {
117 
118 		try {
119 			final Document document = createSurroundingXMLResp();
120 			final Element resElement = createResultElement(document);
121 
122 			final List<Element> resources = createCollectionElement(
123 					availableResources, document);
124 			for (final Element resource : resources) {
125 				resElement.appendChild(resource);
126 			}
127 			document.appendChild(resElement);
128 			return createStream(document);
129 		} catch (final ParserConfigurationException exc) {
130 			throw new JaxRxException(exc);
131 		}
132 	}
133 
134 	/**
135 	 * Creates an output stream from the specified document.
136 	 *
137 	 * @param doc
138 	 *            document input
139 	 * @return output stream
140 	 */
141 	public static StreamingOutput createStream(final Document doc) {
142 		return new StreamingOutput() {
143 
144 			@Override
145 			public void write(final OutputStream output) {
146 				synchronized (output) {
147 					final DOMSource domSource = new DOMSource(doc);
148 					final StreamResult streamResult = new StreamResult(output);
149 					Transformer transformer;
150 					try {
151 						transformer = TransformerFactory.newInstance()
152 								.newTransformer();
153 						transformer.transform(domSource, streamResult);
154 					} catch (final TransformerException exc) {
155 						exc.printStackTrace();
156 						throw new JaxRxException(exc);
157 					}
158 
159 				}
160 			}
161 		};
162 	}
