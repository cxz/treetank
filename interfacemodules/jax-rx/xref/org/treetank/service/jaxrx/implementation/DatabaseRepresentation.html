138         synchronized (resourceName) {

151     /**
152      * This method is responsible to deliver the whole database. Additional
153      * parameters can be set (wrap, revision, output) which change the response
154      * view.
155      * 
156      * @param resourceName
157      *            The name of the requested database.
158      * @param queryParams
159      *            The optional query parameters.
160      * @return The XML database resource, depending on the query parameters.
161      * @throws JaxRxException
162      *             The exception occurred.
163      */
164     public StreamingOutput getResource(final String resourceName,
165         final Map<QueryParameter, String> queryParams) throws JaxRxException {
166         final StreamingOutput streamingOutput = new StreamingOutput() {
167             @Override
168             public void write(final OutputStream output) throws IOException, JaxRxException {
169                 final String revision = queryParams.get(QueryParameter.REVISION);
170                 final String wrap = queryParams.get(QueryParameter.WRAP);
171                 final String nodeId = queryParams.get(QueryParameter.OUTPUT);
172                 final boolean wrapResult = (wrap == null) ? false : wrap.equalsIgnoreCase(YESSTRING);
173                 final boolean nodeid = (nodeId == null) ? false : nodeId.equalsIgnoreCase(YESSTRING);
174                 try {
175                     if (revision == null) {
176                         serialize(resourceName, null, nodeid, output, wrapResult);
177                     } else {
178                         // pattern which have to match against the input
179                         final Pattern pattern = Pattern.compile("[0-9]+[-]{1}[1-9]+");
180 
181                         final Matcher matcher = pattern.matcher(revision);
182 
183                         if (matcher.matches()) {
184                             getModificHistory(resourceName, revision, nodeid, output, wrapResult);
185                         } else {
186                             serialize(resourceName, Long.valueOf(revision), nodeid, output, wrapResult);
187                         }
188                     }
189                 } catch (final NumberFormatException exce) {
190                     throw new JaxRxException(400, exce.getMessage());
191                 } catch (final TTException exce) {
192                     throw new JaxRxException(exce);
193                 }
194             }
195         };
196         return streamingOutput;
197     }

212     public StreamingOutput performQueryOnResource(final String resource, final String query,
213         final Map<QueryParameter, String> otherParams) {
214         final StreamingOutput streamingOutput = new StreamingOutput() {
215             @Override
216             public void write(final OutputStream output) throws IOException, JaxRxException {
217                 final String revision = otherParams.get(QueryParameter.REVISION);
218                 final String wrap = otherParams.get(QueryParameter.WRAP);
219                 final String nodeId = otherParams.get(QueryParameter.OUTPUT);
220                 final boolean wrapResult = (wrap == null) ? true : wrap.equalsIgnoreCase(YESSTRING);
221                 final boolean nodeid = (nodeId == null) ? false : nodeId.equalsIgnoreCase(YESSTRING);
222                 final Long rev = revision == null ? null : Long.valueOf(revision);
223                 final RestXPathProcessor xpathProcessor = new RestXPathProcessor(mDatabase);
224                 try {
225                     xpathProcessor.getXpathResource(resource, query, nodeid, rev, output, wrapResult);
226                 } catch (final TTException exce) {
227                     throw new JaxRxException(exce);
228                 }
229             }
230         };
231         return streamingOutput;
232     }

286     /**
287      * This method is responsible to save the XML file, which is in an {@link InputStream}, as a TreeTank
288      * object.
289      * 
290      * @param xmlInput
291      *            The XML file in an {@link InputStream}.
292      * @param resource
293      *            The name of the resource.
294      * @return <code>true</code> when the shredding process has been successful. <code>false</code> otherwise.
295      * @throws TTException
296      */
297     public final boolean shred(final InputStream xmlInput, final String resource) throws TTException {
298         boolean allOk;
299         INodeWriteTrx wtx = null;
300         IPageWriteTrx pWtx = null;
301         ISession session = null;
302         boolean abort = false;
303         try {
304             if (!mDatabase.existsResource(resource)) {
305                 Properties properties =
306                     StandardSettings.getProps(mDatabase.getLocation().getAbsolutePath(),
307                         resource);
308 
309                 mDatabase.createResource(new ResourceConfiguration(properties, mStorageFac, mRevision,
310                     NODEFACTORY, METAFAC));
311             }
312 
313             session = mDatabase.getSession(new SessionConfiguration(resource, StandardSettings.KEY));
314             pWtx = session.beginPageWriteTransaction();
315             wtx = new NodeWriteTrx(session, pWtx, HashKind.Rolling);
316             wtx.moveTo(ROOT_NODE);
317             final XMLShredder shredder =
318                 new XMLShredder(wtx, RESTXMLShredder.createReader(xmlInput), EShredderInsert.ADDASFIRSTCHILD);
319             shredder.call();
320             allOk = true;
321         } catch (final Exception exce) {
322             abort = true;
323             throw new JaxRxException(exce);
324         } finally {
325             WorkerHelper.closeWTX(abort, wtx, session);
326         }
327         return allOk;
328     }

330     /**
331      * This method is responsible to build an {@link OutputStream} containing an
332      * XML file out of the TreeTank file.
333      * 
334      * @param resource
335      *            The name of the resource that will be offered as XML.
336      * @param nodeid
337      *            To response the resource with a restid for each node ( <code>true</code>) or without (
338      *            <code>false</code>).
339      * @param revision
340      *            The revision of the requested resource. If <code>null</code>,
341      *            than response the latest revision.
342      * @return The {@link OutputStream} containing the serialized XML file.
343      * @throws IOException
344      * @throws TTException
345      * @throws WebApplicationException
346      */
347     private final OutputStream serialize(final String resource, final Long revision, final boolean nodeid,
348         final OutputStream output, final boolean wrapResult) throws IOException, JaxRxException, TTException {
349 
350         if (mDatabase.existsResource(resource)) {
351             try {
352                 if (wrapResult) {
353                     output.write(beginResult.getBytes());
354                     serializIt(resource, revision, output, nodeid);
355                     output.write(endResult.getBytes());
356                 } else {
357                     serializIt(resource, revision, output, nodeid);
358                 }
359             } catch (final Exception exce) {
360                 throw new JaxRxException(exce);
361             }
362         } else {
363             throw new JaxRxException(404, "Not found");
364         }
365 
366         return output;
367     }

380     public long getLastRevision(final String resourceName) throws JaxRxException, TTException {
381 
382         long lastRevision;
383         if (mDatabase.existsResource(resourceName)) {
384             ISession session = null;
385             try {
386                 session = mDatabase.getSession(new SessionConfiguration(resourceName, StandardSettings.KEY));

400     /**
401      * This method reads the existing database, and offers all modifications of
402      * the two given revisions
403      * 
404      * @param resourceName
405      *            The name of the existing database.
406      * @param revisionRange
407      *            Contains the range of revisions
408      * @param nodeid
409      *            To response the resource with a restid for each node ( <code>true</code>) or without (
410      *            <code>false</code>).
411      * @param output
412      *            The OutputStream reference which have to be modified and
413      *            returned
414      * @param wrap
415      *            <code>true</code> if the results have to be wrapped. <code>false</code> otherwise.
416      * @return The {@link OutputStream} containing the result
417      * @throws TTException
418      * @throws WebApplicationException
419      *             The Exception occurred.
420      */
421     public OutputStream getModificHistory(final String resourceName, // NOPMD this method needs alls these
422         // functions
423         final String revisionRange, final boolean nodeid, final OutputStream output, final boolean wrap)
424         throws JaxRxException, TTException {
425 
426         // extract both revision from given String value
427         final StringTokenizer tokenizer = new StringTokenizer(revisionRange, "-");
428         final long revision1 = Long.valueOf(tokenizer.nextToken());
429         final long revision2 = Long.valueOf(tokenizer.nextToken());
430 
431         if (revision1 < revision2 && revision2 <= getLastRevision(resourceName)) {
432 
433             // variables for highest rest-id in respectively revision
434             long maxRestidRev1 = 0;
435             long maxRestidRev2 = 0;
436 
437             // Connection to treetank, creating a session
438             AbsAxis axis = null;
439             INodeReadTrx rtx = null;
440             ISession session = null;
441             // List for all restIds of modifications
442             final List<Long> modificRestids = new LinkedList<Long>();
443 
444             // List of all restIds of revision 1
445             final List<Long> restIdsRev1 = new LinkedList<Long>();
446 
447             try {
448                 session = mDatabase.getSession(new SessionConfiguration(resourceName, StandardSettings.KEY));
449 
450                 // get highest rest-id from given revision 1
451                 rtx = new NodeReadTrx(session.beginPageReadTransaction(revision1));
452                 axis = new XPathAxis(rtx, ".//*");
453 
454                 while (axis.hasNext()) {
455                     if (rtx.getNode().getNodeKey() > maxRestidRev1) {
456                         maxRestidRev1 = rtx.getNode().getNodeKey();
457                     }
458                     // stores all restids from revision 1 into a list
459                     restIdsRev1.add(rtx.getNode().getNodeKey());
460                 }
461                 rtx.moveTo(ROOT_NODE);
462                 rtx.close();
463 
464                 // get highest rest-id from given revision 2
465                 rtx = new NodeReadTrx(session.beginPageReadTransaction(revision2));
466                 axis = new XPathAxis(rtx, ".//*");
467 
468                 while (axis.hasNext()) {
469                     final Long nodeKey = rtx.getNode().getNodeKey();
470                     if (nodeKey > maxRestidRev2) {
471                         maxRestidRev2 = rtx.getNode().getNodeKey();
472                     }
473                     if (nodeKey > maxRestidRev1) {
474                         /*
475                          * writes all restids of revision 2 higher than the
476                          * highest restid of revision 1 into the list
477                          */
478                         modificRestids.add(nodeKey);
479                     }
480                     /*
481                      * removes all restids from restIdsRev1 that appears in
482                      * revision 2 all remaining restids in the list can be seen
483                      * as deleted nodes
484                      */
485                     restIdsRev1.remove(nodeKey);
486                 }
487                 rtx.moveTo(ROOT_NODE);
488                 rtx.close();
489 
490                 rtx = new NodeReadTrx(session.beginPageReadTransaction(revision1));
491 
492                 // linked list for holding unique restids from revision 1
493                 final List<Long> restIdsRev1New = new LinkedList<Long>();
494 
495                 /*
496                  * Checks if a deleted node has a parent node that was deleted
497                  * too. If so, only the parent node is stored in new list to
498                  * avoid double print of node modification
499                  */
500                 for (Long nodeKey : restIdsRev1) {
501                     rtx.moveTo(nodeKey);
502                     final long parentKey = rtx.getNode().getParentKey();
503                     if (!restIdsRev1.contains(parentKey)) {
504                         restIdsRev1New.add(nodeKey);
505                     }
506                 }
507                 rtx.moveTo(ROOT_NODE);
508                 rtx.close();
509 
510                 if (wrap) {
511                     output.write(beginResult.getBytes());
512                 }
513                 /*
514                  * Shred modified restids from revision 2 to xml fragment Just
515                  * modifications done by post commands
516                  */
517                 rtx = new NodeReadTrx(session.beginPageReadTransaction(revision2));
518 
519                 for (Long nodeKey : modificRestids) {
520                     rtx.moveTo(nodeKey);
521                     WorkerHelper.serializeXML(session, output, false, nodeid, nodeKey, revision2).call();
522                 }
523                 rtx.moveTo(ROOT_NODE);
524                 rtx.close();
525 
526                 /*
527                  * Shred modified restids from revision 1 to xml fragment Just
528                  * modifications done by put and deletes
529                  */
530                 rtx = new NodeReadTrx(session.beginPageReadTransaction(revision1));
531                 for (Long nodeKey : restIdsRev1New) {
532                     rtx.moveTo(nodeKey);
533                     WorkerHelper.serializeXML(session, output, false, nodeid, nodeKey, revision1).call();
534                 }
535                 if (wrap) {
536                     output.write(endResult.getBytes());
537                 }
538 
539                 rtx.moveTo(ROOT_NODE);
540 
541             } catch (final Exception globExcep) {
542                 throw new JaxRxException(globExcep);
543             } finally {
544                 WorkerHelper.closeRTX(rtx, session);
545             }
546         } else {
547             throw new JaxRxException(400, "Bad user request");
548         }
549 
550         return output;
551     }

553     /**
554      * The XML serializer to a given tnk file.
555      * 
556      * @param resource
557      *            The resource that has to be serialized.
558      * @param revision
559      *            The revision of the document.
560      * @param output
561      *            The output stream where we write the XML file.
562      * @param nodeid
563      *            <code>true</code> when you want the result nodes with node
564      *            id's. <code>false</code> otherwise.
565      * @throws WebApplicationException
566      *             The exception occurred.
567      * @throws TTException
568      */
569     private void serializIt(final String resource, final Long revision, final OutputStream output,
570         final boolean nodeid) throws JaxRxException, TTException {
571         // Connection to treetank, creating a session
572         ISession session = null;
573         // INodeReadTrx rtx = null;
574         try {
575             session = mDatabase.getSession(new SessionConfiguration(resource, StandardSettings.KEY));
576             // and creating a transaction
577             // if (revision == null) {
578             // rtx = session.beginReadTransaction();
579             // } else {
580             // rtx = session.beginReadTransaction(revision);
581             // }
582             final XMLSerializerBuilder builder;
583             if (revision == null)
584                 builder = new XMLSerializerBuilder(session, output);
585             else
586                 builder = new XMLSerializerBuilder(session, output, revision);
587             builder.setREST(nodeid);
588             builder.setID(nodeid);
589             builder.setDeclaration(false);
590             final XMLSerializer serializer = builder.build();
591             serializer.call();
592         } catch (final Exception exce) {
593             throw new JaxRxException(exce);
594         } finally {
595             // closing the treetank storage
596             WorkerHelper.closeRTX(null, session);
597         }
598     }

601      * This method reverts the latest revision data to the requested.
602      * 
603      * @param resourceName
604      *            The name of the XML resource.
605      * @param backToRevision
606      *            The revision value, which has to be set as the latest.
607      * @throws WebApplicationException
608      * @throws TTException
609      */
610     public void revertToRevision(final String resourceName, final long backToRevision) throws JaxRxException,
611         TTException {
612         ISession session = null;
613         INodeWriteTrx wtx = null;
614         boolean abort = false;
615         try {
616             session = mDatabase.getSession(new SessionConfiguration(resourceName, StandardSettings.KEY));
617             wtx = new NodeWriteTrx(session, session.beginPageWriteTransaction(), HashKind.Rolling);
618             wtx.revertTo(backToRevision);
619             wtx.commit();
620         } catch (final TTException exce) {
621             abort = true;
622             throw new JaxRxException(exce);
623         } finally {
624             WorkerHelper.closeWTX(abort, wtx, session);
625         }
