25  package org.treetank.node;
26  
27  import java.io.DataInput;
28  import java.io.IOException;
29  
30  import org.treetank.api.INode;
31  import org.treetank.api.INodeFactory;
32  import org.treetank.exception.TTIOException;
33  
34  /**
35   * This factory is used to deserialize {@link ByteNode}
36   * 
37   * @author Andreas Rain, University of Konstanz
38   */
39  public class ByteNodeFactory implements INodeFactory {
40  
41      /**
42       * {@inheritDoc}
43       * 
44       */
45      @Override
46      public INode deserializeNode(DataInput input) throws TTIOException {
47          try {
48              int size = input.readInt();
49              long index = input.readLong();
50              long nodeKey = input.readLong();
51              long previousNodeKey = input.readLong();
52              long nextNodeKey = input.readLong();
53              byte[] data = new byte[size];
54              input.readFully(data);
55  
56              ByteNode node = new ByteNode(nodeKey, data);
57              node.setIndex(index);
58              node.setNextNodeKey(nextNodeKey);
59              node.setPreviousNodeKey(previousNodeKey);
60              return node;
61          } catch (final IOException exc) {
62              throw new TTIOException(exc);
63          }
64      }
65  
66  }
