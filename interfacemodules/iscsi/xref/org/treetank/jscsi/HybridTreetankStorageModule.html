47  /**
48   * <h1>TreetankStorageModule</h1>
49   * <p>
50   * This implementation is used to store data into treetank via an iscsi target.
51   * </p>
52   * 
53   * @author Andreas Rain
54   */
55  public class HybridTreetankStorageModule implements IStorageModule {
56  
57      /**
58       * Number of Blocks in one Cluster. 8 equals 4KB datas; 16 equals 8KB datas
59       * ...
60       * 
61       * When using berkeley db as a backend consider that if you increase the
62       * size of the blocks in datas up to 12800 datas can be held in ram. If you
63       * have a 1gb storage maximally 1gb heap is needed. However if the storage
64       * is bigger and the blocks per data is considerably high (e.g. 256kb per
65       * node) you might have to increase the ram for the jvm.
66       */
67      public static final int BLOCKS_IN_DATA = 8;
68  
69      /** Threshold when commit should occur in number of bytes. */
70      private static final int COMMIT_THRESHOLD = 67108864;
71  
72      /** Number of Bytes in Bucket. */
73      public final static int BYTES_IN_DATA = BLOCKS_IN_DATA * VIRTUAL_BLOCK_SIZE;
74  
75      private static final Logger LOGGER = LoggerFactory.getLogger(HybridTreetankStorageModule.class);
76  
77      /**
78       * The number of datas in the storage resulting in mDataNumbers *
79       * BLOCKS_IN_DATA * VIRTUAL_BLOCK_SIZE bytes
80       * 
81       * @see #VIRTUAL_BLOCK_SIZE
82       */
83      private final long mDataNumbers;
84  
85      /**
86       * The mSession this storage module uses to access the storage device.
87       */
88      private final ISession mSession;
89  
90      /**
91       * {@link IIscsiWriteTrx} that is used to write/read from treetank.
92       */
93      private final IIscsiWriteTrx mRtx;
94  
95      /**
96       * Bytewriter counter - If a certain amount of bytes have been written, a
97       * commit is made to treetank.
98       */
99      private volatile int mByteCounter;
100 
101     /** FileStorageModule mirror for faster response times (filesystem backend) */
102     private JCloudsStorageModule jCloudsStorageModule;
103 
104     /** ExecutorService to perform WriteTasks */
105     private ExecutorService mWriteTaskExecutor;

107     /**
108      * Creates a storage module that is used by the target to handle I/O.
109      * 
110      * @param pNodeNumber
111      *            Define how many nodes the storage holds.
112      * @param pSession
113      *            Pass the session associated to the location to this class.
114      * @throws TTException
115      *             will be thrown if there are problems creating this storage.
116      */
117     public HybridTreetankStorageModule(final long pNodeNumber, final ISession pSession) throws TTException {
118 
119         mDataNumbers = pNodeNumber;
120 
121         LOGGER.debug("Initializing storagemodule with: number of nodes=" + mDataNumbers + ", blockSize="
122             + IStorageModule.VIRTUAL_BLOCK_SIZE);
123 
124         mSession = pSession;
125         mRtx = new IscsiWriteTrx(mSession.beginBucketWtx(), mSession);
126 
127         mWriteTaskExecutor = Executors.newSingleThreadExecutor(new WriteTaskThreadFactory());
128 
129         createStorage();
130         System.out.println("Device ready");
131     }
132 
133     /**
134      * Bootstrap a new device as a treetank storage using nodes to abstract the
135      * device.
136      * 
137      * @throws IOException
138      *             is thrown if a node couldn't be created due to errors in the
139      *             backend.
140      */
141     private void createStorage() throws TTException {
142 
143         LOGGER.debug("Creating storage with " + mDataNumbers + " nodes containing " + BLOCKS_IN_DATA
144             + " blocks with " + IStorageModule.VIRTUAL_BLOCK_SIZE + " bytes each.");
145 
146         // Creating mirror
147         jCloudsStorageModule =
148             new JCloudsStorageModule(BLOCKS_IN_DATA * VIRTUAL_BLOCK_SIZE, Files.createTempDir());
149 
150         IData data = this.mRtx.getCurrentData();
151 
152         if (data != null) {
153             return;
154         }
155 
156         for (int i = 0; i < mDataNumbers; i++) {
157 
158             // Bootstrapping nodes containing clusterSize -many blocks/sectors.
159             LOGGER.debug("Bootstraping node " + i + "\tof " + (mDataNumbers - 1));
160             this.mRtx.bootstrap(new byte[HybridTreetankStorageModule.BYTES_IN_DATA]);
161         }
162 
163         this.mRtx.commit();
164     }

166     /**
167      * {@inheritDoc}
168      */
169     public int checkBounds(long logicalBlockAddress, int transferLengthInBlocks) {
170         // Checking if the logical block address is out of bounds
171         if (logicalBlockAddress < 0 || logicalBlockAddress >= getSizeInBlocks()) {
172             return 1;
173         } else
174         // if the logical block address is in bounds but the transferlength
175         // either exceeds
176         // the device size or is faulty return 2
177         if (transferLengthInBlocks < 0 || logicalBlockAddress + transferLengthInBlocks > getSizeInBlocks()) {
178             return 2;
179         } else {
180             return 0;
181         }
182     }

203     /**
204      * {@inheritDoc}
205      */
206     public void write(byte[] bytes, long storageIndex) throws IOException {
207 
208         LOGGER.debug("Starting to write with param: " + "\nstorageIndex = " + storageIndex
209             + "\nbytes.length = " + bytes.length);
210 
211         jCloudsStorageModule.write(bytes, storageIndex);
212 
213         // Submitting into treetank
214         mWriteTaskExecutor.submit(new WriteTask(bytes, storageIndex));
215 
216     }

218     /**
219      * WriteTask for writing data to treetank.
220      * 
221      * @author Andreas Rain
222      * 
223      */
224     private class WriteTask implements Callable<Void> {
225 
226         private final byte[] mBytes;
227         private final long mStorageIndex;
228 
229         public WriteTask(byte[] pBytes, long pStorageIndex) {
230             mBytes = pBytes;
231             mStorageIndex = pStorageIndex;
232         }
233 
234         @Override
235         public Void call() throws Exception {
236             long startIndex = mStorageIndex / BYTES_IN_DATA;
237             int startIndexOffset = (int)(mStorageIndex % BYTES_IN_DATA);
238 
239             long endIndex = (mStorageIndex + mBytes.length) / BYTES_IN_DATA;
240             int endIndexMax = (int)((mStorageIndex + mBytes.length) % BYTES_IN_DATA);
241 
242             int bytesWritten =
243                 mBytes.length + startIndexOffset > BYTES_IN_DATA ? BYTES_IN_DATA - startIndexOffset
244                     : mBytes.length;
245 
246             try {
247                 checkState(mRtx.moveTo(startIndex));
248                 byte[] data = mRtx.getValueOfCurrentData();
249                 System.arraycopy(mBytes, 0, data, startIndexOffset, bytesWritten);
250                 mRtx.setValue(data);
251 
252                 for (long i = startIndex + 1; i < endIndex; i++) {
253                     checkState(mRtx.moveTo(i));
254                     data = mRtx.getValueOfCurrentData();
255                     System.arraycopy(mBytes, bytesWritten, data, 0, data.length);
256                     mRtx.setValue(data);
257                     bytesWritten = bytesWritten + data.length;
258 
259                 }
260 
261                 if (startIndex != endIndex && endIndex < mDataNumbers) {
262                     checkState(mRtx.moveTo(endIndex));
263                     data = mRtx.getValueOfCurrentData();
264                     System.arraycopy(mBytes, bytesWritten, data, 0, endIndexMax);
265                     mRtx.setValue(data);
266 
267                     bytesWritten += endIndexMax;
268                 }
269 
270                 // Bytes written is the actual number of bytes that have been written.
271                 // The two lengths have to match, otherwise not enough bytes have been written (or too much?).
272                 checkState(bytesWritten == mBytes.length);
273 
274                 // Incrementing bytewriter counter
275                 mByteCounter += bytesWritten;
276 
277                 // If 1024 nodes have been fully written.
278                 if (mByteCounter >= COMMIT_THRESHOLD) {
279                     mRtx.commit();
280 
281                     mByteCounter = 0;
282                 }
283 
284             } catch (TTException exc) {
285                 throw new IOException(exc);
286             }
287             return null;
288         }
289 
290     }

292     private class WriteTaskThreadFactory implements ThreadFactory {
293         public Thread newThread(Runnable r) {
294             Thread t = new Thread(r);
295             t.setPriority(Thread.MIN_PRIORITY);
296             return t;
297         }
298     }

310         }
