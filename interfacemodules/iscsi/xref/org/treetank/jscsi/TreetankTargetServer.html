25  package org.treetank.jscsi;
26  
27  import java.io.File;
28  import java.io.IOException;
29  
30  import org.jscsi.target.TargetServer;
31  import org.treetank.access.conf.StorageConfiguration;
32  
33  import com.google.common.io.Files;
34  
35  public class TreetankTargetServer {
36    
37    /**
38     * Starts the jSCSI target.
39     * 
40     * Argument one has to be an empty (or already created) storage folder!
41     * Argument to can be additionally added if you don't want the default target configuration.
42     * It has to be conform with the default target configuration schema file, which you can find
43     * in the resources of the target.
44     * @param args
45     *          Argument 1 = the storage path
46     *          Argument 2 = path to a customized target configuration xml file
47     * @throws IOException
48     */
49    public static void main(String[] args) throws Exception {
50  
51      TargetServer target;
52      File file;
53      StorageConfiguration configuration;
54      switch (args.length) {
55        case 0:
56          file = new File(new StringBuilder(Files.createTempDir()
57              .getAbsolutePath()).append(File.separator).append("tnk")
58              .append(File.separator).append("path1").toString());
59          configuration = new StorageConfiguration(file);
60          target = new TargetServer(TreetankConfiguration.create(
61              TreetankConfiguration.CONFIGURATION_SCHEMA_FILE,
62              TreetankConfiguration.CONFIGURATION_CONFIG_FILE, file,
63              configuration, 512));
64          break;
65        case 1:
66          file = new File(args[0]);
67          configuration = new StorageConfiguration(file);
68          target = new TargetServer(TreetankConfiguration.create(
69              TreetankConfiguration.CONFIGURATION_SCHEMA_FILE,
70              TreetankConfiguration.CONFIGURATION_CONFIG_FILE, file,
71              configuration, 512));
72  
73          break;
74        case 2:
75          file = new File(args[0]);
76          configuration = new StorageConfiguration(file);
77          target = new TargetServer(TreetankConfiguration.create(
78              TreetankConfiguration.CONFIGURATION_SCHEMA_FILE,
79              new File(args[1]), file,
80              configuration, 512));
81          break;
82        default:
83          throw new IllegalArgumentException(
84              "Only zero or one Parameter (Path to Configuration-File) allowed!");
85      }
86      
87      target.call();
88    }
89  }
