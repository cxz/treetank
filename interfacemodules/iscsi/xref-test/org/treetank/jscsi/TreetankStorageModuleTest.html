81      @Test(groups = "Initial read write")
82      public void testReadAndWrite() throws TTException, IOException {
83  
84          final byte[] writeArray = new byte[TEST_FILE_SIZE];
85          for (int i = 0; i < TEST_FILE_SIZE; ++i)
86              writeArray[i] = (byte)(CoreTestHelper.random.nextDouble() * 256);
87          final byte[] readArray = new byte[TEST_FILE_SIZE];
88  
89          System.arraycopy(writeArray, 0, readArray, 0, TEST_FILE_SIZE);
90  
91          // write
92          storageModule.write(writeArray,// bytes (source)
93              0,// bytesOffset
94              TEST_FILE_SIZE - 50,// length
95              0);
96  
97          // read
98          storageModule.read(readArray,// bytes (destination)
99              0,// bytesOffset
100             TEST_FILE_SIZE - 50,// length
101             0);
102 
103         // check for errors
104         assertTrue(Arrays.equals(writeArray, readArray));
105 
106         byte[][] splitBytes = new byte[16][8192];
107 
108         ByteArrayInputStream writeArrayInputStream = new ByteArrayInputStream(writeArray);
109 
110         for (int i = 0; i < splitBytes.length; i++) {
111             splitBytes[i] = new byte[8192];
112             storageModule.read(splitBytes[i],// bytes (destination)
113                 0,// bytesOffset
114                 8192,// length
115                 i * 16 * 512);
116             byte[] b = new byte[8192];
117             System.out.println(i);
118             writeArrayInputStream.read(b, 0, 8192);
119 
120             assertTrue(Arrays.equals(splitBytes[i], b));
121         }
122 
123     }
